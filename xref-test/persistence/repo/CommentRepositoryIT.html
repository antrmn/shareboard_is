32      @BeforeAll
33      public void populate() throws Exception {
34          doTransactional(() -> {
35              users = IntStream.range(1, 4).mapToObj(n -> {
36                  User user = new User();
37                  user.setUsername("user" + n);
38                  user.setEmail("user" + n + "@email.it");
39                  user.setDescription("Description for user " + n);
40                  user.setAdmin(n % 2 == 0);
41                  user.setPicture("picture" + n);
42                  user.setPassword((n + "password").getBytes(StandardCharsets.UTF_8));
43                  user.setSalt((n + "salt").getBytes(StandardCharsets.UTF_8));
44                  return user;
45              }).map(genericRepository::insert).collect(Collectors.toList());
46  
47              sections = IntStream.range(1,4).mapToObj(n -> {
48                  Section section = new Section();
49                  section.setName("section"+n);
50                  section.setPicture("picture" + n);
51                  section.setBanner("banner"+n);
52                  return section;
53              }).map(genericRepository::insert).collect(Collectors.toList());
54  
55              posts = IntStream.range(1,10).mapToObj(n -> {
56                  Post post = new Post();
57                  post.setContent("content"+n);
58                  post.setTitle("title"+n);
59                  post.setType(n % 2 == 0 ? Post.Type.TEXT : Post.Type.IMG);
60                  post.setAuthor( users.get( n % users.size() ) );
61                  post.setSection( sections.get(n % sections.size()) );
62                  return post;
63              }).map(genericRepository::insert).collect(Collectors.toList());
64  
65              IntStream.range(1,30).forEach(n -> {
66                  Comment comment = new Comment();
67                  comment.setContent("content" + n);
68                  comment.setPost( posts.get(n % posts.size() ));
69                  List<Comment> postComments = comments.computeIfAbsent(comment.getPost().getId(), ArrayList::new);
70  
71                  if(postComments.size()<=3){
72                      comment.setParentComment(null);
73                  } else {
74                      comment.setParentComment(postComments.get(n % postComments.size()));
75                  }
76  
77                  postComments.add(comment);
78                  genericRepository.insert(comment);
79              });
80          });
81      }

83      @Test
84      public void testRetrieveByPost() throws Exception {
85          Post post = posts.get(0);
86          List<Comment> commentByPosts = commentRepository.getByPost(post, 999);
87          commentByPosts.forEach((comment) -> {
88              Comment foundComment = comments.get(post.getId()).stream()
89                      .filter(x -> x.getId().equals(comment.getId()))
90                      .findFirst().orElse(null);
91  
92              Assertions.assertNotNull(foundComment);
93              Assertions.assertEquals(comment.getId(),foundComment.getId());
94              Assertions.assertEquals(comment.getPost().getId(),foundComment.getPost().getId());
95              Assertions.assertTrue(comment.getParentComment() == null && foundComment.getParentComment() == null
96                      || comment.getParentComment().getId().equals(foundComment.getParentComment().getId()));
97          });
98      }
