49      @Test
50      void interceptorsOrderTest() throws ServletException, IOException {
51          {
52              List<Integer> sequence = new ArrayList<>();
53              HttpServletRequest req = Mockito.mock(HttpServletRequest.class);
54              when(req.getAttribute(any())).thenReturn(sequence);
55              InterceptableServlet servlet = new SampleInterceptableServlet();
56              when(req.getMethod()).thenReturn("GET");
57              HttpServletResponse resp = Mockito.mock(HttpServletResponse.class);
58              servlet.service(req,resp);
59              assertEquals(sequence,List.of(3,2,1,4));
60          }
61          {
62              List<Integer> sequence = new ArrayList<>();
63              HttpServletRequest req = Mockito.mock(HttpServletRequest.class);
64              when(req.getAttribute(any())).thenReturn(sequence);
65              InterceptableServlet servlet = new SampleInterceptableServlet();
66              when(req.getMethod()).thenReturn("POST");
67              HttpServletResponse resp = Mockito.mock(HttpServletResponse.class);
68              servlet.service((ServletRequest) req,(ServletResponse) resp);
69              assertEquals(sequence,List.of(2,5,1));
70          }
71          {
72              List<Integer> sequence = new ArrayList<>();
73              HttpServletRequest req = Mockito.mock(HttpServletRequest.class);
74              when(req.getAttribute(any())).thenReturn(sequence);
75              InterceptableServlet servlet = new SampleInterceptableServlet();
76              when(req.getMethod()).thenReturn("PUT");
77              HttpServletResponse resp = Mockito.mock(HttpServletResponse.class);
78              assertThrows(NullPointerException.class, () -> servlet.service(req,resp));
79              assertEquals(sequence,List.of());
80          }

103     @Target({ElementType.METHOD,ElementType.TYPE})
104     @Retention(RetentionPolicy.RUNTIME)
105     private @interface LowPriority{
106         int value();
107     }
108 
109     private static class LowPriorityInterceptor extends ServletInterceptor<LowPriority>{
110         public int value;
111         public LowPriorityInterceptor(){
112 
113         }
114 
115         @Override
116         protected void init(LowPriority annotation) {
117             value = annotation.value();
118         }
119 
120         @Override
121         public void handle(HttpServletRequest req, HttpServletResponse resp, HttpServletBiConsumer next) throws ServletException, IOException {
122             List<Integer> sequence = (List<Integer>) req.getAttribute("test");
123             sequence.add(this.value);
124             next.handle(req,resp);
125         }
126 
127         @Override
128         public int priority() {
129             return Integer.MAX_VALUE;
130         }
131     }
