64      private void startDb() throws ManagedProcessException {
65          DBConfigurationBuilder config = DBConfigurationBuilder.newBuilder();
66          config.setPort(0); // random port
67          db = DB.newEmbeddedDB(config.build());
68          db.start();
69          db.createDB("testdb", "root", "root");
70          db.source("schemaonly.sql", "root", "root", "testdb");
71      }

73      //resources.xml equivalent
74      @Configuration
75      public Properties config() throws ManagedProcessException {
76          startDb();
77          Properties p = new Properties();
78          p.put("db", "new://Resource?type=DataSource");
79          p.put("db.JdbcDriver", "com.mysql.cj.jdbc.Driver");
80          p.put("db.JdbcUrl", "jdbc:mysql://localhost:"+db.getConfiguration().getPort()+"/testdb?user=root&password=root");
81          return p;

85      @Module
86      public PersistenceUnit setupPU(){
87          PersistenceUnit pu = new PersistenceUnit("testPU");
88          pu.setProvider(HibernatePersistenceProvider.class);
89          pu.setJtaDataSource("db");
90          pu.setProperty("tomee.jpa.factory.lazy", "true");
91          pu.setProperty("hibernate.format_sql" ,"true");
92          pu.setProperty("hibernate.use_sql_comments" ,"true");
93          pu.setProperty("hibernate.dialect" ,"org.hibernate.dialect.MySQL8Dialect");
94          pu.setProperty("hibernate.physical_naming_strategy",
95                         "org.hibernate.boot.model.naming.CamelCaseToUnderscoresNamingStrategy");
96          return pu;

135     protected void doTransactional(Worker worker) throws Exception{
136         userTransaction.begin();
137         try {
138             worker.run();
139         } catch (RuntimeException e){
140             userTransaction.setRollbackOnly();
141             throw e;
142         } finally {
143             if (userTransaction.getStatus() == Status.STATUS_MARKED_ROLLBACK)
144                 userTransaction.rollback();
145             else
146                 userTransaction.commit();
147         }
148     }
