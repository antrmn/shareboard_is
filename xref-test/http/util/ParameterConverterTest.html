34      private <R> R convert(String value, BiFunction<ParameterConverter,String,R> converterMethod){
35          HttpServletRequest request = mock(HttpServletRequest.class);
36          when(request.getParameter("iWillReturnValue")).thenReturn(value);
37  
38          ParameterConverter parameterConverter = new ParameterConverter(request);
39          return converterMethod.apply(parameterConverter, "iWillReturnValue");
40      }

42      @ParameterizedTest
43      @ValueSource(ints={0,1,-5,Integer.MAX_VALUE, Integer.MIN_VALUE})
44      void testIntParameter(int n){
45          String str = String.valueOf(n);
46          OptionalInt optionalInt = convert(str, ParameterConverter::getIntParameter);
47          assertFalse(optionalInt.isEmpty());
48          assertEquals(n, optionalInt.getAsInt());
49      }

51      @Test
52      void testIntParameterOverflow(){
53          long max = Integer.MAX_VALUE;
54          max++;
55          OptionalInt optionalInt = convert(String.valueOf(max), ParameterConverter::getIntParameter);
56          assertTrue(optionalInt.isEmpty());
57  
58          long min = Integer.MIN_VALUE;
59          min--;
60          OptionalInt optionalIntMin = convert(String.valueOf(min), ParameterConverter::getIntParameter);
61          assertTrue(optionalIntMin.isEmpty());
