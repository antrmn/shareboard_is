25  class ImageServiceTest extends ServiceTest {
26      @Mock
27      MediaRepository mediaRepository;
28      @Inject ImageService imageService;
29  
30      @Test
31      void getImage() throws IOException {
32          byte[] bytes = "GIF8".getBytes(StandardCharsets.UTF_8);
33          InputStream stream = new BufferedInputStream(new ByteArrayInputStream(bytes));
34          when(mediaRepository.get(any())).thenReturn(stream);
35          byte[] result = new byte[bytes.length];
36          imageService.getImage("file").read(result);
37          assertArrayEquals(bytes,result);
38      }
39  
40      @Test
41      void imageNotFound() throws IOException {
42          when(mediaRepository.get(any())).thenReturn(null);
43          assertNull(imageService.getImage("file"));
44      }
45  
46      @Test
47      void audioInsteadOfImage() throws IOException{

53      }
