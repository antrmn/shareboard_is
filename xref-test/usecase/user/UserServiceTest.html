142     @Test
143     public void testShowUsers() {
144         List<User> users = IntStream.range(1, 10).mapToObj(n -> {
145             User user = new User();
146             user.setUsername("usecase/user" + n);
147             user.setId(n);
148             user.setEmail("email" + n + "@email.it");
149             user.setAdmin(false);
150             return user;
151         }).collect(Collectors.toList());
152 
153         List<UserProfile> usersDto =
154                 users.stream().map(user -> UserProfile.builder()
155                         .id(user.getId())
156                         .email(user.getEmail())
157                         .creationDate(user.getCreationDate())
158                         .description(user.getDescription())
159                         .picture(user.getPicture())
160                         .username(user.getUsername())
161                         .isAdmin(user.getAdmin()).build())
162                 .collect(Collectors.toList());
163 
164         when(genericRepository.findAll(User.class)).thenReturn(users);
165         assertEquals(service.showUsers(), usersDto);
166     }
