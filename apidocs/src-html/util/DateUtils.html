001package util;
002
003import java.time.Instant;
004import java.time.LocalDate;
005import java.time.ZoneId;
006import java.time.temporal.ChronoUnit;
007
008public final class DateUtils {
009    public static String printTimeSince(Instant then){
010        long n;
011        if ((n = Math.abs(Instant.now().until(then, ChronoUnit.SECONDS))) < 60){
012            return n == 1 ? n + " secondo" : n + " secondi";
013        } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.MINUTES))) < 60){
014            return n == 1 ? n + " minuto" : n + " minuti";
015        } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.HOURS))) < 24){
016            return n == 1 ? n + " ora" : n + " ore";
017        } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.DAYS))) < 30) {
018            return n == 1 ? n + " giorno" : n + " giorni";
019        } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.DAYS))) < 365) {
020            n /= 30;
021            return n == 1 ? n + " mese" : n + " mesi";
022        } else {
023            int thenYear = then.atZone(ZoneId.systemDefault()).toLocalDate().getYear();
024            n = Math.abs(thenYear - LocalDate.now().getYear());
025            return n == 1 ? n + " anno" : n + " anni";
026        }
027    }
028}
