001package util;
002
003import java.time.Instant;
004import java.time.temporal.ChronoUnit;
005
006public final class DateUtils {
007    public static String printTimeSince(Instant then){
008        long n;
009        if ((n = Math.abs(Instant.now().until(then, ChronoUnit.SECONDS))) < 60){
010            return n == 1 ? n + " secondo" : n + " secondi";
011        } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.MINUTES))) < 60){
012            return n == 1 ? n + " minuto" : n + " minuti";
013        } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.HOURS))) < 24){
014            return n == 1 ? n + " ora" : n + " ore";
015        } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.DAYS))) < 30) {
016            return n == 1 ? n + " giorno" : n + " giorni";
017        } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.DAYS))) < 365) {
018            n /= 30;
019            return n == 1 ? n + " mese" : n + " mesi";
020        } else {
021            n = Math.abs(Instant.now().until(then, ChronoUnit.YEARS));
022            return n == 1 ? n + " anno" : n + " anni";
023        }
024    }
025}
