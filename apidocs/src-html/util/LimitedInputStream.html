001package util;
002
003import java.io.FilterInputStream;
004import java.io.IOException;
005import java.io.InputStream;
006
007public class LimitedInputStream extends FilterInputStream {
008    private final long byteLimit;
009    private long byteRead = 0;
010
011
012    public LimitedInputStream(InputStream in, long byteLimit){
013        super(in);
014        this.byteLimit = byteLimit;
015    }
016
017    @Override
018    public int read() throws IOException {
019        int bite = super.read();
020        this.byteRead++;
021        if(this.byteRead > byteLimit)
022            throw new ReadLimitExceededException();
023        return bite;
024    }
025
026    @Override
027    public int read(byte[] b) throws IOException {
028        return super.read(b);
029    }
030
031    @Override
032    public int read(byte[] b, int off, int len) throws IOException {
033        int byteRead = super.read(b,off,len);
034        this.byteRead += byteRead;
035        if(this.byteRead > byteLimit)
036            throw new ReadLimitExceededException();
037        return byteRead;
038    }
039}
