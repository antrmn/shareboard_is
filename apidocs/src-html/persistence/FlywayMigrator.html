001package persistence;
002
003import org.flywaydb.core.Flyway;
004import org.flywaydb.core.api.configuration.FluentConfiguration;
005
006import javax.annotation.PostConstruct;
007import javax.annotation.Resource;
008import javax.ejb.*;
009import javax.sql.DataSource;
010
011@Singleton
012@Startup
013@TransactionManagement(TransactionManagementType.BEAN)
014public class FlywayMigrator {
015
016    @Resource(name = "jdbc/shareboardDB")
017    private DataSource dataSource;
018
019    @PostConstruct
020    private void onStartup(){
021        if (dataSource == null) {
022            throw new EJBException("Cannot migrate, no datasource configured!");
023        }
024        FluentConfiguration config = Flyway.configure()
025                .dataSource(dataSource)
026                .createSchemas(true)
027                .validateOnMigrate(false)
028                .locations("classpath:db/migration");
029        Flyway flyway = new Flyway(config);
030        flyway.migrate();
031    }
032
033
034
035}
