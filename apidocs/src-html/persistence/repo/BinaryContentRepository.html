001package persistence.repo;
002
003
004import java.io.*;
005import java.nio.file.Files;
006import java.nio.file.Path;
007import java.util.UUID;
008
009public class BinaryContentRepository implements Serializable{
010
011    private Path uploadRoot = Path.of(System.getProperty("openejb.home"), "uploads");
012
013    public BinaryContentRepository(){}
014
015    public BinaryContentRepository(Path uploadRoot){
016        this.uploadRoot = uploadRoot;
017    }
018
019    public String insert(InputStream stream, String filename) throws IOException {
020        File file = new File(uploadRoot.toFile(), filename);
021        Files.copy(stream, file.toPath());
022        return filename;
023    }
024
025    public String insert(InputStream stream) throws IOException {
026        return insert(stream, UUID.randomUUID().toString());
027    }
028
029    public void remove(String filename) throws IOException {
030        boolean successful = Files.deleteIfExists(uploadRoot.resolve(filename));
031        // if (!successful) { ???? }
032    }
033
034    public InputStream get(String filename){
035        File file = new File(uploadRoot.toFile(), filename);
036        if (!file.exists() || !file.isFile())
037            return null;
038        try {
039            return new FileInputStream(file);
040        } catch (FileNotFoundException e) {
041            return null;
042        }
043    }
044
045}
