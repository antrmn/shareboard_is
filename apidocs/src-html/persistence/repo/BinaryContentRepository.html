001package persistence.repo;
002
003
004import java.io.*;
005import java.nio.file.Files;
006import java.nio.file.Path;
007import java.util.UUID;
008
009public class BinaryContentRepository implements Serializable{
010
011    private Path uploadRoot = Path.of(System.getProperty("openejb.home"), "uploads");
012
013    /**
014     *  Costruttore vuoto
015     * @return nuova istanza di BinaryContentRepository
016     */
017    public BinaryContentRepository(){}
018
019    /**
020     *  Costruttore con parametro per settare uploadRoot
021     * @param uploadRoot oggetto Path per settare uploadRoot
022     * @return nuova istanza di BinaryContentRepository
023     */
024    public BinaryContentRepository(Path uploadRoot){
025        this.uploadRoot = uploadRoot;
026    }
027
028    /**
029     *  Salva un file nel filesystem e ne restituisce il nome
030     * @param stream stream di dati
031     * @param filename nome del file da salvare
032     * @throws IOException
033     * @return nuova istanza di BinaryContentRepository
034     */
035    public String insert(InputStream stream, String filename) throws IOException {
036        File file = new File(uploadRoot.toFile(), filename);
037        Files.copy(stream, file.toPath());
038        return filename;
039    }
040
041    /**
042     *  Salva un file nel filesystem e ne restituisce un nome univoco generato in maniera casuale
043     * @param stream stream di dati
044     * @throws IOException
045     * @return nuova istanza di BinaryContentRepository
046     */
047    public String insert(InputStream stream) throws IOException {
048        return insert(stream, UUID.randomUUID().toString());
049    }
050
051    /**
052     *  Rimuove un file dal filesystem dato un nome se esiste
053     * @param filename nome del file
054     * @throws IOException
055     */
056    public void remove(String filename) throws IOException {
057        boolean successful = Files.deleteIfExists(uploadRoot.resolve(filename));
058        // if (!successful) { ???? }
059    }
060
061    /**
062     *  Restituisce un file dal filesystem dato un nome se esiste altrimenti restituisce null
063     * @param filename nome del file
064     * @return stream di dati del file o null
065     */
066    public InputStream get(String filename){
067        File file = new File(uploadRoot.toFile(), filename);
068        if (!file.exists() || !file.isFile())
069            return null;
070        try {
071            return new FileInputStream(file);
072        } catch (FileNotFoundException e) {
073            return null;
074        }
075    }
076
077}
