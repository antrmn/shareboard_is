001package persistence.repo;
002
003import javax.persistence.EntityManager;
004import javax.persistence.LockModeType;
005import javax.persistence.PersistenceContext;
006import javax.persistence.criteria.CriteriaQuery;
007import java.io.Serializable;
008import java.util.List;
009
010/**
011 * Interfaccia usata per interagire con {@link javax.persistence.EntityManager}
012 * Istanze che implementano questa interfaccia fungono da facciata alle funzionalità di JPA, fornendo i metodi
013 * essenziali (CRUD) per una specifica entità di persistenza
014 * @param <T> La classe delle entità di persistenza gestite da classi che implementano questa interfaccia
015 * @param <ID> La classe della chiave primaria delle entità di persistenza gestite da classi che implementano questa
016 *            interfaccia
017 * @see javax.persistence.EntityManager
018 */
019public class GenericRepository implements Serializable {
020
021    @PersistenceContext
022    protected EntityManager em;
023
024    /**
025     *  Trova per chiave primaria
026     * @param id La chiave primaria dell'oggetto da cercare
027     * @return L'entità trovata o null se l'entità non esiste
028     * @throws IllegalArgumentException se la chiave primaria fornita è null o non valida per l'entità da cercare
029     * @see javax.persistence.EntityManager#find(Class, Object)
030     */
031    public <T> T findById(Class<T> entityClass, Object primaryKey, boolean loadLazily) {
032        return loadLazily ? em.getReference(entityClass, primaryKey) :
033                            em.find(entityClass, primaryKey, LockModeType.NONE);
034    }
035
036    /**
037     * Trova per chiave primaria, restituendo un'entità caricata pigramente (lazy-loaded). L'entità restituita viene
038     * caricata al primo accesso effettuato. Se l'istanza non esiste, viene lanciato {@link javax.persistence.EntityNotFoundException}
039     * quando l'istanza viene acceduta per la prima volta.
040     * @param id La chiave primaria dell'oggetto da cercare
041     * @return L'entità caricata pigramente
042     * @throws IllegalArgumentException se la chiave primaria fornita è null o non valida per l'entità da cercare
043     * @throws javax.persistence.EntityNotFoundException se l'entità non può essere acceduta
044     * @see javax.persistence.EntityManager#getReference(Class, Object)
045     */
046    public <T> T findById(Class<T> entityClass, Object primaryKey){
047        return findById(entityClass,primaryKey,false);
048    }
049
050    /**
051     * Restituisce una lista tipata contenente tutte le istanze di una determinata entità di persistenza
052     * @return La lista dei risultati
053     * @throws javax.persistence.QueryTimeoutException In caso di timeout raggiunto
054     * @throws javax.persistence.PersistenceException
055     */
056    public <T> List<T> findAll(Class<T> entityClass) {
057        CriteriaQuery<T> c = em.getCriteriaBuilder().createQuery(entityClass);
058        c.select(c.from(entityClass));
059        return em.createQuery(c).getResultList();
060    }
061
062    /**
063     * Restituisce il numero indicante la quantità di istanze di una determinata entità di persistenza
064     * @return La quantità di istanze
065     * @throws javax.persistence.QueryTimeoutException In caso di timeout raggiunto
066     * @throws javax.persistence.PersistenceException
067     */
068    public <T> Long getCount(Class<T> entityClass) {
069        CriteriaQuery<Long> c =
070                em.getCriteriaBuilder().createQuery(Long.class);
071        c.select(em.getCriteriaBuilder().count(c.from(entityClass)));
072        return em.createQuery(c).getSingleResult();
073    }
074
075    /**
076     * Crea un'istanza persistente e gestita (managed entity)
077     * @param entity L'istanza di entità
078     * @return L'istanza passata come parametro
079     * @throws javax.persistence.EntityExistsException se l'entità esiste già
080     * @see javax.persistence.EntityManager#persist(Object)
081     */
082    public <T> T insert(T entity) {
083        em.persist(entity);
084        em.flush();
085        return entity;
086    }
087
088    /**
089     * Unisce lo stato di una data entità allo stato nel contesto di persistenza
090     * @param entity
091     * @return Una copia <i>managed</i> dell'entità passata come parametro
092     * @see javax.persistence.EntityManager#merge(Object)
093     */
094    public <T> T merge(T entity) {
095        return em.merge(entity);
096    }
097
098    /**
099     * Rimuove l'istanza dal contesto di persistenza
100     * @param entity L'entità managed da rimuovere
101     * @throws IllegalArgumentException se l'entità passata non è managed
102     */
103    public <T> void remove(T entity) {
104        em.remove(entity);
105    }
106}
