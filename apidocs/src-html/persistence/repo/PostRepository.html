001package persistence.repo;
002
003import persistence.model.Post;
004import persistence.model.Section;
005import persistence.model.User;
006
007import javax.persistence.EntityManager;
008import javax.persistence.PersistenceContext;
009import javax.persistence.TypedQuery;
010import javax.persistence.criteria.*;
011import java.io.Serializable;
012import java.time.Instant;
013import java.util.ArrayList;
014import java.util.Collections;
015import java.util.List;
016
017public class PostRepository implements Serializable {
018
019    @PersistenceContext
020    protected EntityManager em;
021
022    public PostFinder getFinder(){
023        return new PostFinder();
024    }
025
026    private enum SortCriteria {OLDEST, NEWEST, MOSTVOTED};
027    public class PostFinder {
028        private User author;
029        private List<Section> sections;
030        private Instant before;
031        private Instant after;
032        private int offset = 0;
033        private int pageSize = 30;
034        private String content;
035        private boolean includeBody = false;
036        private User joinUserFollows;
037        private SortCriteria sortCriteria = PostRepository.SortCriteria.NEWEST;
038
039        protected PostFinder(){ }
040
041        public PostFinder byAuthor(User author){
042            this.author = author;
043            return this;
044        }
045
046        public PostFinder byContent(String text){
047            this.content = text;
048            return this;
049        }
050
051        public PostFinder bySections(List<Section> sections){
052            this.sections = sections;
053            return this;
054        }
055
056        public PostFinder bySection(Section section){
057            this.sections = Collections.singletonList(section);
058            return this;
059        }
060
061        public PostFinder postedAfter(Instant after){
062            this.after = after;
063            return this;
064        }
065
066        public PostFinder postedBefore(Instant before){
067            this.before = before;
068            return this;
069        }
070
071        public PostFinder offset(int n){
072            offset = n;
073            return this;
074        }
075
076        public PostFinder limit(int n){
077            pageSize = n;
078            return this;
079        }
080
081        public PostFinder getOldest(){
082            sortCriteria = SortCriteria.OLDEST;
083            return this;
084        }
085
086        public PostFinder getNewest(){
087            sortCriteria = SortCriteria.NEWEST;
088            return this;
089        }
090
091        public PostFinder getMostVoted(){
092            sortCriteria = SortCriteria.MOSTVOTED;
093            return this;
094        }
095
096        public PostFinder includeBody(){
097            includeBody = true;
098            return this;
099        }
100
101        public PostFinder joinUserFollows(User user){
102            joinUserFollows = user;
103            return this;
104        }
105
106        public List<Post> getResults(){
107            CriteriaBuilder cb = em.getCriteriaBuilder();
108            CriteriaQuery<Post> cq = cb.createQuery(Post.class);
109            Root<Post> root = cq.from(Post.class);
110
111            List<Predicate> predicates = new ArrayList<>();
112
113            if(author != null) {
114                predicates.add(cb.equal(root.get("author"), author));
115            }
116
117            if(sections != null && !sections.isEmpty()) {
118                predicates.add(root.get("section").in(sections));
119            }
120
121            if(after != null) {
122                predicates.add(cb.greaterThanOrEqualTo(root.get("creationDate"), after));
123            }
124
125            if(before != null) {
126                predicates.add(cb.lessThanOrEqualTo(root.get("creationDate"), before));
127            }
128
129            if(content != null) {
130                if(includeBody) {
131                    predicates.add(
132                            cb.or(
133                                    cb.like(root.get("title"), '%' + content + '%'),
134                                    cb.and(
135                                            cb.notEqual(root.get("type"), Post.Type.IMG),
136                                            cb.like(root.get("content"), '%' + content + '%')
137                                    )));
138                } else {
139                    predicates.add(cb.like(root.get("title"), '%' + content + '%'));
140                }
141            }
142
143            if(joinUserFollows != null){
144                Join<Object, Object> joinFollow = root.join("section").join("follows");
145                predicates.add(cb.equal(joinFollow.get("user"), joinUserFollows));
146            }
147
148            cq.where(cb.and(predicates.toArray(Predicate[]::new)));
149
150            switch (sortCriteria) {
151                case MOSTVOTED:
152                    cq.orderBy(cb.desc(root.get("votesCount")));
153                    break;
154                case OLDEST:
155                    cq.orderBy(cb.asc(root.get("creationDate")));
156                    break;
157                case NEWEST:
158                default:
159                    cq.orderBy(cb.desc(root.get("creationDate")));
160                    break;
161            }
162
163
164            TypedQuery<Post> tq = em.createQuery(cq);
165            if(offset >= 0)
166                tq.setFirstResult(offset);
167            if(pageSize>=0)
168                tq.setMaxResults(pageSize);
169
170            return tq.getResultList();
171        }
172    }
173}
