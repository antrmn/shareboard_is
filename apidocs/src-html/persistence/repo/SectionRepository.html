001package persistence.repo;
002
003import org.hibernate.Session;
004import org.hibernate.SimpleNaturalIdLoadAccess;
005import persistence.model.Section;
006
007import javax.persistence.EntityManager;
008import javax.persistence.PersistenceContext;
009import java.io.Serializable;
010import java.time.Instant;
011import java.util.List;
012
013
014public class SectionRepository implements Serializable {
015
016    @PersistenceContext
017    protected EntityManager em;
018
019    /**
020     * Restituisce i dati di una sezione dato un nome
021     * @param name nome della sezione
022     * @return entità Section
023     */
024    public Section getByName(String name){
025        return getByName(name, false);
026    }
027
028    /**
029     * Restituisce i dati di una sezione dato un nome
030     * @param name nome della sezione
031     * @param getReference booleano per attivare lazy loading
032     * @return entità Section
033     */
034    public Section getByName(String name, boolean getReference){
035        SimpleNaturalIdLoadAccess<Section> section = em.unwrap(Session.class).bySimpleNaturalId(Section.class);
036        return getReference ? section.getReference(name) : section.load(name);
037    }
038
039    /**
040     * Restituisce una lista con i dati delle sezioni con più follows
041     * @return lista di entità Section delle sezioni con più follows
042     */
043    public List<Section> getMostFollowedSections(){
044        return em.createQuery("from Section s order by follows.size desc", Section.class).getResultList();
045    }
046
047    /**
048     * Restituisce una lista con i dati delle sezioni con più follows in un dato arco di tempo
049     * @param after data dopo la quale contare i follows delle sezioni
050     * @return lista di entità Section delle sezioni con più follows
051     */
052    public List<Section> getMostFollowedSections(Instant after){
053        return em.createQuery(
054                "select s from Section s left join s.follows f where f.followDate >= :after " +
055                "group by s " +
056                "order by count(f) desc", Section.class)
057                .setParameter("after",after).getResultList();
058    }
059}
