001package persistence.repo;
002
003import org.hibernate.Session;
004import org.hibernate.SimpleNaturalIdLoadAccess;
005import persistence.model.User;
006
007import javax.persistence.EntityManager;
008import javax.persistence.PersistenceContext;
009import java.io.Serializable;
010
011/**
012 * Interfaccia usata per interagire con {@link javax.persistence.EntityManager}
013 * Istanze che implementano questa interfaccia fungono da facciata alle funzionalità di JPA, fornendo i metodi
014 * essenziali (CRUD) per una specifica entità di persistenza
015 * @see javax.persistence.EntityManager
016 */
017public class UserRepository implements Serializable {
018
019    @PersistenceContext
020    protected EntityManager em;
021
022    /**
023     * Restituisce i dati di un utente dato un nome
024     * @param name nome dell'utente
025     * @return entità User
026     */
027    public User getByName(String name) {
028        return getByName(name, false);
029    }
030
031    /**
032     * Restituisce i dati di un utente dato un nome
033     * @param name nome dell'utente
034     * @param getReference booleano per attivare lazy loading
035     * @return entità User
036     */
037    public User getByName(String name, boolean getReference) {
038        SimpleNaturalIdLoadAccess<User> user = em.unwrap(Session.class).bySimpleNaturalId(User.class);
039        return getReference ? user.getReference(name) : user.load(name);
040    }
041
042    /**
043     * Restituisce i dati di un utente data un email
044     * @param email email dell'utente
045     * @return entità User
046     */
047    public User getByEmail(String email) {
048        return em.createQuery("from User u where u.email=:email", User.class).setParameter("email", email).getSingleResult();
049    }
050}
