001package persistence.repo;
002
003import org.hibernate.Session;
004import org.hibernate.SimpleNaturalIdLoadAccess;
005import persistence.model.User;
006
007import javax.persistence.EntityManager;
008import javax.persistence.PersistenceContext;
009import java.io.Serializable;
010
011
012public class UserRepository implements Serializable {
013
014    @PersistenceContext
015    protected EntityManager em;
016
017    /**
018     * Restituisce i dati di un utente dato un nome
019     * @param name nome dell'utente
020     * @return entità User
021     */
022    public User getByName(String name) {
023        return getByName(name, false);
024    }
025
026    /**
027     * Restituisce i dati di un utente dato un nome
028     * @param name nome dell'utente
029     * @param getReference booleano per attivare lazy loading
030     * @return entità User
031     */
032    public User getByName(String name, boolean getReference) {
033        SimpleNaturalIdLoadAccess<User> user = em.unwrap(Session.class).bySimpleNaturalId(User.class);
034        return getReference ? user.getReference(name) : user.load(name);
035    }
036
037    /**
038     * Restituisce i dati di un utente data un email
039     * @param email email dell'utente
040     * @return entità User
041     */
042    public User getByEmail(String email) {
043        return em.createQuery("from User u where u.email=:email", User.class).setParameter("email", email).getSingleResult();
044    }
045}
