001package persistence.repo;
002
003import persistence.model.Comment;
004import persistence.model.Post;
005
006import javax.persistence.EntityManager;
007import javax.persistence.PersistenceContext;
008import java.io.Serializable;
009import java.util.List;
010
011public class CommentRepository implements Serializable {
012
013    @PersistenceContext
014    protected EntityManager em;
015
016    public List<Comment> getByPost(Post post, int depth){
017        return em.createQuery(
018                        "from Comment c where c.post = :post and length(c.path) <= (7 * :depth) + 1 order by c.path",
019                        Comment.class)
020                .setParameter("depth", depth)
021                .setParameter("post", post)
022                .getResultList();
023    }
024
025    public List<Comment> getReplies(Comment comment, int depth){
026        return em.createQuery(
027                    "from Comment c where c.path like :path and length(c.path) <= (7 * :depth) + 1 order by c.path",
028                    Comment.class)
029                .setParameter("depth", depth)
030                .setParameter("path", comment.getPath() + '%')
031                .getResultList();
032    }
033}
