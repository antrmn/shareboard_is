001package persistence.repo;
002
003import persistence.model.Comment;
004import persistence.model.Post;
005
006import javax.persistence.EntityManager;
007import javax.persistence.PersistenceContext;
008import java.io.Serializable;
009import java.util.List;
010
011/**
012 * Interfaccia usata per interagire con {@link javax.persistence.EntityManager}
013 * Istanze che implementano questa interfaccia fungono da facciata alle funzionalità di JPA, fornendo i metodi
014 * essenziali (CRUD) per una specifica entità di persistenza
015 * @param <T> La classe delle entità di persistenza gestite da classi che implementano questa interfaccia
016 * @param <ID> La classe della chiave primaria delle entità di persistenza gestite da classi che implementano questa
017 *            interfaccia
018 * @see javax.persistence.EntityManager
019 */
020public class CommentRepository implements Serializable {
021
022    @PersistenceContext
023    protected EntityManager em;
024
025    /**
026     *  Trova tutti i commenti di un determinato post
027     * @param post entità Post di cui si vuole ottenere i commenti
028     * @param depth profondità della ricorsione per le risposte ai post
029     * @return Lista di commenti
030     */
031    public List<Comment> getByPost(Post post, int depth){
032        return em.createQuery(
033                        "from Comment c where c.post = :post and length(c.path) <= (7 * :depth) + 1 order by c.path",
034                        Comment.class)
035                .setParameter("depth", depth)
036                .setParameter("post", post)
037                .getResultList();
038    }
039
040    /**
041     *  Trova tutti i commenti di riposta a un determinato commento
042     * @param comment Entità Comment di cui si vuole ottenere i commenti
043     * @param depth profondità della ricorsione per le risposte ai post
044     * @return Lista di commenti
045     */
046    public List<Comment> getReplies(Comment comment, int depth){
047        return em.createQuery(
048                    "from Comment c where c.path like :path and length(c.path) <= (7 * :depth) + 1 order by c.path",
049                    Comment.class)
050                .setParameter("depth", depth)
051                .setParameter("path", comment.getPath() + '%')
052                .getResultList();
053    }
054}
