001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.controller.interceptor.JSONError;
005import http.util.ParameterConverter;
006import http.util.interceptor.InterceptableServlet;
007import service.VoteService;
008
009import javax.inject.Inject;
010import javax.servlet.ServletException;
011import javax.servlet.annotation.WebServlet;
012import javax.servlet.http.HttpServletRequest;
013import javax.servlet.http.HttpServletResponse;
014import java.io.IOException;
015
016import static http.controller.interceptor.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
017
018@WebServlet("/vote")
019@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
020public class VoteServlet extends InterceptableServlet {
021    @Inject private VoteService service;
022
023    @Override
024    @JSONError
025    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
026        ParameterConverter converter = new ParameterConverter(request);
027        int id = converter.getIntParameter("id").orElse(0);
028        String vote = request.getParameter("vote");
029        String type = request.getParameter("type");
030
031        if("post".equalsIgnoreCase(type)){
032            if("upvote".equalsIgnoreCase(vote))
033                service.upvotePost(id);
034            else if("downvote".equalsIgnoreCase(vote))
035                service.downvotePost(id);
036            else
037                throw new IllegalArgumentException();
038        }else if("comment".equalsIgnoreCase(type)){
039            if("upvote".equalsIgnoreCase(vote))
040                service.upvoteComment(id);
041            else if("downvote".equalsIgnoreCase(vote))
042                service.downvoteComment(id);
043            else
044                throw new IllegalArgumentException();
045        }else{
046            throw new IllegalArgumentException();
047        }
048    }
049
050    @Override
051    @JSONError
052    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
053        doGet(request,response);
054    }
055}
