001package http.controller.filter;
002
003import service.dto.CurrentUser;
004
005import javax.inject.Inject;
006import javax.inject.Named;
007import javax.servlet.FilterChain;
008import javax.servlet.ServletException;
009import javax.servlet.annotation.WebFilter;
010import javax.servlet.http.HttpFilter;
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013import java.io.IOException;
014
015@WebFilter("*")
016public class CheckCurrentUserFilter extends HttpFilter {
017    @Inject @Named("currentUser") CurrentUser currentUser;
018
019    @Override
020    protected void doFilter(HttpServletRequest request, HttpServletResponse response, FilterChain chain) throws IOException, ServletException {
021        request.setAttribute("currentUser",currentUser);
022        super.doFilter(request, response, chain);
023    }
024}
