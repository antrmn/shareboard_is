001package http.controller.filter;
002
003import service.dto.CurrentUser;
004
005import javax.inject.Inject;
006import javax.servlet.FilterChain;
007import javax.servlet.ServletException;
008import javax.servlet.annotation.WebFilter;
009import javax.servlet.http.HttpFilter;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpServletResponse;
012import java.io.IOException;
013import java.net.URI;
014import java.net.URISyntaxException;
015
016@WebFilter("*")
017public class AliasFilter extends HttpFilter {
018    @Inject CurrentUser currentUser;
019
020    @Override
021    protected void doFilter(HttpServletRequest req, HttpServletResponse res, FilterChain chain) throws IOException, ServletException {
022        String requri;
023        try {
024            requri = new URI(req.getRequestURI())
025                    .normalize() //rimuove slash multipli e simili
026                    .toString()
027                    .substring(req.getContextPath().length() +1 );
028        } catch (URISyntaxException e) {
029            chain.doFilter(req, res);
030            return;
031        }
032        String[] splits = requri.split("/+");
033        switch(splits[0]){
034            case "me":{
035                if(currentUser.isLoggedIn()){
036                    req.getRequestDispatcher("/user?name=" + currentUser.getUsername() + '&' + req.getQueryString())
037                            .forward(req, res);
038                    return;
039                } else {
040                    res.sendError(HttpServletResponse.SC_UNAUTHORIZED);
041                    return;
042                }
043            }
044            case "popular": {
045                req.getRequestDispatcher("/home?view=popular" + '&' + req.getQueryString()).forward(req, res);
046                return;
047            }
048            case "feed":{
049                req.getRequestDispatcher("/home?view=following" + '&' + req.getQueryString()).forward(req, res);
050                return;
051            }
052            case "s":{
053                if(splits.length >= 2) {
054                    req.getRequestDispatcher("/s?section=" + splits[1] + '&' + req.getQueryString()).forward(req, res);
055                    return;
056                }
057                break;
058            }
059            case "u": {
060                if (splits.length >= 2) {
061                    req.getRequestDispatcher("/user?name=" + splits[1] + '&' + req.getQueryString()).forward(req, res);
062                    return;
063                }
064                break;
065            }
066            case "post":{
067                if(splits.length >= 2){
068                    req.getRequestDispatcher("/post?id=" + splits[1] + '&' + req.getQueryString()).forward(req, res);
069                    return;
070                }
071                break;
072            }
073        }
074
075        chain.doFilter(req, res);
076    }
077}
