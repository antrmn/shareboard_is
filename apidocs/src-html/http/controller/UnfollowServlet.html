001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.controller.interceptor.JSONError;
005import http.util.ParameterConverter;
006import http.util.interceptor.InterceptableServlet;
007import service.FollowService;
008
009import javax.inject.Inject;
010import javax.servlet.ServletException;
011import javax.servlet.annotation.WebServlet;
012import javax.servlet.http.HttpServletRequest;
013import javax.servlet.http.HttpServletResponse;
014import java.io.IOException;
015
016import static http.controller.interceptor.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
017
018@WebServlet("/unfollow")
019@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
020public class UnfollowServlet extends InterceptableServlet {
021    @Inject private FollowService service;
022
023    @Override
024    @JSONError
025    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
026        ParameterConverter converter = new ParameterConverter(request);
027        int sectionId = converter.getIntParameter("section").orElse(0);
028        service.unFollow(sectionId);
029    }
030
031    @Override
032    @JSONError
033    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
034        doGet(request, response);
035    }
036}
