001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.util.interceptor.InterceptableServlet;
005import service.UserService;
006import service.dto.UserProfile;
007
008import javax.inject.Inject;
009import javax.servlet.ServletException;
010import javax.servlet.annotation.WebServlet;
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013import java.io.IOException;
014import java.util.List;
015
016import static http.controller.interceptor.AuthorizationConstraints.Types.ADMINS_ONLY;
017
018@WebServlet("/admin/showusers")
019@AuthorizationConstraints(ADMINS_ONLY)
020public class ShowUsersServlet extends InterceptableServlet {
021
022    @Inject private UserService service;
023
024    @Override
025    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
026        List<UserProfile> usersDto = service.showUsers();
027        request.setAttribute("users", usersDto);
028        request.getRequestDispatcher("/WEB-INF/views/crm/show-users.jsp").forward(request,response);
029    }
030}
