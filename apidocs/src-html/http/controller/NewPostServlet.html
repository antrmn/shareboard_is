001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.controller.interceptor.ForwardOnError;
005import http.util.ParameterConverter;
006import http.util.interceptor.InterceptableServlet;
007import service.PostService;
008import service.SectionService;
009
010import javax.inject.Inject;
011import javax.servlet.ServletException;
012import javax.servlet.annotation.MultipartConfig;
013import javax.servlet.annotation.WebServlet;
014import javax.servlet.http.HttpServletRequest;
015import javax.servlet.http.HttpServletResponse;
016import javax.servlet.http.Part;
017import java.io.BufferedInputStream;
018import java.io.IOException;
019
020import static http.controller.interceptor.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
021
022@WebServlet("/newpost")
023@MultipartConfig
024@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
025public class NewPostServlet extends InterceptableServlet {
026    @Inject private PostService service;
027    @Inject private SectionService sectionService;
028
029    private static final String NEW_POST_PAGE = "/WEB-INF/views/section/create-post.jsp";
030
031    @Override
032    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
033        request.getRequestDispatcher(NEW_POST_PAGE).forward(request, response);
034    }
035
036    @Override
037    @ForwardOnError("/WEB-INF/views/section/create-post.jsp")
038    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
039        ParameterConverter converter = new ParameterConverter(request);
040        int sectionId = converter.getIntParameter("section").orElse(0);
041        String sectionName = sectionService.showSection(sectionId).getName();
042
043        String title = request.getParameter("title");
044        String type = request.getParameter("type");
045        String content = request.getParameter("content");
046        Part picture = request.getPart("picture");
047
048        int newPostId;
049        if(type.equalsIgnoreCase("text")){
050            newPostId = service.newPost(title,content,sectionName);
051        }else{
052            BufferedInputStream stream = picture.getSize() > 0 ? new BufferedInputStream(picture.getInputStream()) : null;
053            newPostId = service.newPost(title,stream, sectionName);
054        }
055        response.sendRedirect(getServletContext().getContextPath() + "/post/" + newPostId);
056    }
057}
