001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.util.ParameterConverter;
005import http.util.interceptor.InterceptableServlet;
006import service.BanService;
007
008import javax.inject.Inject;
009import javax.servlet.ServletException;
010import javax.servlet.annotation.WebServlet;
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013import java.io.IOException;
014
015import static http.controller.interceptor.AuthorizationConstraints.Types.ADMINS_ONLY;
016
017@WebServlet("/admin/deleteban")
018@AuthorizationConstraints(ADMINS_ONLY)
019public class DeleteBanServlet extends InterceptableServlet {
020    @Inject private BanService service;
021
022    @Override
023    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
024        ParameterConverter converter = new ParameterConverter(request);
025        int banId = converter.getIntParameter("banId").orElse(0);
026        service.removeBan(banId);
027        response.sendRedirect(getServletContext().getContextPath() + "/admin/showbans?userId=" + request.getParameter("userId"));
028    }
029
030    @Override
031    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
032        doGet(request,response);
033    }
034}
