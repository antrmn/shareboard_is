001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.controller.interceptor.ForwardOnError;
005import http.util.ParameterConverter;
006import http.util.interceptor.InterceptableServlet;
007import persistence.model.Post;
008import service.PostService;
009import service.dto.PostEditDTO;
010import service.dto.PostPage;
011
012import javax.inject.Inject;
013import javax.servlet.ServletException;
014import javax.servlet.annotation.MultipartConfig;
015import javax.servlet.annotation.WebServlet;
016import javax.servlet.http.HttpServletRequest;
017import javax.servlet.http.HttpServletResponse;
018import javax.servlet.http.Part;
019import java.io.BufferedInputStream;
020import java.io.IOException;
021
022import static http.controller.interceptor.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
023
024@WebServlet("/editpost")
025@MultipartConfig
026@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
027public class EditPostServlet extends InterceptableServlet {
028    @Inject private PostService service;
029
030    private static final String EDIT_POST_PAGE = "/WEB-INF/views/section/edit-post.jsp";
031
032    @Override
033    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
034        //check necessario?
035        ParameterConverter converter = new ParameterConverter(request);
036        int postId = converter.getIntParameter("id").orElse(0);
037        PostPage post =  service.getPost(postId);
038        request.setAttribute("post", post);
039        request.getRequestDispatcher(EDIT_POST_PAGE).forward(request, response);
040    }
041
042    @Override
043    @ForwardOnError(EDIT_POST_PAGE)
044    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
045        ParameterConverter converter = new ParameterConverter(request);
046        String title = request.getParameter("title");
047        String type = request.getParameter("type");
048        Post.Type postType = type != null && type.equalsIgnoreCase("text") ? Post.Type.TEXT : Post.Type.IMG;
049        Part picture = request.getPart("picture");
050        String content = request.getParameter("content");
051
052        PostEditDTO postToEdit = new PostEditDTO(title,content,postType);
053        //in caso di immagine il content sarà null infatti non sarà utilizzato nel servizio EditPostIMG
054
055        int postId = converter.getIntParameter("id").orElse(0);
056        if(postType == Post.Type.TEXT){
057            service.editPost(postToEdit,postId);
058        }else{
059            BufferedInputStream buff = new BufferedInputStream(picture.getInputStream());
060            service.editPost(postToEdit,postId,buff);
061        }
062        response.sendRedirect(getServletContext().getContextPath() + "/post/" + postId);
063    }
064}
