001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.controller.interceptor.ForwardOnError;
005import http.util.ParameterConverter;
006import http.util.interceptor.InterceptableServlet;
007import service.UserService;
008import service.auth.AuthorizationException;
009import service.dto.CurrentUser;
010import service.dto.UserEditPage;
011import service.dto.UserProfile;
012
013import javax.inject.Inject;
014import javax.servlet.ServletException;
015import javax.servlet.annotation.MultipartConfig;
016import javax.servlet.annotation.WebServlet;
017import javax.servlet.http.HttpServletRequest;
018import javax.servlet.http.HttpServletResponse;
019import javax.servlet.http.Part;
020import java.io.BufferedInputStream;
021import java.io.IOException;
022
023import static http.controller.interceptor.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
024
025@WebServlet("/edituser")
026@MultipartConfig
027@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
028public class EditUserServlet extends InterceptableServlet {
029    @Inject private UserService service;
030    @Inject private CurrentUser currentUser;
031
032    private static final String EDIT_USER_PAGE = "/WEB-INF/views/edit-user.jsp";
033
034    @Override
035    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
036        ParameterConverter converter = new ParameterConverter(request);
037        int userId = converter.getIntParameter("id").orElse(0);
038        if(!currentUser.isAdmin() && currentUser.getId() != userId)
039            throw new AuthorizationException();
040        UserProfile user = service.getUser(userId);
041        request.setAttribute("user", user);
042        request.getRequestDispatcher("/WEB-INF/views/edit-user.jsp").forward(request, response);
043    }
044
045    @Override
046    @ForwardOnError(EDIT_USER_PAGE)
047    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
048        ParameterConverter converter = new ParameterConverter(request);
049        int userId = converter.getIntParameter("id").orElse(0);
050        String email = request.getParameter("email");
051        String description = request.getParameter("description");
052        Part picture = request.getPart("picture");
053        String pass = request.getParameter("pass");
054        String pass2 = request.getParameter("pass2");
055
056        if((!pass.isEmpty() || !pass2.isEmpty()) && !pass.equals(pass2))
057            throw new IllegalArgumentException("Le password devono coincidere");
058
059        BufferedInputStream stream = picture.getSize() > 0 ? new BufferedInputStream(picture.getInputStream()) : null;
060        UserEditPage userEditPage = new UserEditPage(description,email,stream ,pass);
061        service.edit(userEditPage,userId);
062
063        response.sendRedirect(request.getContextPath() + "/u/" + service.getUsernameById(userId));
064    }
065}
