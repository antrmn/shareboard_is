001package http.controller;
002
003import http.util.interceptor.InterceptableServlet;
004
005import javax.servlet.ServletException;
006import javax.servlet.annotation.WebServlet;
007import javax.servlet.http.HttpServletRequest;
008import javax.servlet.http.HttpServletResponse;
009import java.io.IOException;
010
011@WebServlet("/logout")
012public class LogoutServlet extends InterceptableServlet {
013
014    @Override
015    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
016        req.getSession(true).invalidate();
017        resp.sendRedirect(req.getContextPath());
018    }
019}
