001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.controller.interceptor.ForwardOnError;
005import http.util.ParameterConverter;
006import http.util.interceptor.InterceptableServlet;
007import service.SectionService;
008import service.dto.SectionPage;
009
010import javax.inject.Inject;
011import javax.servlet.ServletException;
012import javax.servlet.annotation.MultipartConfig;
013import javax.servlet.annotation.WebServlet;
014import javax.servlet.http.HttpServletRequest;
015import javax.servlet.http.HttpServletResponse;
016import javax.servlet.http.Part;
017import java.io.BufferedInputStream;
018import java.io.IOException;
019
020import static http.controller.interceptor.AuthorizationConstraints.Types.ADMINS_ONLY;
021
022@WebServlet("/admin/editsection")
023@MultipartConfig
024@AuthorizationConstraints(ADMINS_ONLY)
025public class EditSectionServlet extends InterceptableServlet {
026    @Inject private SectionService service;
027
028    private static final String EDIT_SECTION_PAGE = "/WEB-INF/views/crm/edit-section.jsp";
029    private static final int MAX_FILE_SIZE = 5 * 1024 * 1024;
030
031    @Override
032    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
033        ParameterConverter converter = new ParameterConverter(request);
034        int sectionId = converter.getIntParameter("sectionId").orElse(0);
035        SectionPage section = service.showSection(sectionId);
036        request.setAttribute("section", section);
037        request.getRequestDispatcher(EDIT_SECTION_PAGE).forward(request, response);
038    }
039
040    @Override
041    @ForwardOnError(EDIT_SECTION_PAGE)
042    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
043        ParameterConverter converter = new ParameterConverter(request);
044        int sectionId = converter.getIntParameter("sectionId").orElse(0);
045        String description = request.getParameter("description");
046        Part picture = request.getPart("picture");
047        Part banner = request.getPart("banner");
048
049        BufferedInputStream buffPicture = null;
050        BufferedInputStream buffBanner = null;
051        if(picture != null && picture.getSize() < MAX_FILE_SIZE) {
052            if (picture.getSize()>0)
053                buffPicture = new BufferedInputStream(picture.getInputStream());
054        }else{
055            throw new IllegalArgumentException("Il file non deve superare i 5MB");
056        }
057
058        if(banner != null && banner.getSize() < MAX_FILE_SIZE) {
059            if (banner.getSize()>0)
060                buffBanner = new BufferedInputStream(banner.getInputStream());
061        }else{
062            throw new IllegalArgumentException("Il file non deve superare i 5MB");
063        }
064
065        SectionPage sectionPage = new SectionPage(sectionId,null,description,picture.getName(),banner.getName(),0,false); //false?
066        service.editSection(sectionPage,sectionId,buffPicture,buffBanner);
067        response.sendRedirect(request.getContextPath()+"/admin/showsections");
068    }
069}
