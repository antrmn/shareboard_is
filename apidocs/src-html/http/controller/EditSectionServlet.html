001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.controller.interceptor.ForwardOnError;
005import http.util.ParameterConverter;
006import http.util.interceptor.InterceptableServlet;
007import service.SectionService;
008import service.dto.SectionPage;
009
010import javax.inject.Inject;
011import javax.servlet.ServletException;
012import javax.servlet.annotation.MultipartConfig;
013import javax.servlet.annotation.WebServlet;
014import javax.servlet.http.HttpServletRequest;
015import javax.servlet.http.HttpServletResponse;
016import javax.servlet.http.Part;
017import java.io.BufferedInputStream;
018import java.io.IOException;
019
020import static http.controller.interceptor.AuthorizationConstraints.Types.ADMINS_ONLY;
021
022@WebServlet("/admin/editsection")
023@MultipartConfig
024@AuthorizationConstraints(ADMINS_ONLY)
025public class EditSectionServlet extends InterceptableServlet {
026    @Inject private SectionService service;
027
028    private static final String EDIT_SECTION_PAGE = "/WEB-INF/views/crm/edit-section.jsp";
029
030    @Override
031    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
032        ParameterConverter converter = new ParameterConverter(request);
033        int sectionId = converter.getIntParameter("sectionId").orElse(0);
034        SectionPage section = service.showSection(sectionId);
035        request.setAttribute("section", section);
036        request.getRequestDispatcher(EDIT_SECTION_PAGE).forward(request, response);
037    }
038
039    @Override
040    @ForwardOnError(EDIT_SECTION_PAGE)
041    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
042        ParameterConverter converter = new ParameterConverter(request);
043        int sectionId = converter.getIntParameter("sectionId").orElse(0);
044        String description = request.getParameter("description");
045        Part picture = request.getPart("picture");
046        Part banner = request.getPart("banner");
047
048        SectionPage sectionPage = new SectionPage(sectionId,null,description,picture.getName(),banner.getName(),0,false); //false?
049        BufferedInputStream buffPicture = new BufferedInputStream(picture.getInputStream());
050        BufferedInputStream buffBanner = new BufferedInputStream(banner.getInputStream());
051        service.editSection(sectionPage,sectionId,buffPicture,buffBanner);
052        response.sendRedirect(request.getContextPath()+"/admin/showsections");
053    }
054}
