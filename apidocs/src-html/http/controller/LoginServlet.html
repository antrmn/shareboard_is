001package http.controller;
002
003import http.util.interceptor.InterceptableServlet;
004import service.AuthenticationService;
005import service.dto.CurrentUser;
006
007import javax.inject.Inject;
008import javax.servlet.ServletException;
009import javax.servlet.annotation.WebServlet;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpServletResponse;
012import java.io.IOException;
013import java.util.Optional;
014
015@WebServlet("/login")
016public class LoginServlet extends InterceptableServlet {
017    @Inject AuthenticationService authenticationService;
018    @Inject CurrentUser currentUser;
019
020    @Override
021    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
022        if(currentUser.isLoggedIn())
023            resp.sendRedirect(req.getContextPath());
024        else
025            req.getRequestDispatcher("/WEB-INF/views/login.jsp").forward(req, resp);
026    }
027
028    @Override
029    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
030        if(currentUser.isLoggedIn()){
031            resp.sendRedirect(req.getContextPath());
032            return;
033        }
034
035        String username = Optional.ofNullable(req.getParameter("username")).orElse("");
036        String password = Optional.ofNullable(req.getParameter("pass")).orElse("");
037
038        boolean loginSuccessful = authenticationService.authenticate(username, password);
039
040        if(loginSuccessful)
041            resp.sendRedirect(req.getContextPath());
042        else
043            req.getRequestDispatcher("/WEB-INF/views/login.jsp?error").forward(req, resp);
044    }
045}
