001package http.controller.interceptor;
002
003import http.util.interceptor.HttpServletBiConsumer;
004import http.util.interceptor.ServletInterceptor;
005
006import javax.servlet.ServletException;
007import javax.servlet.http.HttpServletRequest;
008import javax.servlet.http.HttpServletResponse;
009import javax.validation.ConstraintViolation;
010import javax.validation.ConstraintViolationException;
011import java.io.IOException;
012import java.util.Collection;
013import java.util.Collections;
014import java.util.List;
015import java.util.stream.Collectors;
016
017import static javax.servlet.http.HttpServletResponse.SC_BAD_REQUEST;
018
019public class ForwardOnErrorInterceptor extends ServletInterceptor<ForwardOnError> {
020
021    private String page;
022
023    @Override
024    protected void init(ForwardOnError annotation) {
025        this.page = annotation.value();
026    }
027
028    @Override
029    public void handle(HttpServletRequest req, HttpServletResponse resp, HttpServletBiConsumer next) throws ServletException, IOException {
030        try{
031            next.handle(req,resp);
032        } catch (IllegalArgumentException e) {
033            forwardErrorPage(req, resp, SC_BAD_REQUEST, e.getMessage());
034        } catch (ConstraintViolationException e) {
035            List<String> collect = e.getConstraintViolations().stream()
036                    .map(ConstraintViolation::getMessage)
037                    .collect(Collectors.toList());
038            forwardErrorPage(req,resp,SC_BAD_REQUEST,collect);
039        }
040    }
041
042    private void forwardErrorPage(HttpServletRequest req, HttpServletResponse resp,
043                                  int code, Collection<String> messages) throws ServletException, IOException {
044        req.setAttribute("errors", messages);
045        resp.setStatus(code);
046        req.getRequestDispatcher(page).forward(req, resp);
047    }
048
049    private void forwardErrorPage(HttpServletRequest req, HttpServletResponse resp, int code, String message) throws IOException, ServletException {
050        List<String> singleton = message == null ? Collections.emptyList() : List.of(message);
051        forwardErrorPage(req, resp, code, singleton);
052    }
053
054    private void forwardErrorPage(HttpServletRequest req, HttpServletResponse resp, int code) throws IOException, ServletException {
055        forwardErrorPage(req, resp,code,Collections.emptyList());
056    }
057
058    @Override
059    public int priority() {
060        return 1;
061    }
062}
