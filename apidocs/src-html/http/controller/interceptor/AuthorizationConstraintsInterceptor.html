001package http.controller.interceptor;
002
003import http.util.interceptor.HttpServletBiConsumer;
004import http.util.interceptor.ServletInterceptor;
005import service.auth.AuthenticationRequiredException;
006import service.auth.AuthorizationException;
007import service.auth.BannedUserException;
008import service.dto.CurrentUser;
009
010import javax.servlet.ServletException;
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013import java.io.IOException;
014import java.util.Arrays;
015import java.util.EnumSet;
016import java.util.Set;
017
018import static http.controller.interceptor.AuthorizationConstraints.Types.*;
019
020public class AuthorizationConstraintsInterceptor extends ServletInterceptor<AuthorizationConstraints> {
021
022    private Set<AuthorizationConstraints.Types> authorizationConstraintTypes;
023
024    @Override
025    protected void init(AuthorizationConstraints annotation) {
026        authorizationConstraintTypes = EnumSet.noneOf(AuthorizationConstraints.Types.class);
027        authorizationConstraintTypes.addAll(Arrays.asList(annotation.value()));
028    }
029
030    @Override
031    public void handle(HttpServletRequest req, HttpServletResponse resp, HttpServletBiConsumer next) throws ServletException, IOException {
032        CurrentUser currentUser = (CurrentUser) req.getAttribute("currentUser");
033        if(authorizationConstraintTypes.contains(ADMINS_ONLY)
034                || authorizationConstraintTypes.contains(REQUIRE_AUTHENTICATION)){
035            //controlla se l'utente Ã¨ autenticato
036            if(!currentUser.isLoggedIn()){
037                throw new AuthenticationRequiredException();
038            }
039        }
040
041        if(authorizationConstraintTypes.contains(ADMINS_ONLY)){
042            if(!currentUser.isAdmin()){
043                throw new AuthorizationException();
044            }
045        }
046
047        if(authorizationConstraintTypes.contains(DENY_BANNED_USERS)){
048            if(currentUser.getBanDuration() != null){
049                throw new BannedUserException();
050            }
051        }
052
053        next.handle(req,resp);
054    }
055
056    @Override
057    public int priority() {
058        return super.priority();
059    }
060}
