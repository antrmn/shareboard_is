001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.controller.interceptor.ForwardOnError;
005import http.util.interceptor.InterceptableServlet;
006import service.SectionService;
007import service.dto.SectionPage;
008
009import javax.inject.Inject;
010import javax.servlet.ServletException;
011import javax.servlet.annotation.MultipartConfig;
012import javax.servlet.annotation.WebServlet;
013import javax.servlet.http.HttpServletRequest;
014import javax.servlet.http.HttpServletResponse;
015import javax.servlet.http.Part;
016import java.io.BufferedInputStream;
017import java.io.IOException;
018
019import static http.controller.interceptor.AuthorizationConstraints.Types.ADMINS_ONLY;
020
021@WebServlet("/admin/newsection")
022@MultipartConfig
023@AuthorizationConstraints(ADMINS_ONLY)
024public class NewSectionServlet extends InterceptableServlet {
025    @Inject private SectionService service;
026
027    private static final String NEW_SECTION_PAGE = "/WEB-INF/views/crm/create-section.jsp";
028
029    @Override
030    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
031        request.getRequestDispatcher("/WEB-INF/views/crm/create-section.jsp").forward(request, response);
032    }
033
034    @Override
035    @ForwardOnError(NEW_SECTION_PAGE)
036    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
037        String name = request.getParameter("name");
038        String description = request.getParameter("description");
039        Part picture = request.getPart("picture");
040        Part banner = request.getPart("banner");
041
042        //gestire errore immagine
043
044        SectionPage sectionPage = new SectionPage(0,name,description,picture.getName(),banner.getName(),0,false);//false?
045        BufferedInputStream buffPicture = new BufferedInputStream(picture.getInputStream());
046        BufferedInputStream buffBanner = new BufferedInputStream(banner.getInputStream());
047        service.newSection(sectionPage,buffPicture,buffBanner);
048        response.sendRedirect(getServletContext().getContextPath()+"/admin/showsections");
049    }
050}
