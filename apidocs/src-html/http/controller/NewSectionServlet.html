001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.controller.interceptor.ForwardOnError;
005import http.util.interceptor.InterceptableServlet;
006import service.SectionService;
007
008import javax.inject.Inject;
009import javax.servlet.ServletException;
010import javax.servlet.annotation.MultipartConfig;
011import javax.servlet.annotation.WebServlet;
012import javax.servlet.http.HttpServletRequest;
013import javax.servlet.http.HttpServletResponse;
014import javax.servlet.http.Part;
015import java.io.BufferedInputStream;
016import java.io.IOException;
017
018import static http.controller.interceptor.AuthorizationConstraints.Types.ADMINS_ONLY;
019
020@WebServlet("/admin/newsection")
021@MultipartConfig
022@AuthorizationConstraints(ADMINS_ONLY)
023public class NewSectionServlet extends InterceptableServlet {
024    @Inject private SectionService service;
025
026    private static final String NEW_SECTION_PAGE = "/WEB-INF/views/crm/create-section.jsp";
027
028    @Override
029    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
030        request.getRequestDispatcher("/WEB-INF/views/crm/create-section.jsp").forward(request, response);
031    }
032
033    @Override
034    @ForwardOnError(NEW_SECTION_PAGE)
035    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
036        String name = request.getParameter("name");
037        String description = request.getParameter("description");
038        Part picture = request.getPart("picture");
039        Part banner = request.getPart("banner");
040
041        BufferedInputStream streamPicture = picture.getSize() > 0 ? new BufferedInputStream(picture.getInputStream()) : null;
042        BufferedInputStream streamBanner = banner.getSize() > 0 ? new BufferedInputStream(picture.getInputStream()) : null;
043
044        service.newSection(name, description ,streamPicture,streamBanner);
045        response.sendRedirect(getServletContext().getContextPath()+"/admin/showsections");
046    }
047}
