001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.util.ParameterConverter;
005import http.util.interceptor.InterceptableServlet;
006import service.CommentService;
007
008import javax.inject.Inject;
009import javax.servlet.ServletException;
010import javax.servlet.annotation.HttpConstraint;
011import javax.servlet.annotation.ServletSecurity;
012import javax.servlet.annotation.WebServlet;
013import javax.servlet.http.HttpServletRequest;
014import javax.servlet.http.HttpServletResponse;
015import java.io.IOException;
016
017import static http.controller.interceptor.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
018
019@WebServlet("/newcomment")
020@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
021public class NewCommentServlet extends InterceptableServlet {
022    @Inject private CommentService service;
023
024    @Override
025    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
026        doPost(request,response);
027    }
028
029    @Override
030    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
031        ParameterConverter converter = new ParameterConverter(request);
032        int postId = converter.getIntParameter("id").orElse(0);
033        int parentId = converter.getIntParameter("parent").orElse(0);
034        String text = request.getParameter("text");
035
036        if(postId > 0){
037            service.newComment(text,postId);
038        }else{
039            service.newCommentReply(text,parentId,postId);
040        }
041        response.sendRedirect(request.getContextPath() + "/post/" + postId +"#comment-container");
042    }
043}
