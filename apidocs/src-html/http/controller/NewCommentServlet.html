001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.util.ParameterConverter;
005import http.util.interceptor.InterceptableServlet;
006import service.CommentService;
007
008import javax.inject.Inject;
009import javax.servlet.ServletException;
010import javax.servlet.annotation.WebServlet;
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013import java.io.IOException;
014
015import static http.controller.interceptor.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
016
017@WebServlet("/newcomment")
018@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
019public class NewCommentServlet extends InterceptableServlet {
020    @Inject private CommentService service;
021
022    @Override
023    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
024        doPost(request,response);
025    }
026
027    @Override
028    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
029        ParameterConverter converter = new ParameterConverter(request);
030        int postId = converter.getIntParameter("id").orElse(0);
031        int parentId = converter.getIntParameter("parent").orElse(0);
032        String text = request.getParameter("text");
033
034        if(postId > 0){
035            service.newComment(text,postId);
036        }else{
037            service.newCommentReply(text,parentId);
038        }
039        response.sendRedirect(request.getContextPath() + "/post/" + postId +"#comment-container");
040    }
041}
