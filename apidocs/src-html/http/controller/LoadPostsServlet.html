001package http.controller;
002
003import http.util.ParameterConverter;
004import http.util.interceptor.InterceptableServlet;
005import service.PostService;
006import service.dto.PostPage;
007import service.dto.PostSearchForm;
008
009import javax.inject.Inject;
010import javax.servlet.ServletException;
011import javax.servlet.annotation.WebServlet;
012import javax.servlet.http.HttpServletRequest;
013import javax.servlet.http.HttpServletResponse;
014import java.io.IOException;
015import java.time.Instant;
016import java.time.LocalDate;
017import java.time.ZoneOffset;
018import java.util.List;
019import java.util.Map;
020import java.util.function.Function;
021
022import static service.dto.PostSearchForm.SortCriteria.*;
023
024@WebServlet("/loadposts")
025public class LoadPostsServlet extends InterceptableServlet {
026    @Inject private PostService postService;
027
028    private static final Function<LocalDate, Instant> LOCALDATE_TO_INSTANT =
029            x -> x.atStartOfDay().toInstant(ZoneOffset.UTC);
030
031    private static final Map<String, PostSearchForm.SortCriteria> SORT_CRITERIA = Map.of(
032            "oldest", OLDEST,
033            "newest", NEWEST,
034            "mostvoted", MOSTVOTED);
035
036    @Override
037    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
038        ParameterConverter converter = new ParameterConverter(req);
039        String content = req.getParameter("content");
040        String section = req.getParameter("section");
041        String author = req.getParameter("author");
042        Instant postedAfter = converter.getDateParameter("postedAfter")
043                .map(LOCALDATE_TO_INSTANT).orElse(null);
044        Instant postedBefore = converter.getDateParameter("postedBefore")
045                .map(LOCALDATE_TO_INSTANT).orElse(null);
046        PostSearchForm.SortCriteria orderBy = SORT_CRITERIA.get(req.getParameter("orderby"));
047        int page = converter.getIntParameter("page").orElse(1);
048        boolean onlyFollow = "on".equals(req.getParameter("onlyfollow"));
049        boolean includeBody = req.getParameter("includeBody") != null;
050
051        PostSearchForm postSearchForm = PostSearchForm.builder()
052                .content(content)
053                .onlyFollow(onlyFollow)
054                .includeBody(includeBody)
055                .sectionName(section)
056                .authorName(author)
057                .orderBy(orderBy)
058                .postedAfter(postedAfter)
059                .postedBefore(postedBefore)
060                .page(page)
061                .build();
062
063        List<PostPage> posts = postService.loadPosts(postSearchForm);
064        req.setAttribute("posts",posts);
065        req.getRequestDispatcher("/WEB-INF/views/partials/post-previews.jsp").forward(req,resp);
066    }
067
068    @Override
069    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
070        doGet(req, resp);
071    }
072}
