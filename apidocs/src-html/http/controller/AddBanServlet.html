001package http.controller;
002
003import http.controller.interceptor.AuthorizationConstraints;
004import http.controller.interceptor.JSONError;
005import http.util.ParameterConverter;
006import http.util.interceptor.InterceptableServlet;
007import service.BanService;
008
009import javax.inject.Inject;
010import javax.servlet.ServletException;
011import javax.servlet.annotation.WebServlet;
012import javax.servlet.http.HttpServletRequest;
013import javax.servlet.http.HttpServletResponse;
014import java.io.IOException;
015import java.time.Instant;
016import java.time.ZoneOffset;
017
018import static http.controller.interceptor.AuthorizationConstraints.Types.ADMINS_ONLY;
019
020@WebServlet("/admin/addban")
021@AuthorizationConstraints(ADMINS_ONLY)
022public class AddBanServlet extends InterceptableServlet {
023    @Inject private BanService service;
024
025    @Override
026    @JSONError
027    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
028        doPost(request,response);
029    }
030
031    @Override
032    @JSONError
033    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
034        ParameterConverter converter = new ParameterConverter(request);
035        Instant endDate = converter.getDateParameter("endDate")
036                .map(date -> date.atStartOfDay().toInstant(ZoneOffset.UTC)) //trasforma solo se l'optional non Ã¨ vuoto
037                .orElse(null);
038        int userId = converter.getIntParameter("userId").orElse(0);
039        service.addBan(endDate,userId);
040    }
041}
