001package http.controller;
002
003import http.controller.interceptor.ForwardOnError;
004import http.util.interceptor.InterceptableServlet;
005import service.AuthenticationService;
006import service.PostService;
007import service.UserService;
008import service.dto.CurrentUser;
009
010import javax.inject.Inject;
011import javax.servlet.ServletException;
012import javax.servlet.annotation.WebServlet;
013import javax.servlet.http.HttpServletRequest;
014import javax.servlet.http.HttpServletResponse;
015import java.io.IOException;
016import java.util.Objects;
017
018
019@WebServlet("/register")
020public class RegisterServlet extends InterceptableServlet {
021    @Inject private UserService userService;
022    @Inject private CurrentUser currentUser;
023    @Inject private AuthenticationService authenticationService;
024    @Inject private PostService service;
025
026    private static final String REGISTER_PAGE = "/WEB-INF/views/register.jsp";
027
028    @Override
029    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
030        if(currentUser.isLoggedIn()){
031            resp.sendRedirect(req.getContextPath());
032            return;
033        }
034        req.getRequestDispatcher(REGISTER_PAGE).forward(req, resp);
035    }
036
037
038    @Override
039    @ForwardOnError(REGISTER_PAGE)
040    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
041        if(currentUser.isLoggedIn()){
042            resp.sendRedirect(req.getContextPath());
043            return;
044        }
045        String email = req.getParameter("mail");
046        String username = req.getParameter("username");
047        String password = req.getParameter("pass");
048        String confirmPassword = req.getParameter("pass2");
049
050        if(!Objects.equals(password, confirmPassword)){
051            throw new IllegalArgumentException("Le password devono coincidere");
052        }
053        userService.newUser(email, username, password);
054        authenticationService.authenticate(username,password);
055        resp.sendRedirect(req.getContextPath());
056
057    }
058}
