001package http.util.interceptor;
002
003import javax.servlet.ServletException;
004import javax.servlet.ServletRequest;
005import javax.servlet.ServletResponse;
006import javax.servlet.http.HttpServlet;
007import javax.servlet.http.HttpServletRequest;
008import javax.servlet.http.HttpServletResponse;
009import java.io.IOException;
010import java.lang.reflect.Method;
011import java.util.Arrays;
012import java.util.Comparator;
013import java.util.Map;
014import java.util.Objects;
015import java.util.concurrent.ConcurrentHashMap;
016import java.util.stream.Stream;
017
018
019public abstract class InterceptableServlet extends HttpServlet {
020    private static final Map<String, String> methods =
021            Map.of(
022                    "GET",     "doGet",
023                    "POST",    "doPost",
024                    "PUT",     "doPut",
025                    "TRACE",   "doTrace",
026                    "OPTIONS", "doOptions",
027                    "DELETE",  "doDelete",
028                    "HEAD",    "doHead"
029            );
030
031    private final Map<String, HttpServletBiConsumer> chains = new ConcurrentHashMap<>();
032
033    private ServletInterceptor<?>[] getInterceptors(String httpMethodName) {
034        String javaMethodName = methods.get(httpMethodName);
035        if(javaMethodName == null){
036            //Not a "doX" method
037            throw new IllegalArgumentException();
038        }
039
040        Method method;
041        try {
042            method = this.getClass()
043                    .getDeclaredMethod(javaMethodName, HttpServletRequest.class, HttpServletResponse.class);
044        } catch (NoSuchMethodException e) {
045            //method has not been overriden by this class.
046            //return empty array
047            return new ServletInterceptor[]{};
048        }
049
050
051        //get annotations and their respective interceptor
052        return Stream.concat(Arrays.stream(getClass().getAnnotations()), Arrays.stream(method.getAnnotations()))
053                .map(ServletInterceptorFactory::instantiate)
054                .filter(Objects::nonNull)
055                .sorted(Comparator.comparingInt(ServletInterceptor::priority))
056                .toArray(ServletInterceptor[]::new);
057    }
058
059    private HttpServletBiConsumer buildChain(ServletInterceptor<?>[] interceptors, HttpServletBiConsumer target){
060        interceptors = (interceptors == null ? new ServletInterceptor[]{} : interceptors);
061
062        //builds the chain backwards
063        HttpServletBiConsumer current = target;
064        for(int i = interceptors.length-1; i >= 0; i--){
065            ServletInterceptor<?> interceptor = interceptors[i];
066
067            HttpServletBiConsumer next = current;
068            current = (req, res) -> interceptor.handle(req, res, next);
069        }
070        return current;
071    }
072
073    //unambiguous super.service (there are 2 service methods. java lambdas hate that.)
074    private void superService(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
075        super.service(req, resp);
076    }
077
078    @Override
079    protected final void service(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
080        String httpMethod = req.getMethod();
081        chains.computeIfAbsent(httpMethod, key -> buildChain(getInterceptors(key), this::superService)).handle(req,resp);
082    }
083
084    @Override
085    public final void service(ServletRequest req, ServletResponse res) throws ServletException, IOException {
086        service((HttpServletRequest) req, (HttpServletResponse) res);
087    }
088
089}
