001package http.util.interceptor;
002
003import javax.servlet.ServletException;
004import javax.servlet.http.HttpServletRequest;
005import javax.servlet.http.HttpServletResponse;
006import java.io.IOException;
007import java.lang.annotation.Annotation;
008
009
010public abstract class ServletInterceptor<A extends Annotation>{
011    protected abstract void init (A annotation);
012
013    public abstract void handle(HttpServletRequest req, HttpServletResponse resp, HttpServletBiConsumer next)
014            throws ServletException, IOException;
015
016    public int priority(){
017        return Integer.MAX_VALUE/2;
018    }
019}
