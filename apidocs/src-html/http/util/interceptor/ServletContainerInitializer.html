001package http.util.interceptor;
002
003import javax.servlet.ServletContext;
004import javax.servlet.annotation.HandlesTypes;
005import java.util.Set;
006
007@HandlesTypes(ServletInterceptor.class)
008public class ServletContainerInitializer implements javax.servlet.ServletContainerInitializer {
009    @Override
010    public void onStartup(Set<Class<?>> set, ServletContext servletContext) {
011        //noinspection unchecked
012        set.stream().map(x -> (Class<? extends ServletInterceptor<?>>) x).forEach(ServletInterceptorFactory::register);
013    }
014}
