001package http.util.interceptor;
002
003import java.lang.annotation.Annotation;
004import java.lang.reflect.Modifier;
005import java.lang.reflect.ParameterizedType;
006import java.util.concurrent.ConcurrentHashMap;
007import java.util.concurrent.ConcurrentMap;
008
009public final class ServletInterceptorFactory {
010    private ServletInterceptorFactory(){}
011
012    private static final
013    ConcurrentMap<Class<? extends Annotation>, Class<? extends ServletInterceptor<?>>> classes =
014            new ConcurrentHashMap<>();
015
016    public static <T extends Annotation> ServletInterceptor<T> instantiate(T annotation){
017        Class<? extends ServletInterceptor<?>> clazz = ServletInterceptorFactory.classes.get(annotation.annotationType());
018        if(clazz == null)
019            return null;
020
021        ServletInterceptor<T> interceptor;
022        try {
023            //noinspection unchecked
024            interceptor = (ServletInterceptor<T>) clazz.getConstructor().newInstance();
025        } catch (ReflectiveOperationException e) {
026            throw new RuntimeException(e);
027        }
028
029        interceptor.init(annotation);
030        return interceptor;
031    }
032
033    public static void register(Class<? extends ServletInterceptor<? extends Annotation>> interceptorClass){
034        if(Modifier.isAbstract(interceptorClass.getModifiers())){
035            throw new IllegalArgumentException("Interceptor class must be concrete");
036        }
037        if(interceptorClass.getTypeParameters().length > 0){
038            throw new IllegalArgumentException("Interceptor class must be non-generic");
039        }
040
041        //get ServletInterceptor.class immediate subclass
042        Class<?> clazz = interceptorClass;
043        while(clazz.getSuperclass() != ServletInterceptor.class){
044            clazz = clazz.getSuperclass();
045        }
046        //get ServletInterceptor.class as ParameterizedType object
047        ParameterizedType genericSuperclass = (ParameterizedType) clazz.getGenericSuperclass();
048
049        @SuppressWarnings("unchecked")
050        Class<? extends Annotation> actualTypeArgument =
051                (Class<? extends Annotation>) genericSuperclass.getActualTypeArguments()[0];
052
053        Class<?> oldValue = classes.putIfAbsent(actualTypeArgument, interceptorClass);
054        if(oldValue != null){
055            throw new IllegalArgumentException("There already is a registered Servlet Interceptor with annotation "
056                    + actualTypeArgument.getName());
057        }
058    }
059}
