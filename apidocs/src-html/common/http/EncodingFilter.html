001package common.http;
002
003import javax.servlet.FilterChain;
004import javax.servlet.ServletException;
005import javax.servlet.annotation.WebFilter;
006import javax.servlet.http.HttpFilter;
007import javax.servlet.http.HttpServletRequest;
008import javax.servlet.http.HttpServletResponse;
009import java.io.IOException;
010
011@WebFilter("*")
012public class EncodingFilter extends HttpFilter {
013    @Override
014    protected void doFilter(HttpServletRequest req, HttpServletResponse resp, FilterChain chain) throws IOException, ServletException {
015        req.setCharacterEncoding("UTF-8");
016        resp.setCharacterEncoding("UTF-8");
017        chain.doFilter(req, resp);
018    }
019}
