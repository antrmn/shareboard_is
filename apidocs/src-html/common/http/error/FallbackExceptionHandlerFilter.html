001package common.http.error;
002
003import usecase.auth.AuthenticationRequiredException;
004import usecase.auth.AuthorizationException;
005import usecase.auth.BannedUserException;
006
007import javax.servlet.FilterChain;
008import javax.servlet.ServletException;
009import javax.servlet.annotation.WebFilter;
010import javax.servlet.http.HttpFilter;
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013import javax.validation.ConstraintViolation;
014import javax.validation.ConstraintViolationException;
015import java.io.IOException;
016import java.time.format.DateTimeFormatter;
017import java.util.stream.Collectors;
018
019import static javax.servlet.http.HttpServletResponse.*;
020
021@WebFilter("*")
022public class FallbackExceptionHandlerFilter extends HttpFilter {
023    //low priority  todo: fare in modo che tutte le eccezioni non gestite vengano loggate da catalina
024
025    @Override
026    protected void doFilter(HttpServletRequest req, HttpServletResponse resp, FilterChain chain) throws IOException, ServletException {
027        try{
028            chain.doFilter(req, resp);
029        } catch (IllegalArgumentException e) {
030            resp.sendError(SC_BAD_REQUEST, e.getMessage());
031        } catch (ConstraintViolationException e) {
032            String messages = e.getConstraintViolations().stream()
033                    .map(ConstraintViolation::getMessage).collect(Collectors.joining("\n"));
034            resp.sendError(SC_BAD_REQUEST, messages);
035        } catch (AuthenticationRequiredException e) {
036            resp.sendError(SC_UNAUTHORIZED, e.getMessage());
037        } catch (BannedUserException e){
038            if (e.getDuration() != null){
039                String end  = DateTimeFormatter.ISO_INSTANT.format(e.getDuration());
040                resp.sendError(SC_FORBIDDEN, "Sei bannato fino a " + end);
041            } else {
042                resp.sendError(SC_FORBIDDEN, e.getMessage());
043            }
044        } catch (AuthorizationException e) {
045            resp.sendError(SC_FORBIDDEN, e.getMessage());
046        }
047    }
048}
