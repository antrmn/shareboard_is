001package domain.entity;
002
003import lombok.Getter;
004import lombok.Setter;
005
006import javax.persistence.*;
007import java.io.Serializable;
008import java.time.Instant;
009
010@Entity
011public class Ban implements Serializable {
012
013    @Getter @Setter
014    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
015    protected Integer id;
016
017    @Getter @Setter
018    @ManyToOne(optional = false, fetch = FetchType.LAZY)
019    protected User user;
020
021    @Getter
022    @Column(nullable = false, insertable = false, updatable = false)
023    protected Instant startTime; //db generated column
024
025    @Getter @Setter
026    @Column(nullable = true)
027    protected Instant endTime;
028
029    public Ban(){}
030
031    @Override
032    public boolean equals(Object o) {
033        if (this == o) return true;
034        if (!(o instanceof Ban)) return false;
035        Ban ban = (Ban) o;
036        return id.equals(ban.id);
037    }
038
039    @Override
040    public int hashCode() {
041        return getClass().hashCode();
042    }
043}
