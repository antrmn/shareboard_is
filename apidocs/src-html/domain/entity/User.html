001package domain.entity;
002
003import lombok.Getter;
004import lombok.Setter;
005import org.hibernate.annotations.NaturalId;
006
007import javax.persistence.*;
008import java.io.Serializable;
009import java.time.Instant;
010import java.util.ArrayList;
011import java.util.Collections;
012import java.util.List;
013
014@Entity
015public class User implements Serializable {
016
017    @Getter @Setter
018    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
019    protected Integer id;
020
021    @Getter @Setter
022    @NaturalId(mutable = false) @Column(length = 30, unique = true, nullable = false)
023    protected String username;
024
025    @Getter @Setter
026    @Column(length = 16, nullable = false)
027    protected byte[] password;
028
029    @Getter @Setter
030    @Column(length = 16, nullable = false)
031    protected byte[] salt;
032
033    @Getter @Setter
034    @Column(length = 255, unique = true, nullable = false)
035    protected String email;
036
037    @Getter @Setter
038    @Column(length = 255)
039    protected String description;
040
041    @Getter @Setter
042    @Column(length = 4096)
043    protected String picture;
044
045    @Getter
046    @Column(insertable = false, updatable = false, nullable = false)
047    protected Instant creationDate;
048
049    @Getter @Setter
050    @Column(nullable = false)
051    protected Boolean admin;
052
053    @OneToMany(mappedBy = "user")
054    @OrderBy("endTime desc")
055    protected List<Ban> bans = new ArrayList<>();
056    public List<Ban> getBans(){
057        return Collections.unmodifiableList(bans);
058    }
059
060    public User(){}
061
062    @Override
063    public boolean equals(Object o) {
064        if (this == o) return true;
065        if (!(o instanceof User)) return false;
066        User user = (User) o;
067        return id.equals(user.id);
068    }
069
070    @Override
071    public int hashCode() {
072        return getClass().hashCode();
073    }
074}
