001package domain.repository;
002
003import domain.entity.User;
004
005import javax.persistence.EntityManager;
006import javax.persistence.PersistenceContext;
007import java.io.Serializable;
008
009
010public class UserRepository implements Serializable {
011
012    @PersistenceContext
013    protected EntityManager em;
014
015    /**
016     * Restituisce i dati di un utente data un email
017     * @param email email dell'utente
018     * @return entit√† User
019     */
020    public User getByEmail(String email) {
021        return em.createQuery("from User u where u.email=:email", User.class).setParameter("email", email)
022                .getResultList().stream().findFirst().orElse(null);
023    }
024}
