001package domain.repository;
002
003import domain.entity.Comment;
004import domain.entity.Post;
005
006import javax.persistence.EntityManager;
007import javax.persistence.PersistenceContext;
008import java.io.Serializable;
009import java.util.List;
010
011
012public class CommentRepository implements Serializable {
013
014    @PersistenceContext
015    protected EntityManager em;
016
017    /**
018     *  Trova tutti i commenti di un determinato post
019     * @param post entità Post di cui si vuole ottenere i commenti
020     * @param depth profondità della ricorsione per le risposte ai post
021     * @return Lista di commenti
022     */
023    public List<Comment> getByPost(Post post, int depth){
024        return em.createQuery(
025                        "from Comment c where c.post = :post and length(c.path) <= (7 * :depth) + 1 order by c.path",
026                        Comment.class)
027                .setParameter("depth", depth)
028                .setParameter("post", post)
029                .getResultList();
030    }
031
032    /**
033     *  Trova tutti i commenti di riposta a un determinato commento
034     * @param comment Entità Comment di cui si vuole ottenere i commenti
035     * @param depth profondità della ricorsione per le risposte ai post
036     * @return Lista di commenti
037     */
038    public List<Comment> getReplies(Comment comment, int depth){
039        return em.createQuery(
040                    "from Comment c where c.path like :path and length(c.path) <= (7 * :depth) + 1 order by c.path",
041                    Comment.class)
042                .setParameter("depth", depth)
043                .setParameter("path", comment.getPath() + '%')
044                .getResultList();
045    }
046}
