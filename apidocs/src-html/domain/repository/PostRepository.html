001package domain.repository;
002
003import domain.entity.Post;
004import domain.entity.Section;
005import domain.entity.User;
006
007import javax.persistence.EntityManager;
008import javax.persistence.PersistenceContext;
009import javax.persistence.TypedQuery;
010import javax.persistence.criteria.*;
011import java.io.Serializable;
012import java.time.Instant;
013import java.util.ArrayList;
014import java.util.Collections;
015import java.util.List;
016
017public class PostRepository implements Serializable {
018
019    @PersistenceContext
020    protected EntityManager em;
021
022    /**
023     * Restituisce una nuova istanza di PostFinder
024     * @return nuova istanza di PostFinder
025     */
026    public PostFinder getFinder(){
027        return new PostFinder();
028    }
029
030    private enum SortCriteria {OLDEST, NEWEST, MOSTVOTED};
031
032    /**
033     *  Classe interna usata per specificare i parametri di ricerca di un post
034     */
035    public class PostFinder {
036        private User author;
037        private List<Section> sections;
038        private Instant before;
039        private Instant after;
040        private int offset = 0;
041        private int pageSize = 30;
042        private String content;
043        private boolean includeBody = false;
044        private User joinUserFollows;
045        private SortCriteria sortCriteria = PostRepository.SortCriteria.NEWEST;
046
047        protected PostFinder(){ }
048
049        /**
050         * Setta il campo author e restituisce l'istanza passata di PostFinder
051         * @param author entità User dell'autore dei post
052         * @return istanza passata di PostFinder
053         */
054        public PostFinder byAuthor(User author){
055            this.author = author;
056            return this;
057        }
058
059        /**
060         * Setta il campo content e restituisce l'istanza passata di PostFinder
061         * @param author text contenuto dei post
062         * @return istanza passata di PostFinder
063         */
064        public PostFinder byContent(String text){
065            this.content = text;
066            return this;
067        }
068
069        /**
070         * Setta il campo sections e restituisce l'istanza passata di PostFinder
071         * @param sections lista di sezioni
072         * @return istanza passata di PostFinder
073         */
074        public PostFinder bySections(List<Section> sections){
075            this.sections = sections;
076            return this;
077        }
078
079        /**
080         * Setta il campo sections e restituisce l'istanza passata di PostFinder
081         * @param sections sezione dei post
082         * @return istanza passata di PostFinder
083         */
084        public PostFinder bySection(Section section){
085            this.sections = Collections.singletonList(section);
086            return this;
087        }
088
089        /**
090         * Setta il campo postedAfter e restituisce l'istanza passata di PostFinder
091         * @param after data dopo la quale i post da trovare sono stati postati
092         * @return istanza passata di PostFinder
093         */
094        public PostFinder postedAfter(Instant after){
095            this.after = after;
096            return this;
097        }
098
099        /**
100         * Setta il campo postedBefore e restituisce l'istanza passata di PostFinder
101         * @param before data prima della quale i post da trovare sono stati postati
102         * @return istanza passata di PostFinder
103         */
104        public PostFinder postedBefore(Instant before){
105            this.before = before;
106            return this;
107        }
108
109        /**
110         * Setta il campo offset e restituisce l'istanza passata di PostFinder
111         * @param n offset per la paginazione
112         * @return istanza passata di PostFinder
113         */
114        public PostFinder offset(int n){
115            offset = n;
116            return this;
117        }
118
119        /**
120         * Setta il campo limit e restituisce l'istanza passata di PostFinder
121         * @param n limite di post da caricare
122         * @return istanza passata di PostFinder
123         */
124        public PostFinder limit(int n){
125            pageSize = n;
126            return this;
127        }
128
129        /**
130         * Setta il campo sortCriteria a oldest e restituisce l'istanza passata di PostFinder
131         * @return istanza passata di PostFinder
132         */
133        public PostFinder getOldest(){
134            sortCriteria = SortCriteria.OLDEST;
135            return this;
136        }
137
138        /**
139         * Setta il campo sortCriteria a newest e restituisce l'istanza passata di PostFinder
140         * @return istanza passata di PostFinder
141         */
142        public PostFinder getNewest(){
143            sortCriteria = SortCriteria.NEWEST;
144            return this;
145        }
146
147        /**
148         * Setta il campo sortCriteria a most voted e restituisce l'istanza passata di PostFinder
149         * @return istanza passata di PostFinder
150         */
151        public PostFinder getMostVoted(){
152            sortCriteria = SortCriteria.MOSTVOTED;
153            return this;
154        }
155
156        /**
157         * Setta il campo includeBody a true e restituisce l'istanza passata di PostFinder
158         * @return istanza passata di PostFinder
159         */
160        public PostFinder includeBody(){
161            includeBody = true;
162            return this;
163        }
164
165        /**
166         * Setta il campo joinUserFollows e restituisce l'istanza passata di PostFinder
167         * @param user entita utente da cui ottenere le sezioni seguite
168         * @return istanza passata di PostFinder
169         */
170        public PostFinder joinUserFollows(User user){
171            joinUserFollows = user;
172            return this;
173        }
174
175        /**
176         * Restituisce tutti i post che rispettano i criteri di ricerca
177         * @return lista di entità Post
178         */
179        public List<Post> getResults(){
180            CriteriaBuilder cb = em.getCriteriaBuilder();
181            CriteriaQuery<Post> cq = cb.createQuery(Post.class);
182            Root<Post> root = cq.from(Post.class);
183
184            List<Predicate> predicates = new ArrayList<>();
185
186            if(author != null) {
187                predicates.add(cb.equal(root.get("author"), author));
188            }
189
190            if(sections != null && !sections.isEmpty()) {
191                predicates.add(root.get("section").in(sections));
192            }
193
194            if(after != null) {
195                predicates.add(cb.greaterThanOrEqualTo(root.get("creationDate"), after));
196            }
197
198            if(before != null) {
199                predicates.add(cb.lessThanOrEqualTo(root.get("creationDate"), before));
200            }
201
202            if(content != null) {
203                if(includeBody) {
204                    predicates.add(
205                            cb.or(
206                                    cb.like(root.get("title"), '%' + content + '%'),
207                                    cb.and(
208                                            cb.notEqual(root.get("type"), Post.Type.IMG),
209                                            cb.like(root.get("content"), '%' + content + '%')
210                                    )));
211                } else {
212                    predicates.add(cb.like(root.get("title"), '%' + content + '%'));
213                }
214            }
215
216            if(joinUserFollows != null){
217                Join<Object, Object> joinFollow = root.join("section").join("follows");
218                predicates.add(cb.equal(joinFollow.get("user"), joinUserFollows));
219            }
220
221            cq.where(cb.and(predicates.toArray(Predicate[]::new)));
222
223            switch (sortCriteria) {
224                case MOSTVOTED:
225                    cq.orderBy(cb.desc(root.get("votesCount")));
226                    break;
227                case OLDEST:
228                    cq.orderBy(cb.asc(root.get("creationDate")));
229                    break;
230                case NEWEST:
231                default:
232                    cq.orderBy(cb.desc(root.get("creationDate")));
233                    break;
234            }
235
236
237            TypedQuery<Post> tq = em.createQuery(cq);
238            if(offset >= 0)
239                tq.setFirstResult(offset);
240            if(pageSize>=0)
241                tq.setMaxResults(pageSize);
242
243            return tq.getResultList();
244        }
245    }
246}
