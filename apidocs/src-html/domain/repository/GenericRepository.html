001package domain.repository;
002
003import org.hibernate.Session;
004import org.hibernate.SimpleNaturalIdLoadAccess;
005
006import javax.persistence.EntityManager;
007import javax.persistence.LockModeType;
008import javax.persistence.PersistenceContext;
009import javax.persistence.criteria.CriteriaQuery;
010import java.io.Serializable;
011import java.util.List;
012
013/**
014 * Interfaccia usata per interagire con {@link javax.persistence.EntityManager}
015 * Istanze che implementano questa interfaccia fungono da facciata alle funzionalità di JPA, fornendo i metodi
016 * essenziali (CRUD) per una specifica entità di persistenza
017 * @see javax.persistence.EntityManager
018 */
019public class GenericRepository implements Serializable {
020
021    @PersistenceContext
022    protected EntityManager em;
023
024    /**
025     *  Trova per chiave primaria
026     * @return L'entità trovata o null se l'entità non esiste
027     * @throws IllegalArgumentException se la chiave primaria fornita è null o non valida per l'entità da cercare
028     * @see javax.persistence.EntityManager#find(Class, Object)
029     */
030    public <T> T findById(Class<T> entityClass, Object primaryKey, boolean loadLazily) {
031        return loadLazily ? em.getReference(entityClass, primaryKey) :
032                            em.find(entityClass, primaryKey, LockModeType.NONE);
033    }
034
035    /**
036     * Trova per chiave primaria, restituendo un'entità caricata pigramente (lazy-loaded). L'entità restituita viene
037     * caricata al primo accesso effettuato. Se l'istanza non esiste, viene lanciato {@link javax.persistence.EntityNotFoundException}
038     * quando l'istanza viene acceduta per la prima volta.
039     * @return L'entità caricata pigramente
040     * @throws IllegalArgumentException se la chiave primaria fornita è null o non valida per l'entità da cercare
041     * @throws javax.persistence.EntityNotFoundException se l'entità non può essere acceduta
042     * @see javax.persistence.EntityManager#getReference(Class, Object)
043     */
044    public <T> T findById(Class<T> entityClass, Object primaryKey){
045        return findById(entityClass,primaryKey,false);
046    }
047
048    public <T> T findByNaturalId(Class<T> entityClass, Object naturalKey) {
049        return findByNaturalId(entityClass,naturalKey,false);
050    }
051
052    public <T> T findByNaturalId(Class<T> entityClass, Object naturalKey, boolean getReference) {
053        SimpleNaturalIdLoadAccess<T> entity = em.unwrap(Session.class).bySimpleNaturalId(entityClass);
054        return getReference ? entity.getReference(naturalKey) : entity.load(naturalKey);
055    }
056
057    /**
058     * Restituisce una lista tipata contenente tutte le istanze di una determinata entità di persistenza
059     * @return La lista dei risultati
060     * @throws javax.persistence.QueryTimeoutException In caso di timeout raggiunto
061     * @throws javax.persistence.PersistenceException
062     */
063    public <T> List<T> findAll(Class<T> entityClass) {
064        CriteriaQuery<T> c = em.getCriteriaBuilder().createQuery(entityClass);
065        c.select(c.from(entityClass));
066        return em.createQuery(c).getResultList();
067    }
068
069    /**
070     * Restituisce il numero indicante la quantità di istanze di una determinata entità di persistenza
071     * @return La quantità di istanze
072     * @throws javax.persistence.QueryTimeoutException In caso di timeout raggiunto
073     * @throws javax.persistence.PersistenceException
074     */
075    public <T> Long getCount(Class<T> entityClass) {
076        CriteriaQuery<Long> c =
077                em.getCriteriaBuilder().createQuery(Long.class);
078        c.select(em.getCriteriaBuilder().count(c.from(entityClass)));
079        return em.createQuery(c).getSingleResult();
080    }
081
082    /**
083     * Crea un'istanza persistente e gestita (managed entity)
084     * @param entity L'istanza di entità
085     * @return L'istanza passata come parametro
086     * @throws javax.persistence.EntityExistsException se l'entità esiste già
087     * @see javax.persistence.EntityManager#persist(Object)
088     */
089    public <T> T insert(T entity) {
090        em.persist(entity);
091        em.flush();
092        return entity;
093    }
094
095    /**
096     * Unisce lo stato di una data entità allo stato nel contesto di persistenza
097     * @param entity
098     * @return Una copia <i>managed</i> dell'entità passata come parametro
099     * @see javax.persistence.EntityManager#merge(Object)
100     */
101    public <T> T merge(T entity) {
102        return em.merge(entity);
103    }
104
105    /**
106     * Rimuove l'istanza dal contesto di persistenza
107     * @param entity L'entità managed da rimuovere
108     * @throws IllegalArgumentException se l'entità passata non è managed
109     */
110    public <T> void remove(T entity) {
111        em.remove(entity);
112    }
113}
