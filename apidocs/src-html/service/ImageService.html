001package service;
002
003import persistence.repo.BinaryContentRepository;
004
005import javax.enterprise.context.ApplicationScoped;
006import javax.inject.Inject;
007import javax.validation.constraints.NotBlank;
008import java.io.BufferedInputStream;
009import java.io.IOException;
010import java.io.InputStream;
011import java.net.URLConnection;
012
013@ApplicationScoped
014public class ImageService {
015    @Inject private BinaryContentRepository binaryContentRepository;
016
017    public InputStream getImage(@NotBlank String filename) throws IOException {
018        InputStream inputStream = binaryContentRepository.get(filename);
019        if(inputStream == null)
020            return null;
021
022        inputStream = new BufferedInputStream(inputStream);
023        String mimetype = URLConnection.guessContentTypeFromStream(inputStream);
024        if(mimetype == null || !mimetype.startsWith("image/"))
025            return  null;
026
027        return inputStream;
028    }
029}
