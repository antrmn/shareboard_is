001package service;
002
003import persistence.repo.BinaryContentRepository;
004
005import javax.enterprise.context.ApplicationScoped;
006import javax.inject.Inject;
007import javax.validation.constraints.NotBlank;
008import java.io.BufferedInputStream;
009import java.io.IOException;
010import java.io.InputStream;
011import java.net.URLConnection;
012
013@ApplicationScoped
014public class ImageService {
015    private BinaryContentRepository binaryContentRepository;
016
017    protected ImageService(){}
018
019    @Inject
020    protected ImageService(BinaryContentRepository binaryContentRepository){
021        this.binaryContentRepository = binaryContentRepository;
022    }
023
024    /**
025     * Ritorna l'input stream di un immagine
026     * @param filename nome dell'immagine
027     * @return input stream dell'immagine
028     * @throws IOException
029     */
030    public InputStream getImage(@NotBlank String filename) throws IOException {
031        InputStream inputStream = binaryContentRepository.get(filename);
032        if(inputStream == null)
033            return null;
034
035        inputStream = new BufferedInputStream(inputStream);
036        String mimetype = URLConnection.guessContentTypeFromStream(inputStream);
037        if(mimetype == null || !mimetype.startsWith("image/"))
038            return  null;
039
040        return inputStream;
041    }
042}
