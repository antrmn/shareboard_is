001package service;
002
003import persistence.model.Section;
004import persistence.model.User;
005import persistence.repo.BinaryContentRepository;
006import persistence.repo.GenericRepository;
007import persistence.repo.SectionRepository;
008import service.auth.AdminsOnly;
009import service.dto.CurrentUser;
010import service.dto.SectionPage;
011import service.validation.Image;
012import service.validation.SectionExists;
013import service.validation.SectionExistsById;
014
015import javax.enterprise.context.ApplicationScoped;
016import javax.enterprise.context.RequestScoped;
017import javax.enterprise.inject.Produces;
018import javax.inject.Inject;
019import javax.inject.Named;
020import javax.transaction.Transactional;
021import java.io.BufferedInputStream;
022import java.io.IOException;
023import java.time.Instant;
024import java.time.temporal.ChronoUnit;
025import java.util.List;
026import java.util.Map;
027import java.util.stream.Collectors;
028
029@ApplicationScoped
030@Transactional
031public class SectionService {
032
033    @Inject private SectionRepository sectionRepo;
034    @Inject private GenericRepository genericRepository;
035    @Inject private BinaryContentRepository bcRepo;
036    @Inject private CurrentUser currentUser;
037
038    /**
039     * Converte section in SectionPage.
040     * @param section sezione da convertire
041     * @return SectionPage con i dati di section
042     */
043    private SectionPage map(Section section){
044        User user = null;
045        if(currentUser.isLoggedIn()){
046            user = genericRepository.findById(User.class,currentUser.getId());
047        }
048        return SectionPage.builder().id(section.getId())
049                .name(section.getName())
050                .picture(section.getPicture())
051                .description(section.getDescription())
052                .banner(section.getBanner())
053                .nFollowers(section.getFollowCount())
054                .isFollowed(user != null && section.getFollow(user) != null).build();
055    }
056
057    /**
058     * Cancella una sezione dato il suo id
059     * @param id l'id di una sezione esistente
060     */
061    @AdminsOnly
062    public void delete(@SectionExistsById int id){
063        genericRepository.remove(genericRepository.findById(Section.class, id));
064    }
065
066    /**
067     * Crea una nuova sezione e ne restituisce l'id
068     * @param sectiondata entità contenente i dati della sezione
069     * @param picture stream relativo alla foto della sezione
070     * @param banner stream relativo al banner della sezione
071     * @throws IOException
072     * @return id della sezione creata
073     */
074    @AdminsOnly
075    public int newSection(SectionPage sectiondata,
076                          @Image BufferedInputStream picture,
077                          @Image BufferedInputStream banner) throws IOException {
078        Section s = new Section();
079        s.setName(sectiondata.getName());
080        s.setDescription(sectiondata.getDescription());
081        if (picture != null){
082            s.setPicture(bcRepo.insert(picture));
083        }
084        if (banner != null) {
085            s.setBanner(bcRepo.insert(banner));
086        }
087
088        return genericRepository.insert(s).getId();
089    }
090
091    /**
092     * Ritorna una lista di tutte le sezioni esistenti
093     * @return lista di sezioni
094     */
095    public List<SectionPage> showSections(){
096        return genericRepository.findAll(Section.class).stream().map(this::map).collect(Collectors.toList());
097    }
098
099    /**
100     * Ritorna una mappa di tutte le sezioni esistenti
101     * @return mappa la cui chiave è l'id della sezione e il valore un entita PostPage che ne contiene i dati
102     */
103    @Named("sections")
104    @RequestScoped
105    @Produces
106    public Map<Integer,SectionPage> getSectionsMap(){
107        return genericRepository.findAll(Section.class).stream().map(this::map)
108                .collect(Collectors.toMap(SectionPage::getId, section -> section));
109    }
110
111    /**
112     * Ritorna un entità sezione dato un certo id
113     * @param id id di una sezione esistente
114     * @return entita SectionPage che contiene i dati della sezione
115     */
116    public SectionPage showSection(@SectionExistsById int id){
117       Section s =  genericRepository.findById(Section.class, id);
118       return map(s);
119    }
120
121    /**
122     * Ritorna un entità sezione con un nome specifico
123     * @param sectionName nome di una sezione esistente
124     * @return entita SectionPage che contiene i dati della sezione
125     */
126    public SectionPage getSection(@SectionExists String sectionName){
127        Section s = sectionRepo.getByName(sectionName);
128        return map(s);
129    }
130
131    /**
132     * Modifica i dati di un entità sezione con un id specifico
133     * @param edit nuovi dati della sezione
134     * @param id id di una sezione esistente
135     * @param picture stream contenente i dati della foto di sezione
136     * @param banner stream contenente i dati del banner di sezione
137     * @throws IOException
138     * @return entita SectionPage che contiene i dati della sezione
139     */
140    @AdminsOnly
141    public void editSection(SectionPage edit,
142                            @SectionExistsById int id,
143                            @Image BufferedInputStream picture,
144                            @Image BufferedInputStream banner) throws IOException {
145        Section s = genericRepository.findById(Section.class,id);
146//        s.setBanner(edit.getBanner());
147//        s.setPicture(edit.getPicture());
148        s.setDescription(edit.getDescription());
149        if (picture != null){
150            s.setPicture(bcRepo.insert(picture));
151        }
152        if (banner != null) {
153            s.setBanner(bcRepo.insert(banner));
154        }
155    }
156
157    /**
158     * Ritorna una lista delle sezioni con più follows
159     * @return lista di sezioni
160     */
161    @Produces
162    @RequestScoped
163    @Named("topSections")
164    public List<SectionPage> getTopSections(){
165        return sectionRepo.getMostFollowedSections().stream().map(this::map).collect(Collectors.toList());
166    }
167
168    /**
169     * Ritorna una lista delle sezioni con più follows negli ultimi 7 giorni
170     * @return lista di sezioni
171     */
172    @Produces
173    @RequestScoped
174    @Named("trendingSections")
175    public List<SectionPage> getTrendingSections(){
176        return sectionRepo.getMostFollowedSections(Instant.now().minus(7, ChronoUnit.DAYS))
177                .stream().map(this::map).collect(Collectors.toList());
178    }
179}
