001package service;
002
003import persistence.model.Section;
004import persistence.model.User;
005import persistence.repo.BinaryContentRepository;
006import persistence.repo.GenericRepository;
007import persistence.repo.SectionRepository;
008import service.auth.AdminsOnly;
009import service.dto.CurrentUser;
010import service.dto.SectionPage;
011import service.validation.SectionExists;
012import service.validation.SectionExistsById;
013
014import javax.enterprise.context.ApplicationScoped;
015import javax.enterprise.context.RequestScoped;
016import javax.enterprise.inject.Produces;
017import javax.inject.Inject;
018import javax.inject.Named;
019import javax.transaction.Transactional;
020import java.io.BufferedInputStream;
021import java.io.IOException;
022import java.time.Instant;
023import java.time.temporal.ChronoUnit;
024import java.util.List;
025import java.util.Map;
026import java.util.stream.Collectors;
027
028@ApplicationScoped
029@Transactional
030public class SectionService {
031
032    @Inject private SectionRepository sectionRepo;
033    @Inject private GenericRepository genericRepository;
034    @Inject private BinaryContentRepository bcRepo;
035    @Inject private CurrentUser currentUser;
036
037    /**
038     * Converte section in SectionPage.
039     * @param section sezione da convertire
040     * @return SectionPage con i dati di section
041     */
042    private SectionPage map(Section section){
043        User user = null;
044        if(currentUser.isLoggedIn()){
045            user = genericRepository.findById(User.class,currentUser.getId());
046        }
047        return SectionPage.builder().id(section.getId())
048                .name(section.getName())
049                .picture(section.getPicture())
050                .description(section.getDescription())
051                .banner(section.getBanner())
052                .nFollowers(section.getFollowCount())
053                .isFollowed(user != null && section.getFollow(user) != null).build();
054    }
055
056    /**
057     * Cancella una sezione dato il suo id
058     * @param id l'id di una sezione esistente
059     */
060    @AdminsOnly
061    public void delete(@SectionExistsById int id){
062        genericRepository.remove(genericRepository.findById(Section.class, id));
063    }
064
065    /**
066     * Crea una nuova sezione e ne restituisce l'id
067     * @param sectiondata entità contenente i dati della sezione
068     * @param picture stream relativo alla foto della sezione
069     * @param banner stream relativo al banner della sezione
070     * @throws IOException
071     * @return id della sezione creata
072     */
073    @AdminsOnly
074    public int newSection(SectionPage sectiondata,
075                          BufferedInputStream picture,
076                          BufferedInputStream banner) throws IOException {
077        Section s = new Section();
078        s.setName(sectiondata.getName());
079        s.setPicture(sectiondata.getPicture());
080        s.setBanner(sectiondata.getBanner());
081        s.setDescription(sectiondata.getDescription());
082        bcRepo.insert(picture);
083        bcRepo.insert(banner);
084
085        return genericRepository.insert(s).getId();
086    }
087
088    /**
089     * Ritorna una lista di tutte le sezioni esistenti
090     * @return lista di sezioni
091     */
092    public List<SectionPage> showSections(){
093        return genericRepository.findAll(Section.class).stream().map(this::map).collect(Collectors.toList());
094    }
095
096    /**
097     * Ritorna una mappa di tutte le sezioni esistenti
098     * @return mappa la cui chiave è l'id della sezione e il valore un entita PostPage che ne contiene i dati
099     */
100    @Named("sections")
101    @RequestScoped
102    @Produces
103    public Map<Integer,SectionPage> getSectionsMap(){
104        return genericRepository.findAll(Section.class).stream().map(this::map)
105                .collect(Collectors.toMap(SectionPage::getId, section -> section));
106    }
107
108    /**
109     * Ritorna un entità sezione dato un certo id
110     * @param id id di una sezione esistente
111     * @return entita SectionPage che contiene i dati della sezione
112     */
113    public SectionPage showSection(@SectionExistsById int id){
114       Section s =  genericRepository.findById(Section.class, id);
115       return map(s);
116    }
117
118    /**
119     * Ritorna un entità sezione con un nome specifico
120     * @param sectionName nome di una sezione esistente
121     * @return entita SectionPage che contiene i dati della sezione
122     */
123    public SectionPage getSection(@SectionExists String sectionName){
124        Section s = sectionRepo.getByName(sectionName);
125        return map(s);
126    }
127
128    /**
129     * Modifica i dati di un entità sezione con un id specifico
130     * @param edit nuovi dati della sezione
131     * @param id id di una sezione esistente
132     * @param picture stream contenente i dati della foto di sezione
133     * @param banner stream contenente i dati del banner di sezione
134     * @throws IOException
135     * @return entita SectionPage che contiene i dati della sezione
136     */
137    @AdminsOnly
138    public void editSection(SectionPage edit,
139                            @SectionExistsById int id,
140                            BufferedInputStream picture,
141                            BufferedInputStream banner) throws IOException {
142        Section s = genericRepository.findById(Section.class,id);
143        s.setBanner(edit.getBanner());
144        s.setPicture(edit.getPicture());
145        s.setDescription(edit.getDescription());
146        bcRepo.insert(picture);
147        bcRepo.insert(banner);
148    }
149
150    /**
151     * Ritorna una lista delle sezioni con più follows
152     * @return lista di sezioni
153     */
154    @Produces
155    @RequestScoped
156    @Named("topSections")
157    public List<SectionPage> getTopSections(){
158        return sectionRepo.getMostFollowedSections().stream().map(this::map).collect(Collectors.toList());
159    }
160
161    /**
162     * Ritorna una lista delle sezioni con più follows negli ultimi 7 giorni
163     * @return lista di sezioni
164     */
165    @Produces
166    @RequestScoped
167    @Named("trendingSections")
168    public List<SectionPage> getTrendingSections(){
169        return sectionRepo.getMostFollowedSections(Instant.now().minus(7, ChronoUnit.DAYS))
170                .stream().map(this::map).collect(Collectors.toList());
171    }
172}
