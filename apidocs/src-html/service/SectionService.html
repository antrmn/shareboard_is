001package service;
002
003import persistence.model.Section;
004import persistence.model.User;
005import persistence.repo.BinaryContentRepository;
006import persistence.repo.GenericRepository;
007import persistence.repo.SectionRepository;
008import service.auth.AdminsOnly;
009import service.dto.CurrentUser;
010import service.dto.SectionPage;
011import service.validation.SectionExists;
012import service.validation.SectionExistsById;
013
014import javax.enterprise.context.ApplicationScoped;
015import javax.enterprise.context.RequestScoped;
016import javax.enterprise.inject.Produces;
017import javax.inject.Inject;
018import javax.inject.Named;
019import javax.transaction.Transactional;
020import java.io.BufferedInputStream;
021import java.io.IOException;
022import java.time.Instant;
023import java.time.temporal.ChronoUnit;
024import java.util.List;
025import java.util.Map;
026import java.util.stream.Collectors;
027
028@ApplicationScoped
029@Transactional
030public class SectionService {
031
032    @Inject private SectionRepository sectionRepo;
033    @Inject private GenericRepository genericRepository;
034    @Inject private BinaryContentRepository bcRepo;
035    @Inject private CurrentUser currentUser;
036
037    private SectionPage map(Section section){
038        User user = null;
039        if(currentUser.isLoggedIn()){
040            user = genericRepository.findById(User.class,currentUser.getId());
041        }
042        return SectionPage.builder().id(section.getId())
043                .name(section.getName())
044                .picture(section.getPicture())
045                .description(section.getDescription())
046                .banner(section.getBanner())
047                .nFollowers(section.getFollowCount())
048                .isFollowed(user != null && section.getFollow(user) != null).build();
049    }
050
051    @AdminsOnly
052    public void delete(@SectionExistsById int id){
053        genericRepository.remove(genericRepository.findById(Section.class, id));
054    }
055
056    @AdminsOnly
057    public int newSection(SectionPage sectiondata,
058                          BufferedInputStream picture,
059                          BufferedInputStream banner) throws IOException {
060        Section s = new Section();
061        s.setName(sectiondata.getName());
062        s.setPicture(sectiondata.getPicture());
063        s.setBanner(sectiondata.getBanner());
064        s.setDescription(sectiondata.getDescription());
065        bcRepo.insert(picture);
066        bcRepo.insert(banner);
067
068        return genericRepository.insert(s).getId();
069    }
070
071    public List<SectionPage> showSections(){
072        return genericRepository.findAll(Section.class).stream().map(this::map).collect(Collectors.toList());
073    }
074
075    @Named("sections")
076    @RequestScoped
077    @Produces
078    public Map<Integer,SectionPage> getSectionsMap(){
079        return genericRepository.findAll(Section.class).stream().map(this::map)
080                .collect(Collectors.toMap(SectionPage::getId, section -> section));
081    }
082
083    public SectionPage showSection(@SectionExistsById int id){
084       Section s =  genericRepository.findById(Section.class, id);
085       return map(s);
086    }
087
088    public SectionPage getSection(@SectionExists String sectionName){
089        Section s = sectionRepo.getByName(sectionName);
090        return map(s);
091    }
092
093    @AdminsOnly
094    public void editSection(SectionPage edit,
095                            @SectionExistsById int id,
096                            BufferedInputStream picture,
097                            BufferedInputStream banner) throws IOException {
098        Section s = genericRepository.findById(Section.class,id);
099        s.setBanner(edit.getBanner());
100        s.setPicture(edit.getPicture());
101        s.setDescription(edit.getDescription());
102        bcRepo.insert(picture);
103        bcRepo.insert(banner);
104    }
105
106    @Produces
107    @RequestScoped
108    @Named("topSections")
109    public List<SectionPage> getTopSections(){
110        return sectionRepo.getMostFollowedSections().stream().map(this::map).collect(Collectors.toList());
111    }
112
113    @Produces
114    @RequestScoped
115    @Named("trendingSections")
116    public List<SectionPage> getTrendingSections(){
117        return sectionRepo.getMostFollowedSections(Instant.now().minus(7, ChronoUnit.DAYS))
118                .stream().map(this::map).collect(Collectors.toList());
119    }
120}
