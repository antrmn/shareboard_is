001package service.validation;
002
003import persistence.repo.UserRepository;
004
005import javax.inject.Inject;
006import javax.persistence.NoResultException;
007import javax.validation.ConstraintValidator;
008import javax.validation.ConstraintValidatorContext;
009import javax.validation.ValidationException;
010
011public class UniqueEmailValidator implements ConstraintValidator<UniqueEmail, String> {
012    @Inject
013    UserRepository userRepository;
014
015
016    @Override
017    public void initialize(UniqueEmail userExists) {
018
019    }
020
021    @Override
022    public boolean isValid(String email, ConstraintValidatorContext constraintValidatorContext) {
023        boolean check;
024        try{
025            check = userRepository.getByEmail(email) == null;
026        }catch (NoResultException e){
027            check = true;
028        }
029        return check;
030    }
031}
