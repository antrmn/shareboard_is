001package service.validation;
002
003import javax.validation.Constraint;
004import javax.validation.Payload;
005import javax.validation.constraints.Pattern;
006import javax.validation.constraints.Size;
007import java.lang.annotation.*;
008
009@Pattern(regexp = "^[\\w\\-]+$")
010@Size(min=3, max=30)
011@Target({ElementType.FIELD, ElementType.METHOD, ElementType.PARAMETER,
012        ElementType.ANNOTATION_TYPE})
013@Retention(RetentionPolicy.RUNTIME)
014@Constraint(validatedBy = {})
015@Documented
016public @interface UsernameFormat {
017    String message() default "Formato username non valido";
018
019    Class<?>[] groups() default {};
020
021    Class<? extends Payload>[] payload() default {};
022}
