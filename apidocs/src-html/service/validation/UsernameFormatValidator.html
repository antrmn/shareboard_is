001package service.validation;
002
003import javax.validation.ConstraintValidator;
004import javax.validation.ConstraintValidatorContext;
005
006public class UsernameFormatValidator implements ConstraintValidator<UsernameFormat, String> {
007
008    @Override
009    public void initialize(UsernameFormat userExists) {
010
011    }
012
013    @Override
014    public boolean isValid(String username, ConstraintValidatorContext constraintValidatorContext) {
015        return (username.length() <= 30 && username.length() >= 3) && username.matches("^[\\w\\-]+$");
016    }
017}
