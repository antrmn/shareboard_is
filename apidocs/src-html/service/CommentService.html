001package service;
002
003import persistence.model.Comment;
004import persistence.model.CommentVote;
005import persistence.model.Post;
006import persistence.model.User;
007import persistence.repo.CommentRepository;
008import persistence.repo.GenericRepository;
009import service.auth.AuthenticationRequired;
010import service.auth.DenyBannedUsers;
011import service.dto.CommentDTO;
012import service.dto.CurrentUser;
013import service.validation.CommentExists;
014import service.validation.PostExists;
015
016import javax.enterprise.context.ApplicationScoped;
017import javax.inject.Inject;
018import javax.transaction.Transactional;
019import javax.validation.constraints.NotBlank;
020import javax.validation.constraints.Size;
021import java.time.Instant;
022import java.util.List;
023import java.util.Map;
024
025import static java.util.stream.Collectors.groupingBy;
026import static java.util.stream.Collectors.toList;
027
028
029@ApplicationScoped
030@Transactional
031public class CommentService {
032    @Inject private GenericRepository genericRepository;
033    @Inject private CommentRepository commentRepo;
034    @Inject private CurrentUser currentUser;
035
036    private static final int MAX_COMMENT_DEPTH = 4;
037
038    private CommentDTO map (Comment comment){
039        CommentVote commentVote = null;
040        if(currentUser.isLoggedIn()){
041            User user = genericRepository.findById(User.class, currentUser.getId());
042            commentVote = comment.getVote(user);
043        }
044
045        return CommentDTO.builder()
046                .id(comment.getId())
047                .authorUsername(comment.getAuthor().getUsername())
048                .authorId(comment.getAuthor().getId())
049                .creationDate(comment.getCreationDate())
050                .postId(comment.getPost().getId())
051                .vote(commentVote == null ? 0 : commentVote.getVote())
052                .content(comment.getContent())
053                .votes(comment.getVotesCount() == null ? 0 : comment.getVotesCount())
054                .creationDate(comment.getCreationDate()  == null ? Instant.now() : comment.getCreationDate())
055                .parentCommentId(comment.getParentComment() == null ? 0 : comment.getParentComment().getId())
056                .build();
057    }
058
059    public Map<Integer,List<CommentDTO>> getPostComments(@PostExists int postId){
060        List<Comment> comments = commentRepo.getByPost(genericRepository.findById(Post.class, postId), MAX_COMMENT_DEPTH);
061        return comments.stream().map(this::map).collect(groupingBy(CommentDTO::getParentCommentId, toList()));
062    }
063
064    public Map<Integer, List<CommentDTO>> getReplies(@CommentExists int commentId){
065        List<Comment> comments = commentRepo.getReplies(genericRepository.findById(Comment.class, commentId), MAX_COMMENT_DEPTH);
066        return comments.stream().map(this::map).collect(groupingBy(CommentDTO::getParentCommentId, toList()));
067    }
068
069    public CommentDTO getComment(@CommentExists int id){
070        return map(genericRepository.findById(Comment.class, id));
071    }
072
073    @AuthenticationRequired
074    public void delete(@CommentExists int id){
075        genericRepository.remove(genericRepository.findById(Comment.class, id));
076    }
077
078    @AuthenticationRequired
079    public void editComment(@CommentExists int id, String text){
080        genericRepository.findById(Comment.class, id).setContent(text);
081    }
082
083    @AuthenticationRequired
084    @DenyBannedUsers
085    public int newComment(@NotBlank @Size String text,
086                       @PostExists int postId){
087        User user = genericRepository.findById(User.class, currentUser.getId());
088
089        Comment comment = new Comment();
090        comment.setAuthor(user);
091        comment.setContent(text);
092        comment.setPost(genericRepository.findById(Post.class, postId));
093        return genericRepository.insert(comment).getId();
094    }
095
096    @AuthenticationRequired
097    public int newCommentReply(@NotBlank @Size String text,
098                               @CommentExists int parentId,
099                               @PostExists int postId){
100        User user = genericRepository.findById(User.class, currentUser.getId());
101        Comment parent = genericRepository.findById(Comment.class, parentId);
102
103        Comment comment = new Comment();
104        comment.setAuthor(user);
105        comment.setContent(text);
106        comment.setPost(genericRepository.findById(Post.class, postId));
107        comment.setParentComment(parent);
108        return genericRepository.insert(comment).getId();
109    }
110
111}
