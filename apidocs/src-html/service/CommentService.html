001package service;
002
003import persistence.model.Comment;
004import persistence.model.CommentVote;
005import persistence.model.Post;
006import persistence.model.User;
007import persistence.repo.CommentRepository;
008import persistence.repo.GenericRepository;
009import service.auth.AuthenticationRequired;
010import service.auth.AuthorizationException;
011import service.auth.DenyBannedUsers;
012import service.dto.CommentDTO;
013import service.dto.CurrentUser;
014import service.validation.CommentExists;
015import service.validation.PostExists;
016
017import javax.enterprise.context.ApplicationScoped;
018import javax.inject.Inject;
019import javax.transaction.Transactional;
020import javax.validation.constraints.NotBlank;
021import javax.validation.constraints.Size;
022import java.time.Instant;
023import java.util.List;
024import java.util.Map;
025
026import static java.util.stream.Collectors.groupingBy;
027import static java.util.stream.Collectors.toList;
028
029
030@ApplicationScoped
031@Transactional
032public class CommentService {
033    private GenericRepository genericRepository;
034    private CommentRepository commentRepo;
035    private CurrentUser currentUser;
036
037    private static final int MAX_COMMENT_DEPTH = 4;
038
039    protected CommentService(){}
040
041    @Inject
042    protected CommentService(GenericRepository genericRepository, CommentRepository commentRepository,
043                                            CurrentUser currentUser){
044        this.genericRepository = genericRepository;
045        this.commentRepo = commentRepository;
046        this.currentUser = currentUser;
047    }
048
049    /**
050     * Converte Comment in CommentDTO.
051     * @param comment commento da convertire
052     * @return commentDTO con i dati di comment
053     */
054    private CommentDTO map (Comment comment){
055        CommentVote commentVote = null;
056        if(currentUser.isLoggedIn()){
057            User user = genericRepository.findById(User.class, currentUser.getId());
058            commentVote = comment.getVote(user);
059        }
060
061        return CommentDTO.builder()
062                .id(comment.getId())
063                .authorUsername(comment.getAuthor().getUsername())
064                .authorId(comment.getAuthor().getId())
065                .creationDate(comment.getCreationDate())
066                .postId(comment.getPost().getId())
067                .vote(commentVote == null ? 0 : commentVote.getVote())
068                .content(comment.getContent())
069                .votes(comment.getVotesCount() == null ? 0 : comment.getVotesCount())
070                .creationDate(comment.getCreationDate()  == null ? Instant.now() : comment.getCreationDate())
071                .parentCommentId(comment.getParentComment() == null ? 0 : comment.getParentComment().getId())
072                .build();
073    }
074
075    /**
076     * Ritorna una mappa la cui chiave è l'id del commento padre e il valore una lista di CommentDTO
077     * @param postId l'id di un post esistente di cui si vuole ottenere i commenti
078     * @return mappa con i commenti del post
079     */
080    public Map<Integer,List<CommentDTO>> getPostComments(@PostExists int postId){
081        List<Comment> comments = commentRepo.getByPost(genericRepository.findById(Post.class, postId), MAX_COMMENT_DEPTH);
082        return comments.stream().map(this::map).collect(groupingBy(CommentDTO::getParentCommentId, toList()));
083    }
084
085    /**
086     * Ritorna una mappa la cui chiave è l'id del commento padre e il valore una lista di CommentDTO
087     * @param commentId l'id di un commento esistente di cui si vuole ottenere le risposte
088     * @return mappa con le risposte al commento
089     */
090    public Map<Integer, List<CommentDTO>> getReplies(@CommentExists int commentId){
091        List<Comment> comments = commentRepo.getReplies(genericRepository.findById(Comment.class, commentId), MAX_COMMENT_DEPTH);
092        return comments.stream().map(this::map).collect(groupingBy(CommentDTO::getParentCommentId, toList()));
093    }
094
095    /**
096     * Ritorna un commento dato il suo id
097     * @param id l'id di un commento esistente
098     * @return commento avente l'id specificato
099     */
100    public CommentDTO getComment(@CommentExists int id){
101        return map(genericRepository.findById(Comment.class, id));
102    }
103
104
105    /**
106     * Cancella un commento dato il suo id
107     * @param id l'id di un commento esistente
108     */
109    @AuthenticationRequired
110    public void delete(@CommentExists int id){
111        Comment comment = genericRepository.findById(Comment.class, id);
112        if(currentUser.getId() != comment.getAuthor().getId() && !currentUser.isAdmin())
113            throw new AuthorizationException();
114        genericRepository.remove(genericRepository.findById(Comment.class, id));
115    }
116
117    /**
118     * Modifica un commento dato il suo id
119     * @param id l'id di un commento esistente
120     * @param text stringa con testo da sostituire
121     */
122    @AuthenticationRequired
123    public void editComment(@CommentExists int id, @NotBlank @Size(max=65535) String text){
124        Comment comment = genericRepository.findById(Comment.class, id);
125        if(currentUser.getId() != comment.getAuthor().getId() && !currentUser.isAdmin())
126            throw new AuthorizationException();
127        comment.setContent(text);
128    }
129
130    /**
131     * Crea un nuovo commento e ne restituisce l'id
132     * @param text una stringa non vuota di massimo 65535 caratteri
133     * @param postId id di un post esistente
134     * @return id del commento creato
135     */
136    @AuthenticationRequired
137    @DenyBannedUsers
138    public int newComment(@NotBlank @Size(max=65535) String text,
139                          @PostExists int postId){
140        User user = genericRepository.findById(User.class, currentUser.getId());
141
142        Comment comment = new Comment();
143        comment.setAuthor(user);
144        comment.setContent(text);
145        comment.setPost(genericRepository.findById(Post.class, postId));
146        return genericRepository.insert(comment).getId();
147    }
148
149    /**
150     * Crea una risposta a un commento e ne restituisce l'id
151     * @param text una stringa non vuota di massimo 1000 caratteri
152     * @param parentCommentId id di un commento esistente
153     * @return id del commento creato
154     */
155    @AuthenticationRequired
156    public int newCommentReply(@NotBlank @Size(max=65535) String text,
157                               @CommentExists int parentCommentId){
158        User user = genericRepository.findById(User.class, currentUser.getId());
159        Comment parent = genericRepository.findById(Comment.class, parentCommentId);
160
161        Comment comment = new Comment();
162        comment.setAuthor(user);
163        comment.setContent(text);
164        comment.setPost(parent.getPost());
165        comment.setParentComment(parent);
166        return genericRepository.insert(comment).getId();
167    }
168
169}
