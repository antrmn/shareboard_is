001package service;
002
003import persistence.model.Comment;
004import persistence.model.CommentVote;
005import persistence.model.Post;
006import persistence.model.User;
007import persistence.repo.CommentRepository;
008import persistence.repo.GenericRepository;
009import service.auth.AuthenticationRequired;
010import service.auth.DenyBannedUsers;
011import service.dto.CommentDTO;
012import service.dto.CurrentUser;
013import service.validation.CommentExists;
014import service.validation.PostExists;
015
016import javax.enterprise.context.ApplicationScoped;
017import javax.inject.Inject;
018import javax.transaction.Transactional;
019import javax.validation.constraints.NotBlank;
020import javax.validation.constraints.Size;
021import java.time.Instant;
022import java.util.List;
023import java.util.Map;
024
025import static java.util.stream.Collectors.groupingBy;
026import static java.util.stream.Collectors.toList;
027
028
029@ApplicationScoped
030@Transactional
031public class CommentService {
032    @Inject private GenericRepository genericRepository;
033    @Inject private CommentRepository commentRepo;
034    @Inject private CurrentUser currentUser;
035
036    private static final int MAX_COMMENT_DEPTH = 4;
037
038    /**
039     * Converte Comment in CommentDTO.
040     * @param comment commento da convertire
041     * @return commentDTO con i dati di comment
042     */
043    private CommentDTO map (Comment comment){
044        CommentVote commentVote = null;
045        if(currentUser.isLoggedIn()){
046            User user = genericRepository.findById(User.class, currentUser.getId());
047            commentVote = comment.getVote(user);
048        }
049
050        return CommentDTO.builder()
051                .id(comment.getId())
052                .authorUsername(comment.getAuthor().getUsername())
053                .authorId(comment.getAuthor().getId())
054                .creationDate(comment.getCreationDate())
055                .postId(comment.getPost().getId())
056                .vote(commentVote == null ? 0 : commentVote.getVote())
057                .content(comment.getContent())
058                .votes(comment.getVotesCount() == null ? 0 : comment.getVotesCount())
059                .creationDate(comment.getCreationDate()  == null ? Instant.now() : comment.getCreationDate())
060                .parentCommentId(comment.getParentComment() == null ? 0 : comment.getParentComment().getId())
061                .build();
062    }
063
064    /**
065     * Ritorna una mappa la cui chiave è l'id del commento padre e il valore una lista di CommentDTO
066     * @param postId l'id di un post esistente di cui si vuole ottenere i commenti
067     * @return mappa con i commenti del post
068     */
069    public Map<Integer,List<CommentDTO>> getPostComments(@PostExists int postId){
070        List<Comment> comments = commentRepo.getByPost(genericRepository.findById(Post.class, postId), MAX_COMMENT_DEPTH);
071        return comments.stream().map(this::map).collect(groupingBy(CommentDTO::getParentCommentId, toList()));
072    }
073
074    /**
075     * Ritorna una mappa la cui chiave è l'id del commento padre e il valore una lista di CommentDTO
076     * @param commentId l'id di un commento esistente di cui si vuole ottenere le risposte
077     * @return mappa con le risposte al commento
078     */
079    public Map<Integer, List<CommentDTO>> getReplies(@CommentExists int commentId){
080        List<Comment> comments = commentRepo.getReplies(genericRepository.findById(Comment.class, commentId), MAX_COMMENT_DEPTH);
081        return comments.stream().map(this::map).collect(groupingBy(CommentDTO::getParentCommentId, toList()));
082    }
083
084    /**
085     * Ritorna un commento dato il suo id
086     * @param id l'id di un commento esistente
087     * @return commento avente l'id specificato
088     */
089    public CommentDTO getComment(@CommentExists int id){
090        return map(genericRepository.findById(Comment.class, id));
091    }
092
093
094    /**
095     * Cancella un commento dato il suo id
096     * @param id l'id di un commento esistente
097     */
098    @AuthenticationRequired
099    public void delete(@CommentExists int id){
100        genericRepository.remove(genericRepository.findById(Comment.class, id));
101    }
102
103    /**
104     * Modifica un commento dato il suo id
105     * @param id l'id di un commento esistente
106     * @param text stringa con testo da sostituire
107     */
108    @AuthenticationRequired
109    public void editComment(@CommentExists int id, String text){
110        genericRepository.findById(Comment.class, id).setContent(text);
111    }
112
113    /**
114     * Crea un nuovo commento e ne restituisce l'id
115     * @param text una stringa non vuota di massimo 1000 caratteri
116     * @param postId id di un post esistente
117     * @return id del commento creato
118     */
119    @AuthenticationRequired
120    @DenyBannedUsers
121    public int newComment(@NotBlank @Size String text,
122                       @PostExists int postId){
123        User user = genericRepository.findById(User.class, currentUser.getId());
124
125        Comment comment = new Comment();
126        comment.setAuthor(user);
127        comment.setContent(text);
128        comment.setPost(genericRepository.findById(Post.class, postId));
129        return genericRepository.insert(comment).getId();
130    }
131
132    /**
133     * Crea una risposta a un commento e ne restituisce l'id
134     * @param text una stringa non vuota di massimo 1000 caratteri
135     * @param parentId id di un commento esistente
136     * @param postId id di un post esistente
137     * @return id del commento creato
138     */
139    @AuthenticationRequired
140    public int newCommentReply(@NotBlank @Size String text,
141                               @CommentExists int parentId,
142                               @PostExists int postId){
143        User user = genericRepository.findById(User.class, currentUser.getId());
144        Comment parent = genericRepository.findById(Comment.class, parentId);
145
146        Comment comment = new Comment();
147        comment.setAuthor(user);
148        comment.setContent(text);
149        comment.setPost(genericRepository.findById(Post.class, postId));
150        comment.setParentComment(parent);
151        return genericRepository.insert(comment).getId();
152    }
153
154}
