001package service;
002
003import persistence.model.Follow;
004import persistence.model.Section;
005import persistence.model.User;
006import persistence.repo.GenericRepository;
007import service.auth.AuthenticationRequired;
008import service.dto.CurrentUser;
009import service.validation.SectionExistsById;
010
011import javax.enterprise.context.ApplicationScoped;
012import javax.inject.Inject;
013import javax.transaction.Transactional;
014
015@ApplicationScoped
016@Transactional
017public class FollowService {
018    @Inject private GenericRepository genericRepository;
019    @Inject private CurrentUser currentUser;
020
021    @AuthenticationRequired
022    public Follow follow(@SectionExistsById int sectionId){
023        User user = genericRepository.findById(User.class, currentUser.getId());
024        Section section = genericRepository.findById(Section.class,sectionId);
025        return genericRepository.insert(new Follow(user,section));
026    }
027
028    @AuthenticationRequired
029    public void unFollow(@SectionExistsById int sectionId){
030        User user = genericRepository.findById(User.class, currentUser.getId());
031        Section section = genericRepository.findById(Section.class,sectionId);
032        Follow follow = section.getFollow(user);
033
034        if(follow != null)
035            genericRepository.remove(follow);
036    }
037
038}
