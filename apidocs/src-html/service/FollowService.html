001package service;
002
003import persistence.model.Follow;
004import persistence.model.Section;
005import persistence.model.User;
006import persistence.repo.GenericRepository;
007import service.auth.AuthenticationRequired;
008import service.dto.CurrentUser;
009import service.validation.SectionExistsById;
010
011import javax.enterprise.context.ApplicationScoped;
012import javax.inject.Inject;
013import javax.transaction.Transactional;
014
015@ApplicationScoped
016@Transactional
017public class FollowService {
018    @Inject private GenericRepository genericRepository;
019    @Inject private CurrentUser currentUser;
020
021    /**
022     * Permette di seguire una sezione
023     * @param sectionId identificativo sezione
024     * @return entit√† follow inserita nel database
025     */
026    @AuthenticationRequired
027    public Follow follow(@SectionExistsById int sectionId){
028        User user = genericRepository.findById(User.class, currentUser.getId());
029        Section section = genericRepository.findById(Section.class,sectionId);
030        return genericRepository.insert(new Follow(user,section));
031    }
032
033    /**
034     * Permette di togliere il follow ad una sezione
035     * @param sectionId identificativo sezione
036     */
037    @AuthenticationRequired
038    public void unFollow(@SectionExistsById int sectionId){
039        User user = genericRepository.findById(User.class, currentUser.getId());
040        Section section = genericRepository.findById(Section.class,sectionId);
041        Follow follow = section.getFollow(user);
042
043        if(follow != null)
044            genericRepository.remove(follow);
045    }
046
047}
