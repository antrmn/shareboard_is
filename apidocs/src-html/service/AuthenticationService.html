001package service;
002
003import persistence.model.User;
004import persistence.repo.GenericRepository;
005import persistence.repo.UserRepository;
006import service.dto.CurrentUser;
007import util.Pbkdf2PasswordHashImpl;
008
009import javax.enterprise.context.RequestScoped;
010import javax.enterprise.context.SessionScoped;
011import javax.enterprise.inject.Produces;
012import javax.inject.Inject;
013import javax.inject.Named;
014import javax.transaction.Transactional;
015import java.io.Serializable;
016import java.time.Instant;
017
018@SessionScoped
019@Transactional
020public class AuthenticationService implements Serializable {
021    @Inject UserRepository userRepository;
022    @Inject GenericRepository genericRepository;
023    @Inject Pbkdf2PasswordHashImpl passwordHash;
024
025    private int currentUserId = 0;
026
027    /**
028     * Autentica un utente
029     * @param username stringa con nome utente
030     * @param password stringa con password
031     * @return esito dell'operazione
032     */
033    public boolean authenticate(String username, String password){
034        User user = userRepository.getByName(username);
035        if(user == null)
036            return false;
037        if (passwordHash.verify(password, user.getPassword(), user.getSalt())){
038            currentUserId = user.getId();
039            return true;
040        } else {
041            return false;
042        }
043    }
044
045    /**
046     * Ritorna l'utente in uso e ne rende accessibile i dati nelle jsp
047     * @return utente in uso
048     */
049    @Named("currentUser") //accessibile nelle jsp con ${currentUser}
050    @RequestScoped
051    @Produces
052    public CurrentUser getCurrentUser(){
053        if(currentUserId <= 0)
054            return new CurrentUser();
055
056        User user = genericRepository.findById(User.class,currentUserId);
057        if(user == null)
058            return new CurrentUser();
059
060        Instant longestCurrentBan = user.getBans().isEmpty() ? null : user.getBans().get(0).getEndTime();
061
062        CurrentUser currentUser = CurrentUser.builder()
063                .username(user.getUsername())
064                .id(user.getId())
065                .isAdmin(user.getAdmin())
066                .picture(user.getPicture())
067                .banDuration(longestCurrentBan)
068                .isLoggedIn(true)
069                .build();
070
071        System.out.println(currentUser);
072        return currentUser;
073    }
074
075}
