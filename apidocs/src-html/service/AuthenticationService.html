001package service;
002
003import persistence.model.User;
004import persistence.repo.GenericRepository;
005import service.dto.CurrentUser;
006import util.Pbkdf2PasswordHashImpl;
007
008import javax.enterprise.context.RequestScoped;
009import javax.enterprise.context.SessionScoped;
010import javax.enterprise.inject.Produces;
011import javax.inject.Inject;
012import javax.inject.Named;
013import javax.transaction.Transactional;
014import java.io.Serializable;
015import java.time.Instant;
016
017@SessionScoped
018@Transactional
019public class AuthenticationService implements Serializable {
020    private GenericRepository genericRepository;
021    private Pbkdf2PasswordHashImpl passwordHash;
022    private int currentUserId = 0;
023
024    protected AuthenticationService(){}
025
026    @Inject
027    protected AuthenticationService(GenericRepository genericRepository,
028                                    Pbkdf2PasswordHashImpl passwordHash){
029        this.genericRepository = genericRepository;
030        this.passwordHash = passwordHash;
031    }
032
033
034    /**
035     * Autentica un utente
036     * @param username stringa con nome utente
037     * @param password stringa con password
038     * @return esito dell'operazione
039     */
040    public boolean authenticate(String username, String password){
041        User user = genericRepository.findByNaturalId(User.class,username);
042        if(user == null)
043            return false;
044        if (passwordHash.verify(password, user.getPassword(), user.getSalt())){
045            currentUserId = user.getId();
046            return true;
047        } else {
048            return false;
049        }
050    }
051
052    /**
053     * Ritorna l'utente in uso e ne rende accessibile i dati nelle jsp
054     * @return utente in uso
055     */
056    @Named("currentUser") //accessibile nelle jsp con ${currentUser}
057    @RequestScoped
058    @Produces
059    public CurrentUser getCurrentUser(){
060        if(currentUserId <= 0)
061            return new CurrentUser();
062
063        User user = genericRepository.findById(User.class,currentUserId);
064        if(user == null)
065            return new CurrentUser();
066
067        Instant longestCurrentBan = user.getBans().isEmpty() ? null : user.getBans().get(0).getEndTime();
068
069        CurrentUser currentUser = CurrentUser.builder()
070                .username(user.getUsername())
071                .id(user.getId())
072                .isAdmin(user.getAdmin())
073                .picture(user.getPicture())
074                .banDuration(longestCurrentBan)
075                .isLoggedIn(true)
076                .build();
077
078        System.out.println(currentUser);
079        return currentUser;
080    }
081
082}
