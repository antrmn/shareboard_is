001package service;
002
003import persistence.model.User;
004import persistence.repo.GenericRepository;
005import persistence.repo.UserRepository;
006import service.dto.CurrentUser;
007import util.Pbkdf2PasswordHashImpl;
008
009import javax.enterprise.context.RequestScoped;
010import javax.enterprise.context.SessionScoped;
011import javax.enterprise.inject.Produces;
012import javax.inject.Inject;
013import javax.inject.Named;
014import javax.transaction.Transactional;
015import java.io.Serializable;
016import java.time.Instant;
017
018@SessionScoped
019@Transactional
020public class AuthenticationService implements Serializable {
021    @Inject UserRepository userRepository;
022    @Inject GenericRepository genericRepository;
023    @Inject Pbkdf2PasswordHashImpl passwordHash;
024
025    private int currentUserId = 0;
026
027    public boolean authenticate(String username, String password){
028        User user = userRepository.getByName(username);
029        if(user == null)
030            return false;
031        if (passwordHash.verify(password, user.getPassword(), user.getSalt())){
032            currentUserId = user.getId();
033            return true;
034        } else {
035            return false;
036        }
037    }
038
039    @Named("currentUser") //accessibile nelle jsp con ${currentUser}
040    @RequestScoped
041    @Produces
042    public CurrentUser getCurrentUser(){
043        if(currentUserId <= 0)
044            return new CurrentUser();
045
046        User user = genericRepository.findById(User.class,currentUserId);
047        if(user == null)
048            return new CurrentUser();
049
050        Instant longestCurrentBan = user.getBans().isEmpty() ? null : user.getBans().get(0).getEndTime();
051
052        CurrentUser currentUser = CurrentUser.builder()
053                .username(user.getUsername())
054                .id(user.getId())
055                .isAdmin(user.getAdmin())
056                .picture(user.getPicture())
057                .banDuration(longestCurrentBan)
058                .isLoggedIn(true)
059                .build();
060
061        System.out.println(currentUser);
062        return currentUser;
063    }
064
065}
