001package service.auth;
002
003import service.dto.CurrentUser;
004
005import javax.annotation.Priority;
006import javax.inject.Inject;
007import javax.interceptor.AroundInvoke;
008import javax.interceptor.Interceptor;
009import javax.interceptor.InvocationContext;
010
011@Interceptor
012@AuthenticationRequired
013@Priority(Interceptor.Priority.APPLICATION+1)
014public class AuthenticationRequiredInterceptor {
015    @Inject private CurrentUser currentUser;
016
017    @AroundInvoke
018    public Object checkAdmin(InvocationContext invocationContext) throws Exception{
019        if(currentUser == null || !currentUser.isLoggedIn()){
020            throw new AuthenticationRequiredException();
021        }
022        return invocationContext.proceed();
023    }
024}
