001package service;
002
003import com.sun.istack.NotNull;
004import persistence.model.User;
005import persistence.repo.BinaryContentRepository;
006import persistence.repo.GenericRepository;
007import service.auth.AdminsOnly;
008import service.auth.AuthenticationRequired;
009import service.auth.AuthorizationException;
010import service.dto.CurrentUser;
011import service.dto.UserEditPage;
012import service.dto.UserProfile;
013import service.validation.*;
014import util.Pbkdf2PasswordHashImpl;
015import util.Pbkdf2PasswordHashImpl.HashedPassword;
016import util.ReadLimitExceededException;
017
018import javax.enterprise.context.ApplicationScoped;
019import javax.inject.Inject;
020import javax.transaction.Transactional;
021import javax.validation.Valid;
022import javax.validation.constraints.NotBlank;
023import java.io.IOException;
024import java.util.List;
025import java.util.stream.Collectors;
026
027@ApplicationScoped
028@Transactional
029public class UserService {
030    private GenericRepository genericRepository;
031    private BinaryContentRepository bcRepo;
032    private CurrentUser currentUser;
033    private Pbkdf2PasswordHashImpl passwordHash;
034
035    protected UserService(){}
036
037    @Inject
038    protected UserService(GenericRepository genericRepository,
039                          BinaryContentRepository binaryContentRepository, Pbkdf2PasswordHashImpl passwordHash,
040                          CurrentUser currentUser){
041        this.genericRepository = genericRepository;
042        this.bcRepo = binaryContentRepository;
043        this.passwordHash = passwordHash;
044        this.currentUser = currentUser;
045    }
046
047    private UserProfile map(User user){
048        return UserProfile.builder()
049                .id(user.getId())
050                .email(user.getEmail())
051                .creationDate(user.getCreationDate())
052                .description(user.getDescription())
053                .picture(user.getPicture())
054                .username(user.getUsername())
055                .isAdmin(user.getAdmin()).build();
056    }
057
058    /**
059     * Inverte lo stato di admin di un utente dato un id
060     * @param id id di un utente esistente
061     */
062    @AdminsOnly
063    public void toggleAdmin(@UserExists int id){
064        User u = genericRepository.findById(User.class,id);
065        u.setAdmin(!u.getAdmin());
066    }
067
068    /**
069     * Ritorna un entita UserProfile dato un nome
070     * @param name di un utente esistente
071     * @return entita UserProfile
072     */
073    public UserProfile getUser(@NotBlank @UserExists String name){
074        User u = genericRepository.findByNaturalId(User.class,name);
075        return map(u);
076    }
077
078    /**
079     * Ritorna un entita UserProfile dato un id
080     * @param id di un utente esistente
081     * @return entita UserProfile
082     */
083    public UserProfile getUser(@UserExists int id){
084        User u = genericRepository.findById(User.class,id);
085        return map(u);
086    }
087
088    /**
089     * Ritorna un lo username di un utente dato un id
090     * @param id di un utente esistente
091     * @return nome dell'utente
092     */
093    public String getUsernameById(@UserExists int id){
094        return genericRepository.findById(User.class,id).getUsername();
095    }
096
097    /**
098     * Ritorna un lista di tutte le gli utenti
099     * @return lista di entita UserIdentityDTO
100     */
101    public List<UserProfile> showUsers(){
102        List<User> users = genericRepository.findAll(User.class);
103        return users.stream().map(this::map).collect(Collectors.toList());
104    }
105
106    /**
107     * Elimina un utente dato un id
108     * @param id di un utente esistente
109     */
110    @AdminsOnly
111    public void delete(@UserExists int id){
112        genericRepository.remove(genericRepository.findById(User.class, id));
113    }
114
115    /**
116     * Modifica i dati di un utente dato un id
117     * @param edit nuovi dati dell'utente
118     * @param id di un utente esistente
119     */
120    @AuthenticationRequired
121    public void edit(@Valid UserEditPage edit,
122                     @UserExists int id){
123        if(currentUser.getId() != id && !currentUser.isAdmin())
124            throw new AuthorizationException();
125
126        User u = genericRepository.findById(User.class,id);
127
128        if(edit.getDescription() != null){
129            u.setDescription(edit.getDescription());
130        }
131        if(edit.getEmail() != null){
132            u.setEmail(edit.getEmail());
133        }
134        if(edit.getPassword() != null){
135            HashedPassword hashedPassword = passwordHash.generate(edit.getPassword());
136            u.setPassword(hashedPassword.getPassword());
137            u.setSalt(hashedPassword.getSalt());
138        }
139        if(edit.getPicture() != null){
140            try {
141                u.setPicture(bcRepo.insert(edit.getPicture()));
142            } catch(ReadLimitExceededException e){
143                throw new IllegalArgumentException("Il file non deve superare i 5MB");
144            } catch (IOException e) {
145                throw new RuntimeException(e); //todo: delet this
146            }
147        }
148    }
149
150    /**
151     * Crea un nuovo utente e ne restituisce l'id
152     * @param email stringa non vuota, unica e in formato email
153     * @param username stringa non vuota e unica
154     * @param password Stringa non vuota con minimo: 3 e massimo: 255 caratteri
155     * @return identificativo dell'utente creato
156     */
157    public int newUser(@NotNull @EmailFormat @UniqueEmail String email,
158                       @NotNull @UsernameFormat @UniqueUsername String username,
159                       @NotNull @PasswordFormat String password){
160        User user = new User();
161        user.setUsername(username);
162        user.setEmail(email);
163        HashedPassword hashedPassword = passwordHash.generate(password);
164        user.setPassword(hashedPassword.getPassword());
165        user.setSalt(hashedPassword.getSalt());
166        user.setAdmin(false);
167
168        user = genericRepository.insert(user);
169        return user.getId();
170    }
171}
