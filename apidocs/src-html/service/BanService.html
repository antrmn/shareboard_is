001package service;
002
003import persistence.model.Ban;
004import persistence.model.User;
005import persistence.repo.GenericRepository;
006import service.auth.AdminsOnly;
007import service.dto.BanDTO;
008import service.validation.BanExists;
009import service.validation.UserExists;
010
011import javax.enterprise.context.ApplicationScoped;
012import javax.inject.Inject;
013import javax.transaction.Transactional;
014import javax.validation.constraints.Future;
015import java.time.Instant;
016import java.util.ArrayList;
017import java.util.List;
018
019@ApplicationScoped
020@Transactional
021public class BanService {
022    @Inject private GenericRepository genericRepository;
023
024    @AdminsOnly
025    public Ban addBan(@Future Instant date, @UserExists int userId) {
026        Ban ban = new Ban();
027        ban.setEndTime(date);
028        User user = new User();
029        user.setId(userId);
030        ban.setUser(user);
031        return genericRepository.insert(ban);
032    }
033
034    @AdminsOnly
035    public void removeBan(@BanExists int banId){
036        genericRepository.remove(genericRepository.findById(Ban.class, banId));
037    }
038
039    @AdminsOnly
040    public List<BanDTO> retrieveUserBan(@UserExists int userId){
041        User user = genericRepository.findById(User.class, userId);
042        List<BanDTO> bans = new ArrayList<>();
043        List<Ban> temp = user.getBans();
044
045        for(Ban b : temp){
046            BanDTO ban = new BanDTO(b.getId(), b.getEndTime(), b.getUser().getId());
047            bans.add(ban);
048        }
049        return bans;
050    }
051}
