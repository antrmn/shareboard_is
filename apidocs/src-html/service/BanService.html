001package service;
002
003import persistence.model.Ban;
004import persistence.model.User;
005import persistence.repo.GenericRepository;
006import service.auth.AdminsOnly;
007import service.dto.BanDTO;
008import service.validation.BanExists;
009import service.validation.UserExists;
010
011import javax.enterprise.context.ApplicationScoped;
012import javax.inject.Inject;
013import javax.transaction.Transactional;
014import javax.validation.constraints.Future;
015import javax.validation.constraints.NotNull;
016import java.time.Instant;
017import java.util.ArrayList;
018import java.util.List;
019
020@ApplicationScoped
021@Transactional
022public class BanService {
023    private GenericRepository genericRepository;
024
025    protected BanService(){}
026
027    @Inject
028    protected BanService(GenericRepository genericRepository){
029        this.genericRepository = genericRepository;
030    }
031
032
033    /**
034     * Aggiunge un ban dato l'id di un utente.
035     * @param date data di fine ban
036     * @param userId identificativo utente
037     * @return entit√† ban aggiunta
038     */
039    @AdminsOnly
040    public int addBan(@NotNull @Future Instant date, @UserExists int userId) {
041        Ban ban = new Ban();
042        ban.setEndTime(date);
043        User user = new User();
044        user.setId(userId);
045        ban.setUser(user);
046        return genericRepository.insert(ban).getId();
047    }
048
049    /**
050     * Rimuove un ban dato il suo id
051     * @param banId identificativo del ban
052     */
053    @AdminsOnly
054    public void removeBan(@BanExists int banId){
055        genericRepository.remove(genericRepository.findById(Ban.class, banId));
056    }
057
058    /**
059     * Ritorna la lista dei ban di un utente
060     * @param userId identificativo utente
061     * @return lista dei ban relativi all'utente
062     */
063    @AdminsOnly
064    public List<BanDTO> retrieveUserBan(@UserExists int userId){
065        User user = genericRepository.findById(User.class, userId);
066        List<BanDTO> bans = new ArrayList<>();
067        List<Ban> temp = user.getBans();
068
069        for(Ban b : temp){
070            BanDTO ban = new BanDTO(b.getId(), b.getEndTime(), b.getUser().getId());
071            bans.add(ban);
072        }
073        return bans;
074    }
075}
