041    private PostPage mapPost(Post post){
042        User user = null;
043        if(currentUser.isLoggedIn())
044            user = genericRepository.findById(User.class,currentUser.getId());
045
046        //converte persistence.Post.Type in service.dto.PostType (ew)
047        PostType postType = post.getType() == IMG ? PostType.IMG : PostType.TEXT;
048
049        return PostPage.builder()
050                .id(post.getId())
051                .title(post.getTitle())
052                .vote(post.getVote(user) == null ? 0 : post.getVote(user).getVote())
053                .votes(post.getVotesCount() == null ? 0 : post.getVotesCount())
054                .sectionName(post.getSection().getName())
055                .authorName(post.getAuthor().getUsername())
056                .sectionId(post.getSection().getId())
057                .authorId(post.getAuthor().getId())
058                .content(post.getContent())
059                .creationDate(post.getCreationDate() == null ? Instant.now() : post.getCreationDate())
060                .nComments(post.getCommentCount())
061                .type(postType)
062                .build();
063    }
064
065    public PostPage getPost(@PostExists int id){
066        Post p = genericRepository.findById(Post.class,id);
067        PostPage post = mapPost(p);
068        return post;
069    }

071    @Transactional
072    public List<PostPage> loadPosts(PostSearchForm form){
073        final int elementsPerPage = 10;
074        PostRepository.PostFinder finder = postRepo.getFinder();
075
076        finder.limit(elementsPerPage);
077
078        int page = form.getPage() > 0 ? form.getPage() : 1;
079        finder.offset((page == 1) ? 0 : (page-1) * elementsPerPage+1);
080
081        if(form.getContent() != null) {
082            finder.byContent(form.getContent());
083        }
084        if(form.isIncludeBody()){
085            finder.includeBody();
086        }
087        if(form.getSectionName() != null){
088            Section section = sectionRepo.getByName(form.getSectionName());
089            if(section != null){
090                finder.bySection(section);
091            }
092        }
093        if(form.getAuthorName() != null){
094            User user = userRepo.getByName(form.getAuthorName());
095            if(user != null){
096                finder.byAuthor(user);
097            }
098        }
099        if(form.isOnlyFollow() && currentUser.isLoggedIn()){
100            finder.joinUserFollows(genericRepository.findById(User.class,currentUser.getId()));
101        }
102        if(form.getPostedAfter() != null){
103            finder.postedAfter(form.getPostedAfter());
104        }
105        if(form.getPostedBefore() != null){
106            finder.postedBefore(form.getPostedBefore());
107        }
108
109        //null-safe switch
110        switch(form.getOrderBy() != null ? form.getOrderBy() : NEWEST){
111            case NEWEST: default: {
112                finder.getNewest();
113                break;
114            }
115            case OLDEST:{
116                finder.getOldest();
117                break;
118            }
119            case MOSTVOTED:{
120                finder.getMostVoted();
121                break;
122            }
123        }
124        return finder.getResults().stream().map(this::mapPost).collect(Collectors.toList());
125    }

142    @AuthenticationRequired
143    public void editPost(PostEditDTO edit,
144                         @PostExists int id,
145                         @Image BufferedInputStream content){
146        Post post = genericRepository.findById(Post.class, id);
147        post.setTitle(edit.getTitle());
148        String fileName;
149        try {
150            fileName = bcRepo.insert(content);
151        } catch (IOException e) {
152            throw new RuntimeException(e);
153        }
154        post.setContent(fileName);
155        post.setType(edit.getType());
156    }

159    @AuthenticationRequired
160    @DenyBannedUsers
161    public int newPost(@NotBlank(message="{post.title.blank}") String title,
162                       String body,
163                       @SectionExists String sectionName){
164        User user = userRepo.getByName(currentUser.getUsername());
165        Section section = sectionRepo.getByName(sectionName);
166
167
168        Post post = new Post();
169        post.setTitle(title);
170        post.setContent(body == null || body.isBlank() ? "" : body);
171        post.setAuthor(user);
172        post.setSection(section);
173        post.setType(TEXT);
174
175        return genericRepository.insert(post).getId();
176    }
