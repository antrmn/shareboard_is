001package usecase.post;
002
003import common.http.ParameterConverter;
004import common.http.interceptor.InterceptableServlet;
005import usecase.comment.CommentDTO;
006import usecase.comment.CommentService;
007
008import javax.inject.Inject;
009import javax.servlet.ServletException;
010import javax.servlet.annotation.WebServlet;
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013import java.io.IOException;
014import java.util.Collections;
015import java.util.List;
016import java.util.Map;
017
018@WebServlet("/post")
019public class PostServlet extends InterceptableServlet {
020
021    @Inject private PostService postService;
022    @Inject private CommentService service;
023
024    @Override
025    public void init() throws ServletException {
026        getServletContext().setAttribute("maxCommentDepth", 3);
027    }
028
029    @Override
030    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
031        ParameterConverter converter = new ParameterConverter(req);
032        int postId = converter.getIntParameter("id").orElse(0); //viene ignorato se commentId != 0
033        int commentId = converter.getIntParameter("comment").orElse(0);
034
035        PostPage post;
036        Map<Integer, List<CommentDTO>> comments;
037        int initialIndex;
038        if (commentId == 0) {
039            post = postService.getPost(postId);
040            comments = service.getPostComments(postId);
041            initialIndex = 0; //si parte dai root comments
042        } else{
043            comments = service.getReplies(commentId);
044            initialIndex = Collections.min(comments.keySet()); //il commento di partenza è quello all'indice più basso
045            postId = comments.get(initialIndex).get(0).getPostId(); //prendi un commento qualsiasi, ricava il postId
046            post = postService.getPost(postId);
047        }
048
049        req.setAttribute("post", post);
050        req.setAttribute("comments", comments);
051        req.setAttribute("initialIndex", initialIndex);
052        req.getRequestDispatcher("/WEB-INF/views/section/post.jsp").forward(req, resp);
053    }
054
055    @Override
056    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
057        doGet(req, resp);
058    }
059}
