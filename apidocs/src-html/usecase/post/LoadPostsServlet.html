001package usecase.post;
002
003import common.http.ParameterConverter;
004import common.http.interceptor.InterceptableServlet;
005
006import javax.inject.Inject;
007import javax.servlet.ServletException;
008import javax.servlet.annotation.WebServlet;
009import javax.servlet.http.HttpServletRequest;
010import javax.servlet.http.HttpServletResponse;
011import java.io.IOException;
012import java.time.Instant;
013import java.time.LocalDate;
014import java.time.ZoneOffset;
015import java.util.List;
016import java.util.Map;
017import java.util.function.Function;
018
019import static usecase.post.PostSearchForm.SortCriteria.*;
020
021@WebServlet("/loadposts")
022public class LoadPostsServlet extends InterceptableServlet {
023    @Inject private PostService postService;
024
025    private static final Function<LocalDate, Instant> LOCALDATE_TO_INSTANT =
026            x -> x.atStartOfDay().toInstant(ZoneOffset.UTC);
027
028    private static final Map<String, PostSearchForm.SortCriteria> SORT_CRITERIA = Map.of(
029            "oldest", OLDEST,
030            "newest", NEWEST,
031            "mostvoted", MOSTVOTED);
032
033    @Override
034    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
035        ParameterConverter converter = new ParameterConverter(req);
036        String content = req.getParameter("content");
037        String section = req.getParameter("section");
038        String author = req.getParameter("author");
039        Instant postedAfter = converter.getDateParameter("postedAfter")
040                .map(LOCALDATE_TO_INSTANT).orElse(null);
041        Instant postedBefore = converter.getDateParameter("postedBefore")
042                .map(LOCALDATE_TO_INSTANT).orElse(null);
043        PostSearchForm.SortCriteria orderBy = SORT_CRITERIA.get(req.getParameter("orderby"));
044        int page = converter.getIntParameter("page").orElse(1);
045        boolean onlyFollow = "on".equals(req.getParameter("onlyfollow"));
046        boolean includeBody = req.getParameter("includeBody") != null;
047
048        PostSearchForm postSearchForm = PostSearchForm.builder()
049                .content(content)
050                .onlyFollow(onlyFollow)
051                .includeBody(includeBody)
052                .sectionName(section)
053                .authorName(author)
054                .orderBy(orderBy)
055                .postedAfter(postedAfter)
056                .postedBefore(postedBefore)
057                .page(page)
058                .build();
059
060        List<PostPage> posts = postService.loadPosts(postSearchForm);
061        req.setAttribute("posts",posts);
062        req.getRequestDispatcher("/WEB-INF/views/partials/post-previews.jsp").forward(req,resp);
063    }
064
065    @Override
066    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
067        doGet(req, resp);
068    }
069}
