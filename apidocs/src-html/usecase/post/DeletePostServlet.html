001package usecase.post;
002
003import common.http.ParameterConverter;
004import common.http.interceptor.InterceptableServlet;
005import usecase.auth.AuthorizationConstraints;
006
007import javax.inject.Inject;
008import javax.servlet.ServletException;
009import javax.servlet.annotation.WebServlet;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpServletResponse;
012import java.io.IOException;
013
014import static usecase.auth.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
015
016@WebServlet(name = "DeletePostServlet", value = "/DeletePostServlet")
017@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
018public class DeletePostServlet extends InterceptableServlet {
019    @Inject private PostService service;
020
021    @Override
022    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
023        ParameterConverter converter = new ParameterConverter(request);
024        int postId = converter.getIntParameter("id").orElse(0);
025        String sectionName = service.getPost(postId).getSectionName();
026        service.delete(postId);
027        response.sendRedirect(getServletContext().getContextPath() + "/s/" + sectionName);
028    }
029
030    @Override
031    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
032        doGet(request,response);
033    }
034}
