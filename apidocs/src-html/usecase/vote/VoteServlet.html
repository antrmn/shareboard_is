001package usecase.vote;
002
003import common.http.ParameterConverter;
004import common.http.error.JSONError;
005import common.http.interceptor.InterceptableServlet;
006import usecase.auth.AuthorizationConstraints;
007
008import javax.inject.Inject;
009import javax.servlet.ServletException;
010import javax.servlet.annotation.WebServlet;
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013import java.io.IOException;
014
015import static usecase.auth.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
016
017@WebServlet("/vote")
018@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
019public class VoteServlet extends InterceptableServlet {
020    @Inject private VoteService service;
021
022    @Override
023    @JSONError
024    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
025        ParameterConverter converter = new ParameterConverter(request);
026        int id = converter.getIntParameter("id").orElse(0);
027        String vote = request.getParameter("vote");
028        String type = request.getParameter("type");
029
030        if("post".equalsIgnoreCase(type)){
031            if("upvote".equalsIgnoreCase(vote))
032                service.upvotePost(id);
033            else if("downvote".equalsIgnoreCase(vote))
034                service.downvotePost(id);
035            else
036                throw new IllegalArgumentException();
037        }else if("comment".equalsIgnoreCase(type)){
038            if("upvote".equalsIgnoreCase(vote))
039                service.upvoteComment(id);
040            else if("downvote".equalsIgnoreCase(vote))
041                service.downvoteComment(id);
042            else
043                throw new IllegalArgumentException();
044        }else{
045            throw new IllegalArgumentException();
046        }
047    }
048
049    @Override
050    @JSONError
051    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
052        doGet(request,response);
053    }
054}
