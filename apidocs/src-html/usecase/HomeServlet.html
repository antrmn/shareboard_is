001package usecase;
002
003import common.http.interceptor.InterceptableServlet;
004
005import javax.servlet.ServletException;
006import javax.servlet.annotation.WebServlet;
007import javax.servlet.http.HttpServletRequest;
008import javax.servlet.http.HttpServletResponse;
009import java.io.IOException;
010
011@WebServlet("/home")
012public class HomeServlet extends InterceptableServlet {
013    @Override
014    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
015        req.getRequestDispatcher("/WEB-INF/views/home.jsp").forward(req,resp);
016    }
017
018    @Override
019    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
020        doGet(req, resp);
021    }
022}
