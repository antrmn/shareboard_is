001package usecase.user;
002
003import common.http.ParameterConverter;
004import common.http.error.ForwardOnError;
005import common.http.interceptor.InterceptableServlet;
006import usecase.auth.AuthorizationConstraints;
007import usecase.auth.AuthorizationException;
008import usecase.auth.CurrentUser;
009
010import javax.inject.Inject;
011import javax.servlet.ServletException;
012import javax.servlet.annotation.MultipartConfig;
013import javax.servlet.annotation.WebServlet;
014import javax.servlet.http.HttpServletRequest;
015import javax.servlet.http.HttpServletResponse;
016import javax.servlet.http.Part;
017import java.io.BufferedInputStream;
018import java.io.IOException;
019
020import static usecase.auth.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
021
022@WebServlet("/edituser")
023@MultipartConfig
024@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
025public class EditUserServlet extends InterceptableServlet {
026    @Inject private UserService service;
027    @Inject private CurrentUser currentUser;
028
029    private static final String EDIT_USER_PAGE = "/WEB-INF/views/edit-user.jsp";
030
031    @Override
032    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
033        ParameterConverter converter = new ParameterConverter(request);
034        int userId = converter.getIntParameter("id").orElse(0);
035        if(!currentUser.isAdmin() && currentUser.getId() != userId)
036            throw new AuthorizationException();
037        UserProfile user = service.getUser(userId);
038        request.setAttribute("user", user);
039        request.getRequestDispatcher("/WEB-INF/views/edit-user.jsp").forward(request, response);
040    }
041
042    @Override
043    @ForwardOnError(EDIT_USER_PAGE)
044    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
045        ParameterConverter converter = new ParameterConverter(request);
046        int userId = converter.getIntParameter("id").orElse(0);
047        String email = request.getParameter("email");
048        String description = request.getParameter("description");
049        Part picture = request.getPart("picture");
050        String pass = request.getParameter("pass");
051        String pass2 = request.getParameter("pass2");
052
053        if((!pass.isEmpty() || !pass2.isEmpty()) && !pass.equals(pass2))
054            throw new IllegalArgumentException("Le password devono coincidere");
055
056        BufferedInputStream stream = picture.getSize() > 0 ? new BufferedInputStream(picture.getInputStream()) : null;
057        UserEditPage userEditPage = new UserEditPage(description,email,stream ,pass);
058        service.edit(userEditPage,userId);
059
060        response.sendRedirect(request.getContextPath() + "/u/" + service.getUsernameById(userId));
061    }
062}
