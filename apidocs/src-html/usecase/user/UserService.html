001package usecase.user;
002
003import com.sun.istack.NotNull;
004import domain.entity.User;
005import domain.repository.GenericRepository;
006import media.MediaRepository;
007import media.ReadLimitExceededException;
008import usecase.auth.*;
009import usecase.auth.Pbkdf2PasswordHash.HashedPassword;
010import usecase.user.validator.*;
011
012import javax.enterprise.context.ApplicationScoped;
013import javax.inject.Inject;
014import javax.transaction.Transactional;
015import javax.validation.Valid;
016import javax.validation.constraints.NotBlank;
017import java.io.IOException;
018import java.util.List;
019import java.util.stream.Collectors;
020
021@ApplicationScoped
022@Transactional
023public class UserService {
024    private GenericRepository genericRepository;
025    private MediaRepository bcRepo;
026    private CurrentUser currentUser;
027    private Pbkdf2PasswordHash passwordHash;
028
029    protected UserService(){}
030
031    @Inject
032    protected UserService(GenericRepository genericRepository,
033                          MediaRepository mediaRepository, Pbkdf2PasswordHash passwordHash,
034                          CurrentUser currentUser){
035        this.genericRepository = genericRepository;
036        this.bcRepo = mediaRepository;
037        this.passwordHash = passwordHash;
038        this.currentUser = currentUser;
039    }
040
041    private UserProfile map(User user){
042        return UserProfile.builder()
043                .id(user.getId())
044                .email(user.getEmail())
045                .creationDate(user.getCreationDate())
046                .description(user.getDescription())
047                .picture(user.getPicture())
048                .username(user.getUsername())
049                .isAdmin(user.getAdmin()).build();
050    }
051
052    /**
053     * Inverte lo stato di admin di un utente dato un id
054     * @param id id di un utente esistente
055     */
056    @AdminsOnly
057    public void toggleAdmin(@UserExists int id){
058        User u = genericRepository.findById(User.class,id);
059        u.setAdmin(!u.getAdmin());
060    }
061
062    /**
063     * Ritorna un entita UserProfile dato un nome
064     * @param name di un utente esistente
065     * @return entita UserProfile
066     */
067    public UserProfile getUser(@NotBlank @UserExists String name){
068        User u = genericRepository.findByNaturalId(User.class,name);
069        return map(u);
070    }
071
072    /**
073     * Ritorna un entita UserProfile dato un id
074     * @param id di un utente esistente
075     * @return entita UserProfile
076     */
077    public UserProfile getUser(@UserExists int id){
078        User u = genericRepository.findById(User.class,id);
079        return map(u);
080    }
081
082    /**
083     * Ritorna un lo username di un utente dato un id
084     * @param id di un utente esistente
085     * @return nome dell'utente
086     */
087    public String getUsernameById(@UserExists int id){
088        return genericRepository.findById(User.class,id).getUsername();
089    }
090
091    /**
092     * Ritorna un lista di tutte le gli utenti
093     * @return lista di entita UserIdentityDTO
094     */
095    public List<UserProfile> showUsers(){
096        List<User> users = genericRepository.findAll(User.class);
097        return users.stream().map(this::map).collect(Collectors.toList());
098    }
099
100    /**
101     * Elimina un utente dato un id
102     * @param id di un utente esistente
103     */
104    @AdminsOnly
105    public void delete(@UserExists int id){
106        genericRepository.remove(genericRepository.findById(User.class, id));
107    }
108
109    /**
110     * Modifica i dati di un utente dato un id
111     * @param edit nuovi dati dell'utente
112     * @param id di un utente esistente
113     */
114    @AuthenticationRequired
115    public void edit(@Valid UserEditPage edit,
116                     @UserExists int id){
117        if(currentUser.getId() != id && !currentUser.isAdmin())
118            throw new AuthorizationException();
119
120        User u = genericRepository.findById(User.class,id);
121
122        if(edit.getDescription() != null){
123            u.setDescription(edit.getDescription());
124        }
125        if(edit.getEmail() != null){
126            u.setEmail(edit.getEmail());
127        }
128        if(edit.getPassword() != null){
129            HashedPassword hashedPassword = passwordHash.generate(edit.getPassword());
130            u.setPassword(hashedPassword.getPassword());
131            u.setSalt(hashedPassword.getSalt());
132        }
133        if(edit.getPicture() != null){
134            try {
135                u.setPicture(bcRepo.insert(edit.getPicture()));
136            } catch(ReadLimitExceededException e){
137                throw new IllegalArgumentException("Il file non deve superare i 5MB");
138            } catch (IOException e) {
139                throw new RuntimeException(e); //todo: delet this
140            }
141        }
142    }
143
144    /**
145     * Crea un nuovo utente e ne restituisce l'id
146     * @param email stringa non vuota, unica e in formato email
147     * @param username stringa non vuota e unica
148     * @param password Stringa non vuota con minimo: 3 e massimo: 255 caratteri
149     * @return identificativo dell'utente creato
150     */
151    public int newUser(@NotNull @EmailFormat @UniqueEmail String email,
152                       @NotNull @UsernameFormat @UniqueUsername String username,
153                       @NotNull @PasswordFormat String password){
154        User user = new User();
155        user.setUsername(username);
156        user.setEmail(email);
157        HashedPassword hashedPassword = passwordHash.generate(password);
158        user.setPassword(hashedPassword.getPassword());
159        user.setSalt(hashedPassword.getSalt());
160        user.setAdmin(false);
161
162        user = genericRepository.insert(user);
163        return user.getId();
164    }
165}
