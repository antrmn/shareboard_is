001package usecase.user;
002
003import common.http.error.ForwardOnError;
004import common.http.interceptor.InterceptableServlet;
005import usecase.auth.AuthenticationService;
006import usecase.auth.CurrentUser;
007
008import javax.inject.Inject;
009import javax.servlet.ServletException;
010import javax.servlet.annotation.WebServlet;
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013import java.io.IOException;
014import java.util.ArrayList;
015import java.util.Objects;
016
017
018@WebServlet("/register")
019public
020class RegisterServlet extends InterceptableServlet {
021    @Inject private UserService userService;
022    @Inject private CurrentUser currentUser;
023    @Inject private AuthenticationService authenticationService;
024
025    private static final String REGISTER_PAGE = "/WEB-INF/views/register.jsp";
026
027    @Override
028    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
029        if(currentUser.isLoggedIn()){
030            resp.sendRedirect(req.getContextPath());
031            return;
032        }
033        req.getRequestDispatcher(REGISTER_PAGE).forward(req, resp);
034    }
035
036
037    @Override
038    @ForwardOnError(REGISTER_PAGE)
039    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
040        if(currentUser.isLoggedIn()){
041            resp.sendRedirect(req.getContextPath());
042            return;
043        }
044        String email = req.getParameter("mail");
045        String username = req.getParameter("username");
046        String password = req.getParameter("pass");
047        String confirmPassword = req.getParameter("pass2");
048
049        if(!Objects.equals(password, confirmPassword)){
050            ArrayList<String> errors = new ArrayList<>();
051            errors.add("Le password devono coincidere");
052            req.setAttribute("errors",errors);
053            req.getRequestDispatcher("/WEB-INF/views/register.jsp").forward(req, resp);
054        }else {
055            userService.newUser(email, username, password);
056            authenticationService.authenticate(username,password);
057            resp.sendRedirect(req.getContextPath());
058        }
059    }
060}
