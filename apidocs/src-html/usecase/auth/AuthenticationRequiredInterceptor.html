001package usecase.auth;
002
003import javax.annotation.Priority;
004import javax.inject.Inject;
005import javax.interceptor.AroundInvoke;
006import javax.interceptor.Interceptor;
007import javax.interceptor.InvocationContext;
008
009@Interceptor
010@AuthenticationRequired
011@Priority(Interceptor.Priority.APPLICATION+1)
012public class AuthenticationRequiredInterceptor {
013    @Inject private CurrentUser currentUser;
014
015    @AroundInvoke
016    public Object checkAdmin(InvocationContext invocationContext) throws Exception{
017        if(!currentUser.isLoggedIn()){
018            throw new AuthenticationRequiredException();
019        }
020        return invocationContext.proceed();
021    }
022}
