001package usecase.auth;
002
003import javax.crypto.SecretKeyFactory;
004import javax.crypto.spec.PBEKeySpec;
005import java.io.Serializable;
006import java.security.NoSuchAlgorithmException;
007import java.security.SecureRandom;
008import java.security.spec.InvalidKeySpecException;
009import java.security.spec.KeySpec;
010import java.util.Arrays;
011
012public class Pbkdf2PasswordHash implements Serializable {
013    public HashedPassword generate(String password, byte[] salt){
014        try {
015            KeySpec spec = new PBEKeySpec(password.toCharArray(), salt, 65536, 128);
016            SecretKeyFactory factory = SecretKeyFactory.getInstance("PBKDF2WithHmacSHA1");
017            byte[] hash = factory.generateSecret(spec).getEncoded();
018
019            return new HashedPassword(hash, salt);
020        } catch (InvalidKeySpecException | NoSuchAlgorithmException e) {
021            throw new RuntimeException(e);
022        }
023    }
024
025    public HashedPassword generate(String password){
026        return generate(password, generateSalt());
027    }
028
029    public boolean verify(String password, byte[] hashedPassword, byte[] salt){
030        return verify(password, new HashedPassword(hashedPassword, salt));
031    }
032
033    public boolean verify(String password, HashedPassword hashedPassword){
034        return hashedPassword.equals(generate(password, hashedPassword.getSalt()));
035    }
036
037    private static byte[] generateSalt(){
038        SecureRandom ss = new SecureRandom();
039        byte[] salt = new byte[16];
040        ss.nextBytes(salt);
041        return salt;
042    }
043
044    public static class HashedPassword{
045        private byte[] password;
046        private byte[] salt;
047
048        private HashedPassword(byte[] password, byte[] salt) {
049            this.password = password;
050            this.salt = salt;
051        }
052
053        public byte[] getPassword() {
054            return password;
055        }
056
057        public byte[] getSalt() {
058            return salt;
059        }
060
061        @Override
062        public boolean equals(Object o) {
063            if (this == o) return true;
064            if (o == null || getClass() != o.getClass()) return false;
065            HashedPassword that = (HashedPassword) o;
066            return Arrays.equals(password, that.password) && Arrays.equals(salt, that.salt);
067        }
068    }
069}
