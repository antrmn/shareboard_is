001package usecase.comment;
002
003import common.http.ParameterConverter;
004import common.http.interceptor.InterceptableServlet;
005import usecase.auth.AuthorizationConstraints;
006
007import javax.inject.Inject;
008import javax.servlet.ServletException;
009import javax.servlet.annotation.WebServlet;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpServletResponse;
012import java.io.IOException;
013
014import static usecase.auth.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
015
016@WebServlet("/editcomment")
017@AuthorizationConstraints(REQUIRE_AUTHENTICATION)
018public class EditCommentServlet extends InterceptableServlet {
019    @Inject private CommentService service;
020
021    @Override
022    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
023        doPost(request, response);
024    }
025
026    @Override
027    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
028        ParameterConverter converter = new ParameterConverter(request);
029        int commentId = converter.getIntParameter("id").orElse(0);
030        String text = request.getParameter("text");
031        service.editComment(commentId,text);
032
033        int parentPostId = service.getComment(commentId).getPostId();
034
035        response.sendRedirect( getServletContext().getContextPath() + "/usecase/post/" + parentPostId);
036    }
037}
