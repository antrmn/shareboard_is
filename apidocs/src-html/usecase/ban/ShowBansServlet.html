001package usecase.ban;
002
003import common.http.ParameterConverter;
004import common.http.interceptor.InterceptableServlet;
005import usecase.auth.AuthorizationConstraints;
006
007import javax.inject.Inject;
008import javax.servlet.ServletException;
009import javax.servlet.annotation.WebServlet;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpServletResponse;
012import java.io.IOException;
013import java.util.List;
014
015import static usecase.auth.AuthorizationConstraints.Types.ADMINS_ONLY;
016
017@WebServlet("/admin/showbans")
018@AuthorizationConstraints(ADMINS_ONLY)
019public class ShowBansServlet extends InterceptableServlet {
020    @Inject private BanService service;
021
022    @Override
023    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
024        ParameterConverter converter = new ParameterConverter(request);
025        int userId = converter.getIntParameter("userId").orElse(0);
026        List<BanDTO> bans = service.retrieveUserBan(userId);
027        request.setAttribute("bans", bans);
028        request.setAttribute("userId", userId);
029        request.getRequestDispatcher("/WEB-INF/views/crm/show-bans.jsp").forward(request, response);
030    }
031
032    @Override
033    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
034        doGet(request,response);
035    }
036}
