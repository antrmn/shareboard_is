001package usecase.ban;
002
003import domain.entity.Ban;
004import domain.entity.User;
005import domain.repository.GenericRepository;
006import domain.validation.BanExists;
007import domain.validation.UserExists;
008import usecase.auth.AdminsOnly;
009
010import javax.enterprise.context.ApplicationScoped;
011import javax.inject.Inject;
012import javax.transaction.Transactional;
013import javax.validation.constraints.Future;
014import javax.validation.constraints.NotNull;
015import java.time.Instant;
016import java.util.ArrayList;
017import java.util.List;
018
019@ApplicationScoped
020@Transactional
021public class BanService {
022    private GenericRepository genericRepository;
023
024    protected BanService(){}
025
026    @Inject
027    protected BanService(GenericRepository genericRepository){
028        this.genericRepository = genericRepository;
029    }
030
031
032    /**
033     * Aggiunge un ban dato l'id di un utente.
034     * @param date data di fine ban
035     * @param userId identificativo utente
036     * @return entit√† ban aggiunta
037     */
038    @AdminsOnly
039    public int addBan(@NotNull @Future Instant date, @UserExists int userId) {
040        Ban ban = new Ban();
041        ban.setEndTime(date);
042        User user = new User();
043        user.setId(userId);
044        ban.setUser(user);
045        return genericRepository.insert(ban).getId();
046    }
047
048    /**
049     * Rimuove un ban dato il suo id
050     * @param banId identificativo del ban
051     */
052    @AdminsOnly
053    public void removeBan(@BanExists int banId){
054        genericRepository.remove(genericRepository.findById(Ban.class, banId));
055    }
056
057    /**
058     * Ritorna la lista dei ban di un utente
059     * @param userId identificativo utente
060     * @return lista dei ban relativi all'utente
061     */
062    @AdminsOnly
063    public List<BanDTO> retrieveUserBan(@UserExists int userId){
064        User user = genericRepository.findById(User.class, userId);
065        List<BanDTO> bans = new ArrayList<>();
066        List<Ban> temp = user.getBans();
067
068        for(Ban b : temp){
069            BanDTO ban = new BanDTO(b.getId(), b.getEndTime(), b.getUser().getId());
070            bans.add(ban);
071        }
072        return bans;
073    }
074}
