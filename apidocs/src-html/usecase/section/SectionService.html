001package usecase.section;
002
003import domain.entity.Section;
004import domain.entity.User;
005import domain.repository.GenericRepository;
006import domain.repository.SectionRepository;
007import domain.validation.SectionExists;
008import domain.validation.UniqueSection;
009import media.MediaRepository;
010import media.ReadLimitExceededException;
011import media.validation.Image;
012import usecase.auth.AdminsOnly;
013import usecase.auth.CurrentUser;
014
015import javax.enterprise.context.ApplicationScoped;
016import javax.enterprise.context.RequestScoped;
017import javax.enterprise.inject.Produces;
018import javax.inject.Inject;
019import javax.inject.Named;
020import javax.transaction.Transactional;
021import javax.validation.constraints.NotBlank;
022import javax.validation.constraints.NotNull;
023import javax.validation.constraints.Size;
024import java.io.BufferedInputStream;
025import java.io.IOException;
026import java.time.Instant;
027import java.time.temporal.ChronoUnit;
028import java.util.List;
029import java.util.Map;
030import java.util.stream.Collectors;
031
032@ApplicationScoped
033@Transactional
034public class SectionService {
035    private SectionRepository sectionRepo;
036    private GenericRepository genericRepository;
037    private MediaRepository bcRepo;
038    private CurrentUser currentUser;
039
040    protected SectionService(){}
041
042    @Inject
043    protected SectionService(GenericRepository genericRepository, SectionRepository sectionRepository,
044                             MediaRepository bcRepo, CurrentUser currentUser){
045        this.genericRepository = genericRepository;
046        this.sectionRepo = sectionRepository;
047        this.bcRepo = bcRepo;
048        this.currentUser = currentUser;
049    }
050
051    /**
052     * Converte section in SectionPage.
053     * @param section sezione da convertire
054     * @return SectionPage con i dati di section
055     */
056    private SectionPage map(Section section){
057        User user = null;
058        if(currentUser.isLoggedIn()){
059            user = genericRepository.findById(User.class,currentUser.getId());
060        }
061        return SectionPage.builder().id(section.getId())
062                .name(section.getName())
063                .picture(section.getPicture())
064                .description(section.getDescription())
065                .banner(section.getBanner())
066                .nFollowers(section.getFollowCount())
067                .isFollowed(user != null && section.getFollow(user) != null).build();
068    }
069
070    /**
071     * Cancella una sezione dato il suo id
072     * @param id l'id di una sezione esistente
073     */
074    @AdminsOnly
075    public void delete(@SectionExists int id){
076        genericRepository.remove(genericRepository.findById(Section.class, id));
077    }
078
079    /**
080     * Crea una nuova sezione e ne restituisce l'id
081     * @param name Nome della sezione
082     * @param description descrizione della sezione
083     * @param picture stream relativo alla foto della sezione
084     * @param banner stream relativo al banner della sezione
085     * @return id della sezione creata
086     */
087    @AdminsOnly
088    public int newSection(@NotBlank @Size(max=50) @UniqueSection String name,
089                          @Size(max=255) String description,
090                          @Image BufferedInputStream picture,
091                          @Image BufferedInputStream banner) {
092        Section s = new Section();
093        s.setName(name);
094        s.setDescription(description);
095        try {
096            if (picture != null)
097                s.setPicture(bcRepo.insert(picture));
098            if (banner != null)
099                s.setBanner(bcRepo.insert(banner));
100        } catch (ReadLimitExceededException e) {
101            throw new IllegalArgumentException("Il file non deve superare i 5MB");
102        } catch (IOException e) {
103            throw new RuntimeException(e);
104        }
105        return genericRepository.insert(s).getId();
106    }
107
108    /**
109     * Ritorna una lista di tutte le sezioni esistenti
110     * @return lista di sezioni
111     */
112    public List<SectionPage> showSections(){
113        return genericRepository.findAll(Section.class).stream().map(this::map).collect(Collectors.toList());
114    }
115
116    /**
117     * Ritorna una mappa di tutte le sezioni esistenti
118     * @return mappa la cui chiave è l'id della sezione e il valore un entita PostPage che ne contiene i dati
119     */
120    @Named("sections")
121    @RequestScoped
122    @Produces
123    public Map<Integer,SectionPage> getSectionsMap(){
124        return genericRepository.findAll(Section.class).stream().map(this::map)
125                .collect(Collectors.toMap(SectionPage::getId, section -> section));
126    }
127
128    /**
129     * Ritorna un entità sezione dato un certo id
130     * @param id id di una sezione esistente
131     * @return entita SectionPage che contiene i dati della sezione
132     */
133    public SectionPage showSection(@SectionExists int id){
134       Section s =  genericRepository.findById(Section.class, id);
135       return map(s);
136    }
137
138    /**
139     * Ritorna un entità sezione con un nome specifico
140     * @param sectionName nome di una sezione esistente
141     * @return entita SectionPage che contiene i dati della sezione
142     */
143    public SectionPage getSection(@NotNull @SectionExists String sectionName){
144        Section s = genericRepository.findByNaturalId(Section.class,sectionName);
145        return map(s);
146    }
147
148    /**
149     * Ritorna una lista delle sezioni con più follows
150     * @return lista di sezioni
151     */
152    @Produces
153    @RequestScoped
154    @Named("topSections")
155    public List<SectionPage> getTopSections(){
156        return sectionRepo.getMostFollowedSections().stream().map(this::map).collect(Collectors.toList());
157    }
158
159    /**
160     * Ritorna una lista delle sezioni con più follows negli ultimi 7 giorni
161     * @return lista di sezioni
162     */
163    @Produces
164    @RequestScoped
165    @Named("trendingSections")
166    public List<SectionPage> getTrendingSections(){
167        return sectionRepo.getMostFollowedSections(Instant.now().minus(7, ChronoUnit.DAYS))
168                .stream().map(this::map).collect(Collectors.toList());
169    }
170}
