001package media;
002
003
004import java.io.*;
005import java.nio.file.Files;
006import java.nio.file.Path;
007import java.util.UUID;
008
009public class MediaRepository implements Serializable{
010
011    private Path uploadRoot = Path.of(System.getProperty("openejb.home"), "uploads");
012    private final long sizeLimit = 5*1024*1024; //todo: parametrizza nel metodo insert
013
014    /**
015     *  Costruttore vuoto
016     * @return nuova istanza di MediaRepository
017     */
018    public MediaRepository(){}
019
020    /**
021     *  Costruttore con parametro per settare uploadRoot
022     * @param uploadRoot oggetto Path per settare uploadRoot
023     * @return nuova istanza di MediaRepository
024     */
025    public MediaRepository(Path uploadRoot){
026        this.uploadRoot = uploadRoot;
027    }
028
029    /**
030     *  Salva un file nel filesystem e ne restituisce il nome
031     * @param stream stream di dati
032     * @param filename nome del file da salvare
033     * @throws ReadLimitExceededException se il file supera i 5MB
034     * @throws IOException
035     * @return nuova istanza di MediaRepository
036     */
037    public String insert(InputStream stream, String filename) throws IOException {
038        Files.createDirectories(uploadRoot);
039        File file = new File(uploadRoot.toFile(), filename);
040        Files.copy(new LimitedInputStream(stream,sizeLimit), file.toPath());
041        return filename;
042    }
043
044    /**
045     *  Salva un file nel filesystem e ne restituisce un nome univoco generato in maniera casuale
046     * @param stream stream di dati
047     * @throws IOException
048     * @return nuova istanza di MediaRepository
049     */
050    public String insert(InputStream stream) throws IOException {
051        return insert(stream, UUID.randomUUID().toString());
052    }
053
054    /**
055     *  Rimuove un file dal filesystem dato un nome se esiste
056     * @param filename nome del file
057     * @throws IOException
058     */
059    public void remove(String filename) throws IOException {
060        boolean successful = Files.deleteIfExists(uploadRoot.resolve(filename));
061        // if (!successful) { ???? }
062    }
063
064    /**
065     *  Restituisce un file dal filesystem dato un nome se esiste altrimenti restituisce null
066     * @param filename nome del file
067     * @return stream di dati del file o null
068     */
069    public InputStream get(String filename){
070        File file = new File(uploadRoot.toFile(), filename);
071        if (!file.exists() || !file.isFile())
072            return null;
073        try {
074            return new FileInputStream(file);
075        } catch (FileNotFoundException e) {
076            return null;
077        }
078    }
079
080}
