3   import java.lang.annotation.Annotation;
4   import java.lang.reflect.Modifier;
5   import java.lang.reflect.ParameterizedType;
6   import java.util.concurrent.ConcurrentHashMap;
7   import java.util.concurrent.ConcurrentMap;
8   
9   public final class ServletInterceptorFactory {
10      private ServletInterceptorFactory(){}
11  
12      private static final
13      ConcurrentMap<Class<? extends Annotation>, Class<? extends ServletInterceptor<?>>> classes =
14              new ConcurrentHashMap<>();
15  
16      public static <T extends Annotation> ServletInterceptor<T> instantiate(T annotation){
17          Class<? extends ServletInterceptor<?>> clazz = ServletInterceptorFactory.classes.get(annotation.annotationType());
18          if(clazz == null)
19              return null;
20  
21          ServletInterceptor<T> interceptor;
22          try {
23              //noinspection unchecked
24              interceptor = (ServletInterceptor<T>) clazz.getConstructor().newInstance();
25          } catch (ReflectiveOperationException e) {
26              throw new RuntimeException(e);
27          }
28  
29          interceptor.init(annotation);
30          return interceptor;
31      }
32  
33      public static void register(Class<? extends ServletInterceptor<? extends Annotation>> interceptorClass){
34          if(Modifier.isAbstract(interceptorClass.getModifiers())){
35              throw new IllegalArgumentException("Interceptor class must be concrete");
36          }
37          if(interceptorClass.getTypeParameters().length > 0){
38              throw new IllegalArgumentException("Interceptor class must be non-generic");
39          }
40  
41          //get ServletInterceptor.class immediate subclass
42          Class<?> clazz = interceptorClass;
43          while(clazz.getSuperclass() != ServletInterceptor.class){
44              clazz = clazz.getSuperclass();
45          }
46          //get ServletInterceptor.class as ParameterizedType object
47          ParameterizedType genericSuperclass = (ParameterizedType) clazz.getGenericSuperclass();
48  
49          @SuppressWarnings("unchecked")
50          Class<? extends Annotation> actualTypeArgument =
51                  (Class<? extends Annotation>) genericSuperclass.getActualTypeArguments()[0];
52  
53          Class<?> oldValue = classes.putIfAbsent(actualTypeArgument, interceptorClass);
54          if(oldValue != null){
55              throw new IllegalArgumentException("There already is a registered Servlet Interceptor with annotation "
56                      + actualTypeArgument.getName());
57          }
58      }
59  }
