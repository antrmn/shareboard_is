3   import http.controller.interceptor.AuthorizationConstraints;
4   import http.controller.interceptor.ForwardOnError;
5   import http.util.ParameterConverter;
6   import http.util.interceptor.InterceptableServlet;
7   import service.SectionService;
8   import service.dto.SectionPage;
9   
10  import javax.inject.Inject;
11  import javax.servlet.ServletException;
12  import javax.servlet.annotation.MultipartConfig;
13  import javax.servlet.annotation.WebServlet;
14  import javax.servlet.http.HttpServletRequest;
15  import javax.servlet.http.HttpServletResponse;
16  import javax.servlet.http.Part;
17  import java.io.BufferedInputStream;
18  import java.io.IOException;
19  
20  import static http.controller.interceptor.AuthorizationConstraints.Types.ADMINS_ONLY;
21  
22  @WebServlet("/admin/editsection")
23  @MultipartConfig
24  @AuthorizationConstraints(ADMINS_ONLY)
25  public class EditSectionServlet extends InterceptableServlet {
26      @Inject private SectionService service;
27  
28      private static final String EDIT_SECTION_PAGE = "/WEB-INF/views/crm/edit-section.jsp";
29      private static final int MAX_FILE_SIZE = 5 * 1024 * 1024;
30  
31      @Override
32      protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
33          ParameterConverter converter = new ParameterConverter(request);
34          int sectionId = converter.getIntParameter("sectionId").orElse(0);
35          SectionPage section = service.showSection(sectionId);
36          request.setAttribute("section", section);
37          request.getRequestDispatcher(EDIT_SECTION_PAGE).forward(request, response);
38      }
39  
40      @Override
41      @ForwardOnError(EDIT_SECTION_PAGE)
42      protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
43          ParameterConverter converter = new ParameterConverter(request);
44          int sectionId = converter.getIntParameter("sectionId").orElse(0);
45          String description = request.getParameter("description");
46          Part picture = request.getPart("picture");
47          Part banner = request.getPart("banner");
48  
49          BufferedInputStream buffPicture = null;
50          BufferedInputStream buffBanner = null;
51          if(picture != null && picture.getSize() < MAX_FILE_SIZE) {
52              if (picture.getSize()>0)
53                  buffPicture = new BufferedInputStream(picture.getInputStream());
54          }else{
55              throw new IllegalArgumentException("Il file non deve superare i 5MB");
56          }
57  
58          if(banner != null && banner.getSize() < MAX_FILE_SIZE) {
59              if (banner.getSize()>0)
60                  buffBanner = new BufferedInputStream(banner.getInputStream());
61          }else{
62              throw new IllegalArgumentException("Il file non deve superare i 5MB");
63          }
64  
65          SectionPage sectionPage = new SectionPage(sectionId,null,description,picture.getName(),banner.getName(),0,false); //false?
66          service.editSection(sectionPage,sectionId,buffPicture,buffBanner);
67          response.sendRedirect(request.getContextPath()+"/admin/showsections");
68      }
69  }
