25  
26  public class JSONErrorInterceptor extends ServletInterceptor<JSONError> {
27  
28      @Override
29      protected void init(JSONError annotation) {
30  
31      }
32  
33      @Override
34      public void handle(HttpServletRequest req, HttpServletResponse resp, HttpServletBiConsumer next) throws ServletException, IOException {
35          try{
36              next.handle(req,resp);
37          } catch (IllegalArgumentException e) {
38              sendJSONError(resp, SC_BAD_REQUEST, e.getMessage());
39          } catch (ConstraintViolationException e) {
40              List<String> collect = e.getConstraintViolations().stream()
41                      .map(ConstraintViolation::getMessage)
42                      .collect(Collectors.toList());
43  
44              sendJSONError(resp,SC_BAD_REQUEST,collect);
45          } catch (AuthenticationRequiredException e) {
46              sendJSONError(resp,SC_UNAUTHORIZED, e.getMessage());
47          } catch (BannedUserException e){
48              if (e.getDuration() != null){
49                  String end  = DateTimeFormatter.ISO_INSTANT.format(e.getDuration());
50                  sendJSONError(resp,SC_FORBIDDEN, "Sei bannato fino a " + end);
51              } else {
52                  sendJSONError(resp, SC_FORBIDDEN);
53              }
54          } catch (AuthorizationException e) {
55              sendJSONError(resp,SC_FORBIDDEN, e.getMessage());
56          } catch (RuntimeException | ServletException | IOException e){
57              sendJSONError(resp,SC_INTERNAL_SERVER_ERROR, "Internal server error");
58              req.getServletContext().log(e.getMessage(), e);
59          }
60      }
61  
62      private void sendJSONError(HttpServletResponse resp, int code, Collection<String> messages) throws IOException {
63          JsonObject errors = Json.createObjectBuilder().add("errors", Json.createArrayBuilder(messages)).build();
64          resp.setStatus(code);
65          resp.setContentType("application/json");
66          PrintWriter writer = resp.getWriter();
67          writer.print(errors.toString());
68          writer.flush();
69  
70      }
71  
72      private void sendJSONError(HttpServletResponse resp, int code, String message) throws IOException {
73          List<String> singleton = message == null ? Collections.emptyList() : List.of(message);
74          sendJSONError(resp, code, singleton);
75      }
76  
77      private void sendJSONError(HttpServletResponse resp, int code) throws IOException{
78          sendJSONError(resp,code,Collections.emptyList());
79      }
80  
81      @Override
82      public int priority() {
83          return 2;
84      }
85  }
