10  import javax.inject.Inject;
11  import javax.servlet.ServletException;
12  import javax.servlet.annotation.MultipartConfig;
13  import javax.servlet.annotation.WebServlet;
14  import javax.servlet.http.HttpServletRequest;
15  import javax.servlet.http.HttpServletResponse;
16  import javax.servlet.http.Part;
17  import java.io.BufferedInputStream;
18  import java.io.IOException;
19  
20  import static http.controller.interceptor.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
21  
22  @WebServlet("/newpost")
23  @MultipartConfig
24  @AuthorizationConstraints(REQUIRE_AUTHENTICATION)
25  public class NewPostServlet extends InterceptableServlet {
26      @Inject private PostService service;
27      @Inject private SectionService sectionService;
28  
29      private static final String NEW_POST_PAGE = "/WEB-INF/views/section/create-post.jsp";
30      private static final int MAX_FILE_SIZE = 5 * 1024 * 1024;
31      @Override
32      protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
33          request.getRequestDispatcher(NEW_POST_PAGE).forward(request, response);
34      }
35  
36      @Override
37      @ForwardOnError("/WEB-INF/views/section/create-post.jsp")
38      protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
39          ParameterConverter converter = new ParameterConverter(request);
40          int sectionId = converter.getIntParameter("section").orElse(0);
41          String sectionName = sectionService.showSection(sectionId).getName();
42  
43          String title = request.getParameter("title");
44          String type = request.getParameter("type");
45          String content = request.getParameter("content");
46          Part picture = request.getPart("picture");
47  
48          int newPostId;
49          if(type.equalsIgnoreCase("text")){
50              newPostId = service.newPost(title,content,sectionName);
51          }else{
52              if(picture != null && picture.getSize() > MAX_FILE_SIZE) {
53                  throw new IllegalArgumentException("Il file non deve superare i 5MB");
54              }
55  
56              if(picture != null && picture.getSize() == 0) {
57                  throw new IllegalArgumentException("Immagine mancante");
58              }
59              BufferedInputStream buffPicture = new BufferedInputStream(picture.getInputStream());
60              newPostId = service.newPost(title,buffPicture, picture.getSize(), sectionName);
61  
62          }
63          response.sendRedirect(getServletContext().getContextPath() + "/post/" + newPostId); //potrebbe mostrare postId = 0?
64      }
65  }
