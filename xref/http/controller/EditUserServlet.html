8   import service.dto.UserEditPage;
9   import service.dto.UserProfile;

11  import javax.inject.Inject;
12  import javax.servlet.ServletException;
13  import javax.servlet.annotation.MultipartConfig;
14  import javax.servlet.annotation.WebServlet;
15  import javax.servlet.http.HttpServletRequest;
16  import javax.servlet.http.HttpServletResponse;
17  import javax.servlet.http.Part;
18  import java.io.BufferedInputStream;
19  import java.io.IOException;
20  
21  import static http.controller.interceptor.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
22  
23  @WebServlet("/edituser")
24  @MultipartConfig
25  @AuthorizationConstraints(REQUIRE_AUTHENTICATION)
26  public class EditUserServlet extends InterceptableServlet {
27      @Inject private UserService service;
28  
29      private static final String EDIT_USER_PAGE = "/WEB-INF/views/edit-user.jsp";
30      private static final int MAX_FILE_SIZE = 5 * 1024 * 1024;
31  
32      @Override
33      protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
34          ParameterConverter converter = new ParameterConverter(request);
35          int userId = converter.getIntParameter("id").orElse(0);
36          UserProfile user = service.getUser(userId);
37          request.setAttribute("user", user);
38          request.getRequestDispatcher("/WEB-INF/views/edit-user.jsp").forward(request, response);
39      }
40  
41      @Override
42      @ForwardOnError(EDIT_USER_PAGE)
43      protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
44          ParameterConverter converter = new ParameterConverter(request);
45          int userId = converter.getIntParameter("id").orElse(0);
46          String email = request.getParameter("email");
47          String description = request.getParameter("description");
48          Part picture = request.getPart("picture");
49          String pass = request.getParameter("pass");
50          String pass2 = request.getParameter("pass2");
51  
52          if((!pass.isEmpty() || !pass2.isEmpty()) && !pass.equals(pass2))
53              throw new IllegalArgumentException("Le password devono coincidere");
54  
55          BufferedInputStream buffPicture = null;
56          if(picture != null && picture.getSize() < MAX_FILE_SIZE) {
57              if (picture.getSize()>0)
58                  buffPicture = new BufferedInputStream(picture.getInputStream());
59          }else{
60              throw new IllegalArgumentException("Il file non deve superare i 5MB");
61          }
62  
63  
64         // BufferedInputStream buffPicture = new BufferedInputStream(picture.getInputStream());
65          UserEditPage userEditPage = new UserEditPage(userId,description,email,buffPicture,pass);
66          service.edit(userEditPage,userId);
67  
68          response.sendRedirect(request.getContextPath() + "/u/" + service.getUsernameById(userId));
69      }
70  }
