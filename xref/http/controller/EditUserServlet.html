13  import javax.inject.Inject;
14  import javax.servlet.ServletException;
15  import javax.servlet.annotation.MultipartConfig;
16  import javax.servlet.annotation.WebServlet;
17  import javax.servlet.http.HttpServletRequest;
18  import javax.servlet.http.HttpServletResponse;
19  import javax.servlet.http.Part;
20  import java.io.BufferedInputStream;
21  import java.io.IOException;
22  
23  import static http.controller.interceptor.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
24  
25  @WebServlet("/edituser")
26  @MultipartConfig
27  @AuthorizationConstraints(REQUIRE_AUTHENTICATION)
28  public class EditUserServlet extends InterceptableServlet {
29      @Inject private UserService service;
30      @Inject private CurrentUser currentUser;
31  
32      private static final String EDIT_USER_PAGE = "/WEB-INF/views/edit-user.jsp";
33  
34      @Override
35      protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
36          ParameterConverter converter = new ParameterConverter(request);
37          int userId = converter.getIntParameter("id").orElse(0);
38          if(!currentUser.isAdmin() && currentUser.getId() != userId)
39              throw new AuthorizationException();
40          UserProfile user = service.getUser(userId);
41          request.setAttribute("user", user);
42          request.getRequestDispatcher("/WEB-INF/views/edit-user.jsp").forward(request, response);
43      }
44  
45      @Override
46      @ForwardOnError(EDIT_USER_PAGE)
47      protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
48          ParameterConverter converter = new ParameterConverter(request);
49          int userId = converter.getIntParameter("id").orElse(0);
50          String email = request.getParameter("email");
51          String description = request.getParameter("description");
52          Part picture = request.getPart("picture");
53          String pass = request.getParameter("pass");
54          String pass2 = request.getParameter("pass2");
55  
56          if((!pass.isEmpty() || !pass2.isEmpty()) && !pass.equals(pass2))
57              throw new IllegalArgumentException("Le password devono coincidere");
58  
59          BufferedInputStream stream = picture.getSize() > 0 ? new BufferedInputStream(picture.getInputStream()) : null;
60          UserEditPage userEditPage = new UserEditPage(description,email,stream ,pass);
61          service.edit(userEditPage,userId);
62  
63          response.sendRedirect(request.getContextPath() + "/u/" + service.getUsernameById(userId));
64      }
65  }
