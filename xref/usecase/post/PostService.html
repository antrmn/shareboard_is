54      /**
55       * Converte post in PostPage.
56       * @param post post da convertire
57       * @return PostPage con i dati di post
58       */
59      private PostPage mapPost(Post post){
60          User user = null;
61          if(currentUser.isLoggedIn())
62              user = genericRepository.findById(User.class,currentUser.getId());
63  
64          //converte persistence.Post.Type in usecase.post.PostType (ew)
65          PostType postType = post.getType() == IMG ? PostType.IMG : PostType.TEXT;
66  
67          return PostPage.builder()
68                  .id(post.getId())
69                  .title(post.getTitle())
70                  .vote(post.getVote(user) == null ? 0 : post.getVote(user).getVote())
71                  .votes(post.getVotesCount() == null ? 0 : post.getVotesCount())
72                  .sectionName(post.getSection().getName())
73                  .authorName(post.getAuthor().getUsername())
74                  .sectionId(post.getSection().getId())
75                  .authorId(post.getAuthor().getId())
76                  .content(post.getContent())
77                  .creationDate(post.getCreationDate() == null ? Instant.now() : post.getCreationDate())
78                  .nComments(post.getCommentCount())
79                  .type(postType)
80                  .build();
81      }

93      /**
94       * Ritorna una lista di post che rispettano determinati parametri
95       * @param form entità con i parametri di ricerca
96       * @return lista di PostPage
97       */
98      @Transactional
99      public List<PostPage> loadPosts(PostSearchForm form){
100         final int elementsPerPage = 10;
101         PostRepository.PostFinder finder = postRepo.getFinder();
102 
103         finder.limit(elementsPerPage);
104 
105         int page = form.getPage() > 0 ? form.getPage() : 1;
106         finder.offset((page == 1) ? 0 : (page-1) * elementsPerPage+1);
107 
108         if(form.getContent() != null) {
109             finder.byContent(form.getContent());
110         }
111         if(form.isIncludeBody()){
112             finder.includeBody();
113         }
114         if(form.getSectionName() != null){
115             Section section = genericRepository.findByNaturalId(Section.class, form.getSectionName());
116             if(section != null){
117                 finder.bySection(section);
118             }
119         }
120         if(form.getAuthorName() != null){
121             User user = genericRepository.findByNaturalId(User.class, form.getAuthorName());
122             if(user != null){
123                 finder.byAuthor(user);
124             }
125         }
126         if(form.isOnlyFollow() && currentUser.isLoggedIn()){
127             finder.joinUserFollows(genericRepository.findById(User.class,currentUser.getId()));
128         }
129         if(form.getPostedAfter() != null){
130             finder.postedAfter(form.getPostedAfter());
131         }
132         if(form.getPostedBefore() != null){
133             finder.postedBefore(form.getPostedBefore());
134         }
135 
136         //null-safe switch
137         switch(form.getOrderBy() != null ? form.getOrderBy() : NEWEST){
138             case NEWEST: default: {
139                 finder.getNewest();
140                 break;
141             }
142             case OLDEST:{
143                 finder.getOldest();
144                 break;
145             }
146             case MOSTVOTED:{
147                 finder.getMostVoted();
148                 break;
149             }
150         }
151         return finder.getResults().stream().map(this::mapPost).collect(Collectors.toList());
152     }

154     /**
155      * Elimina un post dato il suo id
156      * @param id identificativo del post
157      */
158     @AuthenticationRequired //TODO: check autore post?
159     @DenyBannedUsers
160     public void delete(@PostExists int id){
161         Post post = genericRepository.findById(Post.class,id);
162         if(currentUser.getId() != post.getAuthor().getId() && !currentUser.isAdmin())
163             throw new AuthorizationException();
164         genericRepository.remove(genericRepository.findById(User.class,id));
165     }

167     /**
168      * Aggiunge un post ad una sezione
169      * @param title titolo del post
170      * @param body testo del post
171      * @param sectionName nome della sezione
172      * @return identificativo del nuovo post creato
173      */
174     @AuthenticationRequired
175     @DenyBannedUsers
176     public int newPost(@NotBlank String title,
177                        @Size(max=65535) String body,
178                        @NotNull @SectionExists String sectionName){
179         User user = genericRepository.findByNaturalId(User.class, currentUser.getUsername());
180         Section section = genericRepository.findByNaturalId(Section.class,sectionName);
181 
182 
183         Post post = new Post();
184         post.setTitle(title);
185         post.setContent(body == null || body.isBlank() ? "" : body);
186         post.setAuthor(user);
187         post.setSection(section);
188         post.setType(TEXT);
189 
190         return genericRepository.insert(post).getId();
191     }

193     /**
194      * Aggiunge un post ad una sezione
195      * @param title titolo del post
196      * @param content file di tipo immagine
197      * @param sectionName nome della sezione
198      * @return identificativo del nuovo post creato
199      */
200     @AuthenticationRequired
201     @DenyBannedUsers
202     public int newPost(@NotBlank @Size(max=255) String title,
203                        @NotNull @Image BufferedInputStream content,
204                        @NotNull @SectionExists String sectionName){
205         User user = genericRepository.findByNaturalId(User.class,currentUser.getUsername());
206         Section section = genericRepository.findByNaturalId(Section.class,sectionName);
207 
208         String fileName;
209         try {
210             fileName = bcRepo.insert(content);
211         } catch (ReadLimitExceededException e){
212             throw new IllegalArgumentException("Il file non deve superare i 5MB");
213         } catch (IOException e) {
214             throw new RuntimeException(e);
215         }
216 
217         Post post = new Post();
218         post.setTitle(title);
219         post.setAuthor(user);
220         post.setContent(fileName);
221         post.setSection(section);
222         post.setType(IMG);
223 
224         return genericRepository.insert(post).getId(); //in caso di errore il file resta
225     }
