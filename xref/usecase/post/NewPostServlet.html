9   import javax.inject.Inject;
10  import javax.servlet.ServletException;
11  import javax.servlet.annotation.MultipartConfig;
12  import javax.servlet.annotation.WebServlet;
13  import javax.servlet.http.HttpServletRequest;
14  import javax.servlet.http.HttpServletResponse;
15  import javax.servlet.http.Part;
16  import java.io.BufferedInputStream;
17  import java.io.IOException;
18  
19  import static usecase.auth.AuthorizationConstraints.Types.REQUIRE_AUTHENTICATION;
20  
21  @WebServlet("/newpost")
22  @MultipartConfig
23  @AuthorizationConstraints(REQUIRE_AUTHENTICATION)
24  class NewPostServlet extends InterceptableServlet {
25      @Inject private PostService service;
26      @Inject private SectionService sectionService;
27  
28      private static final String NEW_POST_PAGE = "/WEB-INF/views/section/create-post.jsp";
29  
30      @Override
31      protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
32          request.getRequestDispatcher(NEW_POST_PAGE).forward(request, response);
33      }
34  
35      @Override
36      @ForwardOnError("/WEB-INF/views/section/create-post.jsp")
37      protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
38          ParameterConverter converter = new ParameterConverter(request);
39          int sectionId = converter.getIntParameter("section").orElse(0);
40          String sectionName = sectionService.showSection(sectionId).getName();
41  
42          String title = request.getParameter("title");
43          String type = request.getParameter("type");
44          String content = request.getParameter("content");
45          Part picture = request.getPart("picture");
46  
47          int newPostId;
48          if(type.equalsIgnoreCase("text")){
49              newPostId = service.newPost(title,content,sectionName);
50          }else{
51              BufferedInputStream stream = picture.getSize() > 0 ? new BufferedInputStream(picture.getInputStream()) : null;
52              newPostId = service.newPost(title,stream, sectionName);
53          }
54          response.sendRedirect(getServletContext().getContextPath() + "/post/" + newPostId);
55      }
56  }
