10  import javax.servlet.annotation.WebServlet;

13  import javax.servlet.http.Part;
14  import java.io.BufferedInputStream;
15  import java.io.IOException;
16  
17  import static usecase.auth.AuthorizationConstraints.Types.ADMINS_ONLY;
18  
19  @WebServlet("/admin/newsection")
20  @MultipartConfig
21  @AuthorizationConstraints(ADMINS_ONLY)
22  class NewSectionServlet extends InterceptableServlet {
23      @Inject private SectionService service;
24  
25      private static final String NEW_SECTION_PAGE = "/WEB-INF/views/crm/create-section.jsp";
26  
27      @Override
28      protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
29          request.getRequestDispatcher("/WEB-INF/views/crm/create-section.jsp").forward(request, response);
30      }
31  
32      @Override
33      @ForwardOnError(NEW_SECTION_PAGE)
34      protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
35          String name = request.getParameter("name");
36          String description = request.getParameter("description");
37          Part picture = request.getPart("picture");
38          Part banner = request.getPart("banner");
39  
40          BufferedInputStream streamPicture = picture.getSize() > 0 ? new BufferedInputStream(picture.getInputStream()) : null;
41          BufferedInputStream streamBanner = banner.getSize() > 0 ? new BufferedInputStream(picture.getInputStream()) : null;
42  
43          service.newSection(name, description ,streamPicture,streamBanner);
44          response.sendRedirect(getServletContext().getContextPath()+"/admin/showsections");
45      }
46  }
