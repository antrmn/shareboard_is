7   import javax.inject.Inject;
8   import javax.servlet.ServletException;
9   import javax.servlet.annotation.WebServlet;
10  import javax.servlet.http.HttpServletRequest;
11  import javax.servlet.http.HttpServletResponse;
12  import java.io.IOException;
13  
14  import static usecase.auth.AuthorizationConstraints.Types.ADMINS_ONLY;
15  
16  @WebServlet(name = "DeleteSectionServlet", value = "/deleteSection")
17  @AuthorizationConstraints(ADMINS_ONLY)
18  class DeleteSectionServlet extends InterceptableServlet {
19      @Inject private SectionService service;
20  
21      @Override
22      protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
23          ParameterConverter converter = new ParameterConverter(request);
24          int sectionId = converter.getIntParameter("sectionId").orElse(0);
25          service.delete(sectionId);
26          response.sendRedirect(getServletContext().getContextPath() + "/admin/showsections");
27      }
28  
29      @Override
30      protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
31          doGet(request,response);
32      }
33  }
