17  @WebServlet("/admin/toggleAdmin")
18  @AuthorizationConstraints(ADMINS_ONLY)
19  class ToggleAdminServlet extends InterceptableServlet {
20  
21      @Inject private UserService service;
22  
23      @Override
24      @JSONError
25      protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
26          ParameterConverter converter = new ParameterConverter(req);
27          int userId = converter.getIntParameter("userId").orElse(0);
28          service.toggleAdmin(userId);
29      }
