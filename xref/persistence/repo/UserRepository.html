1   package persistence.repo;

3   import org.hibernate.Session;
4   import org.hibernate.SimpleNaturalIdLoadAccess;
5   import persistence.model.User;
6   
7   import javax.persistence.EntityManager;
8   import javax.persistence.PersistenceContext;
9   import java.io.Serializable;
10  
11  public class UserRepository implements Serializable {
12  
13      @PersistenceContext
14      protected EntityManager em;
15  
16      public User getByName(String name) {
17          return getByName(name, false);
18      }
19  
20      public User getByName(String name, boolean getReference) {
21          SimpleNaturalIdLoadAccess<User> user = em.unwrap(Session.class).bySimpleNaturalId(User.class);
22          return getReference ? user.getReference(name) : user.load(name);
23      }
24  
25      public User getByEmail(String email) {
26          return em.createQuery("from User u where u.email=:email", User.class).setParameter("email", email).getSingleResult();
27      }
28  }
