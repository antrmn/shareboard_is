5   import persistence.model.Section;

7   import javax.persistence.EntityManager;
8   import javax.persistence.PersistenceContext;
9   import java.io.Serializable;
10  import java.time.Instant;
11  import java.util.List;
12  
13  public class SectionRepository implements Serializable {
14  
15      @PersistenceContext
16      protected EntityManager em;
17  
18      public Section getByName(String name){
19          return getByName(name, false);
20      }
21  
22      public Section getByName(String name, boolean getReference){
23          SimpleNaturalIdLoadAccess<Section> user = em.unwrap(Session.class).bySimpleNaturalId(Section.class);
24          return getReference ? user.getReference(name) : user.load(name);
25      }
26  
27      public List<Section> getMostFollowedSections(){
28          return em.createQuery("from Section s order by follows.size desc", Section.class).getResultList();
29      }
30  
31      public List<Section> getMostFollowedSections(Instant after){
32          return em.createQuery(
33                  "select s from Section s left join s.follows f where f.followDate >= :after " +
34                  "group by s " +
35                  "order by count(f) desc", Section.class)
36                  .setParameter("after",after).getResultList();
37      }
38  }
