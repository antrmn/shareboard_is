23  @TransactionManagement(TransactionManagementType.BEAN)
24  class FlywayMigrator {
25  
26      private Path uploadRoot = Path.of(System.getProperty("openejb.home"), "uploads");
27  
28      @Resource(name = "jdbc/shareboardDB")
29      private DataSource dataSource;
30  
31      public void postConstruct(@Observes @Initialized(ApplicationScoped.class) Object o) throws IOException {
32          if (dataSource == null) {
33              throw new RuntimeException("Cannot migrate, no datasource configured!");
34          }
35          FluentConfiguration config = Flyway.configure()
36                  .dataSource(dataSource)
37                  .validateOnMigrate(false)
38                  .locations("classpath:db/migration");
39          Flyway flyway = new Flyway(config);
40          MigrateResult migrate = flyway.migrate();
41          if(!migrate.migrations.isEmpty()){
42              Files.createDirectories(uploadRoot);
43  
44              InputStream resourceAsStream = getClass().getClassLoader().getResourceAsStream("sample/pictures.zip");
45              if(resourceAsStream == null){
46                  throw new RuntimeException("pictures.zip not found");
47              }
48  
49              try(ZipInputStream zipStream = new ZipInputStream(resourceAsStream)){
50                  ZipEntry entry = zipStream.getNextEntry();
51                  byte[] buffer = new byte[1024];
52                  while (entry != null) {
53                      File newFile = new File(uploadRoot.toFile(), entry.getName());
54                      // fix for Windows-created archives
55                      File parent = newFile.getParentFile();
56                      if (!parent.isDirectory() && !parent.mkdirs()) {
57                          throw new IOException("Failed to create directory " + parent);
58                      }
59  
60                      // write file content
61                      try(FileOutputStream fos = new FileOutputStream(newFile)){
62                          int len;
63                          while ((len = zipStream.read(buffer)) > 0) {
64                              fos.write(buffer, 0, len);
65                          }
66                          entry = zipStream.getNextEntry();
67                      }
68                  }
69              }
70  
71          }
72      }
73  }
