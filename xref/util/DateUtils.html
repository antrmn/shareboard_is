1   package util;

3   import java.time.Instant;
4   import java.time.LocalDate;
5   import java.time.ZoneId;
6   import java.time.temporal.ChronoUnit;
7   
8   public final class DateUtils {
9       public static String printTimeSince(Instant then){
10          long n;
11          if ((n = Math.abs(Instant.now().until(then, ChronoUnit.SECONDS))) < 60){
12              return n == 1 ? n + " secondo" : n + " secondi";
13          } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.MINUTES))) < 60){
14              return n == 1 ? n + " minuto" : n + " minuti";
15          } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.HOURS))) < 24){
16              return n == 1 ? n + " ora" : n + " ore";
17          } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.DAYS))) < 30) {
18              return n == 1 ? n + " giorno" : n + " giorni";
19          } else if ((n = Math.abs(Instant.now().until(then, ChronoUnit.DAYS))) < 365) {
20              n /= 30;
21              return n == 1 ? n + " mese" : n + " mesi";
22          } else {
23              int thenYear = then.atZone(ZoneId.systemDefault()).toLocalDate().getYear();
24              n = Math.abs(thenYear - LocalDate.now().getYear());
25              return n == 1 ? n + " anno" : n + " anni";
26          }
27      }
28  }
