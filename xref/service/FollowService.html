15  @ApplicationScoped
16  @Transactional
17  public class FollowService {
18      @Inject private GenericRepository genericRepository;
19      @Inject private CurrentUser currentUser;
20  
21      /**
22       * Permette di seguire una sezione
23       * @param sectionId identificativo sezione
24       * @return entit√† follow inserita nel database
25       */
26      @AuthenticationRequired
27      public Follow follow(@SectionExistsById int sectionId){
28          User user = genericRepository.findById(User.class, currentUser.getId());
29          Section section = genericRepository.findById(Section.class,sectionId);
30          return genericRepository.insert(new Follow(user,section));
31      }
32  
33      /**
34       * Permette di togliere il follow ad una sezione
35       * @param sectionId identificativo sezione
36       */
37      @AuthenticationRequired
38      public void unFollow(@SectionExistsById int sectionId){
39          User user = genericRepository.findById(User.class, currentUser.getId());
40          Section section = genericRepository.findById(Section.class,sectionId);
41          Follow follow = section.getFollow(user);
42  
43          if(follow != null)
44              genericRepository.remove(follow);
45      }
