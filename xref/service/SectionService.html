33  @ApplicationScoped
34  @Transactional
35  public class SectionService {
36      private SectionRepository sectionRepo;
37      private GenericRepository genericRepository;
38      private BinaryContentRepository bcRepo;
39      private CurrentUser currentUser;
40  
41      protected SectionService(){}
42  
43      @Inject
44      protected SectionService(GenericRepository genericRepository, SectionRepository sectionRepository,
45                               BinaryContentRepository bcRepo, CurrentUser currentUser){
46          this.genericRepository = genericRepository;
47          this.sectionRepo = sectionRepository;
48          this.bcRepo = bcRepo;
49          this.currentUser = currentUser;
50      }
51  
52      /**
53       * Converte section in SectionPage.
54       * @param section sezione da convertire
55       * @return SectionPage con i dati di section
56       */
57      private SectionPage map(Section section){
58          User user = null;
59          if(currentUser.isLoggedIn()){
60              user = genericRepository.findById(User.class,currentUser.getId());
61          }
62          return SectionPage.builder().id(section.getId())
63                  .name(section.getName())
64                  .picture(section.getPicture())
65                  .description(section.getDescription())
66                  .banner(section.getBanner())
67                  .nFollowers(section.getFollowCount())
68                  .isFollowed(user != null && section.getFollow(user) != null).build();
69      }
70  
71      /**
72       * Cancella una sezione dato il suo id
73       * @param id l'id di una sezione esistente
74       */
75      @AdminsOnly
76      public void delete(@SectionExists int id){
77          genericRepository.remove(genericRepository.findById(Section.class, id));
78      }
79  
80      /**
81       * Crea una nuova sezione e ne restituisce l'id
82       * @param name Nome della sezione
83       * @param description descrizione della sezione
84       * @param picture stream relativo alla foto della sezione
85       * @param banner stream relativo al banner della sezione
86       * @return id della sezione creata
87       */
88      @AdminsOnly
89      public int newSection(@NotBlank @Size(max=50) @UniqueSection String name,
90                            @Size(max=255) String description,
91                            @Image BufferedInputStream picture,
92                            @Image BufferedInputStream banner) {
93          Section s = new Section();
94          s.setName(name);
95          s.setDescription(description);
96          try {
97              if (picture != null)
98                  s.setPicture(bcRepo.insert(picture));
99              if (banner != null)
100                 s.setBanner(bcRepo.insert(banner));
101         } catch (ReadLimitExceededException e) {
102             throw new IllegalArgumentException("Il file non deve superare i 5MB");
103         } catch (IOException e) {
104             throw new RuntimeException(e);
105         }
106         return genericRepository.insert(s).getId();
107     }

109     /**
110      * Ritorna una lista di tutte le sezioni esistenti
111      * @return lista di sezioni
112      */
113     public List<SectionPage> showSections(){
114         return genericRepository.findAll(Section.class).stream().map(this::map).collect(Collectors.toList());
115     }
116 
117     /**
118      * Ritorna una mappa di tutte le sezioni esistenti
119      * @return mappa la cui chiave è l'id della sezione e il valore un entita PostPage che ne contiene i dati
120      */
121     @Named("sections")
122     @RequestScoped
123     @Produces
124     public Map<Integer,SectionPage> getSectionsMap(){
125         return genericRepository.findAll(Section.class).stream().map(this::map)
126                 .collect(Collectors.toMap(SectionPage::getId, section -> section));
127     }

129     /**
130      * Ritorna un entità sezione dato un certo id
131      * @param id id di una sezione esistente
132      * @return entita SectionPage che contiene i dati della sezione
133      */
134     public SectionPage showSection(@SectionExists int id){
135        Section s =  genericRepository.findById(Section.class, id);
136        return map(s);
137     }

139     /**
140      * Ritorna un entità sezione con un nome specifico
141      * @param sectionName nome di una sezione esistente
142      * @return entita SectionPage che contiene i dati della sezione
143      */
144     public SectionPage getSection(@NotNull @SectionExists String sectionName){
145         Section s = genericRepository.findByNaturalId(Section.class,sectionName);
146         return map(s);
147     }

164     @Produces
