30  @ApplicationScoped
31  @Transactional
32  public class CommentService {
33      private GenericRepository genericRepository;
34      private CommentRepository commentRepo;
35      private CurrentUser currentUser;
36  
37      private static final int MAX_COMMENT_DEPTH = 4;
38  
39      protected CommentService(){}
40  
41      @Inject
42      protected CommentService(GenericRepository genericRepository, CommentRepository commentRepository,
43                                              CurrentUser currentUser){
44          this.genericRepository = genericRepository;
45          this.commentRepo = commentRepository;
46          this.currentUser = currentUser;
47      }
48  
49      /**
50       * Converte Comment in CommentDTO.
51       * @param comment commento da convertire
52       * @return commentDTO con i dati di comment
53       */
54      private CommentDTO map (Comment comment){
55          CommentVote commentVote = null;
56          if(currentUser.isLoggedIn()){
57              User user = genericRepository.findById(User.class, currentUser.getId());
58              commentVote = comment.getVote(user);
59          }
60  
61          return CommentDTO.builder()
62                  .id(comment.getId())
63                  .authorUsername(comment.getAuthor().getUsername())
64                  .authorId(comment.getAuthor().getId())
65                  .creationDate(comment.getCreationDate())
66                  .postId(comment.getPost().getId())
67                  .vote(commentVote == null ? 0 : commentVote.getVote())
68                  .content(comment.getContent())
69                  .votes(comment.getVotesCount() == null ? 0 : comment.getVotesCount())
70                  .creationDate(comment.getCreationDate()  == null ? Instant.now() : comment.getCreationDate())
71                  .parentCommentId(comment.getParentComment() == null ? 0 : comment.getParentComment().getId())
72                  .build();
73      }
74  
75      /**
76       * Ritorna una mappa la cui chiave è l'id del commento padre e il valore una lista di CommentDTO
77       * @param postId l'id di un post esistente di cui si vuole ottenere i commenti
78       * @return mappa con i commenti del post
79       */
80      public Map<Integer,List<CommentDTO>> getPostComments(@PostExists int postId){
81          List<Comment> comments = commentRepo.getByPost(genericRepository.findById(Post.class, postId), MAX_COMMENT_DEPTH);
82          return comments.stream().map(this::map).collect(groupingBy(CommentDTO::getParentCommentId, toList()));
83      }
84  
85      /**
86       * Ritorna una mappa la cui chiave è l'id del commento padre e il valore una lista di CommentDTO
87       * @param commentId l'id di un commento esistente di cui si vuole ottenere le risposte
88       * @return mappa con le risposte al commento
89       */
90      public Map<Integer, List<CommentDTO>> getReplies(@CommentExists int commentId){
91          List<Comment> comments = commentRepo.getReplies(genericRepository.findById(Comment.class, commentId), MAX_COMMENT_DEPTH);
92          return comments.stream().map(this::map).collect(groupingBy(CommentDTO::getParentCommentId, toList()));
93      }
94  
95      /**
96       * Ritorna un commento dato il suo id
97       * @param id l'id di un commento esistente
98       * @return commento avente l'id specificato
99       */
100     public CommentDTO getComment(@CommentExists int id){
101         return map(genericRepository.findById(Comment.class, id));
102     }

105     /**
106      * Cancella un commento dato il suo id
107      * @param id l'id di un commento esistente
108      */
109     @AuthenticationRequired
110     public void delete(@CommentExists int id){
111         Comment comment = genericRepository.findById(Comment.class, id);
112         if(currentUser.getId() != comment.getAuthor().getId() && !currentUser.isAdmin())
113             throw new AuthorizationException();
114         genericRepository.remove(genericRepository.findById(Comment.class, id));
115     }
116 
117     /**
118      * Modifica un commento dato il suo id
119      * @param id l'id di un commento esistente
120      * @param text stringa con testo da sostituire
121      */
122     @AuthenticationRequired
123     public void editComment(@CommentExists int id, @NotBlank @Size(max=65535) String text){
124         Comment comment = genericRepository.findById(Comment.class, id);
125         if(currentUser.getId() != comment.getAuthor().getId() && !currentUser.isAdmin())
126             throw new AuthorizationException();
127         comment.setContent(text);
128     }
129 
130     /**
131      * Crea un nuovo commento e ne restituisce l'id
132      * @param text una stringa non vuota di massimo 65535 caratteri
133      * @param postId id di un post esistente
134      * @return id del commento creato
135      */
136     @AuthenticationRequired
137     @DenyBannedUsers
138     public int newComment(@NotBlank @Size(max=65535) String text,
139                           @PostExists int postId){
140         User user = genericRepository.findById(User.class, currentUser.getId());
141 
142         Comment comment = new Comment();
143         comment.setAuthor(user);
144         comment.setContent(text);
145         comment.setPost(genericRepository.findById(Post.class, postId));
146         return genericRepository.insert(comment).getId();
147     }
148 
149     /**
150      * Crea una risposta a un commento e ne restituisce l'id
151      * @param text una stringa non vuota di massimo 1000 caratteri
152      * @param parentCommentId id di un commento esistente
153      * @return id del commento creato
154      */
155     @AuthenticationRequired
156     public int newCommentReply(@NotBlank @Size(max=65535) String text,
157                                @CommentExists int parentCommentId){
158         User user = genericRepository.findById(User.class, currentUser.getId());
159         Comment parent = genericRepository.findById(Comment.class, parentCommentId);
160 
161         Comment comment = new Comment();
162         comment.setAuthor(user);
163         comment.setContent(text);
164         comment.setPost(parent.getPost());
165         comment.setParentComment(parent);
166         return genericRepository.insert(comment).getId();
167     }
