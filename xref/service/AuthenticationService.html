9   import javax.enterprise.context.RequestScoped;
10  import javax.enterprise.context.SessionScoped;
11  import javax.enterprise.inject.Produces;
12  import javax.inject.Inject;
13  import javax.inject.Named;
14  import javax.transaction.Transactional;
15  import java.io.Serializable;
16  import java.time.Instant;
17  
18  @SessionScoped
19  @Transactional
20  public class AuthenticationService implements Serializable {
21      @Inject UserRepository userRepository;
22      @Inject GenericRepository genericRepository;
23      @Inject Pbkdf2PasswordHashImpl passwordHash;
24  
25      private int currentUserId = 0;
26  
27      /**
28       * Autentica un utente
29       * @param username stringa con nome utente
30       * @param password stringa con password
31       * @return esito dell'operazione
32       */
33      public boolean authenticate(String username, String password){
34          User user = userRepository.getByName(username);
35          if(user == null)
36              return false;
37          if (passwordHash.verify(password, user.getPassword(), user.getSalt())){
38              currentUserId = user.getId();
39              return true;
40          } else {
41              return false;
42          }
43      }
44  
45      /**
46       * Ritorna l'utente in uso e ne rende accessibile i dati nelle jsp
47       * @return utente in uso
48       */
49      @Named("currentUser") //accessibile nelle jsp con ${currentUser}
50      @RequestScoped
51      @Produces
52      public CurrentUser getCurrentUser(){
53          if(currentUserId <= 0)
54              return new CurrentUser();
55  
56          User user = genericRepository.findById(User.class,currentUserId);
57          if(user == null)
58              return new CurrentUser();
59  
60          Instant longestCurrentBan = user.getBans().isEmpty() ? null : user.getBans().get(0).getEndTime();
61  
62          CurrentUser currentUser = CurrentUser.builder()
63                  .username(user.getUsername())
64                  .id(user.getId())
65                  .isAdmin(user.getAdmin())
66                  .picture(user.getPicture())
67                  .banDuration(longestCurrentBan)
68                  .isLoggedIn(true)
69                  .build();
70  
71          System.out.println(currentUser);
72          return currentUser;
73      }
74  
75  }
