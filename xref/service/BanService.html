11  import javax.enterprise.context.ApplicationScoped;
12  import javax.inject.Inject;
13  import javax.transaction.Transactional;
14  import javax.validation.constraints.Future;
15  import java.time.Instant;
16  import java.util.ArrayList;
17  import java.util.List;
18  
19  @ApplicationScoped
20  @Transactional
21  public class BanService {
22      @Inject private GenericRepository genericRepository;
23  
24      @AdminsOnly
25      public Ban addBan(@Future Instant date, @UserExists int userId) {
26          Ban ban = new Ban();
27          ban.setEndTime(date);
28          User user = new User();
29          user.setId(userId);
30          ban.setUser(user);
31          return genericRepository.insert(ban);
32      }
33  
34      @AdminsOnly
35      public void removeBan(@BanExists int banId){
36          genericRepository.remove(genericRepository.findById(Ban.class, banId));
37      }
38  
39      @AdminsOnly
40      public List<BanDTO> retrieveUserBan(@UserExists int userId){
41          User user = genericRepository.findById(User.class, userId);
42          List<BanDTO> bans = new ArrayList<>();
43          List<Ban> temp = user.getBans();
44  
45          for(Ban b : temp){
46              BanDTO ban = new BanDTO(b.getId(), b.getEndTime(), b.getUser().getId());
47              bans.add(ban);
48          }
49          return bans;
50      }
51  }
