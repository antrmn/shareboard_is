11  import javax.enterprise.context.ApplicationScoped;
12  import javax.inject.Inject;
13  import javax.transaction.Transactional;
14  import javax.validation.constraints.Future;
15  import javax.validation.constraints.NotNull;
16  import java.time.Instant;
17  import java.util.ArrayList;
18  import java.util.List;

20  @ApplicationScoped
21  @Transactional
22  public class BanService {
23      private GenericRepository genericRepository;
24  
25      protected BanService(){}
26  
27      @Inject
28      protected BanService(GenericRepository genericRepository){
29          this.genericRepository = genericRepository;
30      }
31  
32  
33      /**
34       * Aggiunge un ban dato l'id di un utente.
35       * @param date data di fine ban
36       * @param userId identificativo utente
37       * @return entit√† ban aggiunta
38       */
39      @AdminsOnly
40      public int addBan(@NotNull @Future Instant date, @UserExists int userId) {
41          Ban ban = new Ban();
42          ban.setEndTime(date);
43          User user = new User();
44          user.setId(userId);
45          ban.setUser(user);
46          return genericRepository.insert(ban).getId();
47      }
48  
49      /**
50       * Rimuove un ban dato il suo id
51       * @param banId identificativo del ban
52       */
53      @AdminsOnly
54      public void removeBan(@BanExists int banId){
55          genericRepository.remove(genericRepository.findById(Ban.class, banId));
56      }
57  
58      /**
59       * Ritorna la lista dei ban di un utente
60       * @param userId identificativo utente
61       * @return lista dei ban relativi all'utente
62       */
63      @AdminsOnly
64      public List<BanDTO> retrieveUserBan(@UserExists int userId){
65          User user = genericRepository.findById(User.class, userId);
66          List<BanDTO> bans = new ArrayList<>();
67          List<Ban> temp = user.getBans();
68  
69          for(Ban b : temp){
70              BanDTO ban = new BanDTO(b.getId(), b.getEndTime(), b.getUser().getId());
71              bans.add(ban);
72          }
73          return bans;
74      }
75  }
